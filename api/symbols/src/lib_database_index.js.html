 74         /**
 75          * Typecast the value to the given columnType. Calls
 76          * typecastValue{ColumnType} if the method exists,
 77          * otherwise returns the value.
 78          *
 79          * @example
 80          * DB.typeCastValue("boolean", 0) //=> false
 81          * DB.typeCastValue("boolean", 1) //=> true
 82          * DB.typeCastValue("timestamp",  '2004-02-01 12:12:12')
 83          *      //=> new patio.sql.TimeStamp(2004, 1, 1, 12, 12, 12);
 84          *
 85          * @throws {patio.DatabaseError} if there is an error converting the value to the column type.
 86          *
 87          * @param {String} columnType the SQL datatype of the column
 88          * @param value the value to typecast.
 89          *
 90          * @return the typecasted value.
 91          * */
 92         typecastValue:function(columnType, value){
 93             if (comb.isNull(value) || comb.isUndefined(value)) {
 94                 return null;
 95             }
 96             var meth = "__typecastValue" + columnType.charAt(0).toUpperCase() + columnType.substr(1).toLowerCase();
 97             try {
 98                 if (comb.isFunction(this[meth])) {
 99                     return this[meth](value);
100                 } else {
101                     return value;
102                 }
103             } catch (e) {
104                 throw e;
105             }
106         },

109         // Typecast the value to true, false, or null
110         __typecastValueBoolean:function(value){
111             if (comb.isBoolean(value)) {
112                 return value;
113             } else if (value === 0 || value === "0" || (comb.isString(value) && value.match(/^f(alse)?$/i) != null)) {
114                 return false;
115             } else {
116                 return comb.isEmpty(value) ? null : true;
117             }

120         // Typecast the value to a Date
121         __typecastValueDate:function(value){
122             if (comb.isDate(value)) {
123                 return value;
124             } else if (comb.isString(value)) {
125                 var ret = patio.stringToDate(value);
126                 if (ret == null) {
127                     throw new DatabaseError(format("Invalid value for date %j", [value]));
128                 }
129                 return ret;
130             } else if (comb.isHash(value) && !comb.isEmpty(value)) {
131                 return new Date(value.year, value.month, value.day);
132             } else {
133                 throw new DatabaseError(format("Invalid value for date %j", [value]));
134             }
135         },

137         // Typecast the value to a patio.sql.DateTime.
138         __typecastValueDatetime:function(value){
139             var ret;
140             if (comb.isInstanceOf(value, sql.DateTime)) {
141                 return value;
142             } else if (comb.isDate(value)) {
143                 ret = value;
144             } else if (comb.isHash(value) && !comb.isEmpty(value)) {
145                 ret = new Date(value.year, value.month, value.day, value.hour, value.minute, value.second);
146             } else if (comb.isString(value)) {
147                 ret = patio.stringToDateTime(value);
148                 if (ret == null) {
149                     throw new DatabaseError(format("Invalid value for datetime %j", [value]));
150                 }
151                 ret = ret.date;
152             } else {
153                 throw new DatabaseError(format("Invalid value for datetime %j", [value]));
154             }
