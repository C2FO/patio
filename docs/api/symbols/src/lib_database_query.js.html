217         /**
218          * Parse the schema from the database.
219          *
220          * @example
221          *
222          *   DB.schema("artists").then(function(schema){
223          *     //example schema
224          *     {
225          *      id :  {
226          *          type : "integer",
227          *          primaryKey : true,
228          *          "default" : "nextval('artist_id_seq'::regclass)",
229          *          jsDefault : null,
230          *          dbType : "integer",
231          *          allowNull : false
232          *       },
233          *      name : {
234          *          type : "string",
235          *          primaryKey : false,
236          *          "default" : null,
237          *          jsDefault  : null,
238          *          dbType : "text",
239          *          allowNull : false
240          *       }
241          *     }
242          *   })
243          *
244          * @param {String|patio.sql.Identifier|patio.sql.QualifiedIdentifier} table the table to get the schema for.
245          * @param {Object} [opts=null] Additinal options.
246          * @param {boolean} [opts.reload=false] Set to true to ignore any cached results.
247          * @param {String|patio.sql.Identifier} [opts.schema] An explicit schema to use.  It may also be implicitly provided
248          *            via the table name.
249          *
250          * @return {Promise} Returns a Promise that is resolved with the schema for the given table as an object
251          * where the key is the column name and the value is and object containg column information. The default
252          * column information returned.
253          * <ul>
254          *     <li>allowNull : Whether NULL is an allowed value for the column.</li>
255          *     <li>dbType : The database type for the column, as a database specific string.</li>
256          *     <li>"default" : The database default for the column, as a database specific string.</li>
257          *     <li>primaryKey : Whether the columns is a primary key column.  If this column is not present,
258          *                 it means that primary key information is unavailable, not that the column
259          *                 is not a primary key.</li>
260          *     <li>jsDefault : The database default for the column, as a javascript object.  In many cases, complex
261          *                  database defaults cannot be parsed into javascript objects.</li>
262          *     <li>type : A string specifying the type, such as "integer" or "string".</li>
263          * <ul>
264          *
265          */
266         schema:function (table, opts) {
267             if (!comb.isFunction(this.schemaParseTable)) {
268                 throw new Error("Schema parsing is not implemented on this database");
269             }
270             var ret = new Promise();
271             opts = opts || {};
272             var schemaParts = this.__schemaAndTable(table);
273             var sch = schemaParts[0], tableName = schemaParts[1];
274             var quotedName = this.__quoteSchemaTable(table);
275             opts = sch && !opts.schema ? comb.merge({schema:sch}, opts) : opts;
276             opts.reload && delete this.schemas[quotedName];
277             if (this.schemas[quotedName]) {
278                 ret.callback(this.schemas[quotedName]);
279                 return ret;
280             } else {
281                 this.schemaParseTable(tableName, opts).then(hitch(this, function (cols) {
282                     var schema = {};
283                     if (!cols || cols.length == 0) {
284                         ret.errback("Error parsing schema, no columns returns, table probably doesnt exist");
285                     } else {
286                         for (var i in cols) {
287                             var c = cols[i];
288                             var name = c[0], c = c[1];
289                             c.jsDefault = this.__columnSchemaToJsDefault(c["default"], c.type);
290                             schema[name] = c;
291                         }
292                         this.schemas[quotedName] = schema;
293                         ret.callback(schema);
294                     }
295                 }), hitch(ret, "errback"));
296             }
297             return ret;
298         },

356         /**
357          * Starts a database transaction.  When a database transaction is used,
358          * either all statements are successful or none of the statements are
359          * successful.
360          * <p>
361          *      <b>Note</b> that MySQL MyISAM tables do not support transactions.</p>
362          * </p>
363          *
364          * @example
365          * //normal transaction
366          * DB.transaction(function() {
367          *       this.execute('DROP TABLE test;');
368          *       this.execute('DROP TABLE test2;');
369          *  });
370          *
371          * //transaction with a save point.
372          * DB.transaction(function() {
373          *      this.transaction({savepoint : true}, function() {
374          *          this.execute('DROP TABLE test;');
375          *          this.execute('DROP TABLE test2;');
376          *      });
377          *});
378          *
379          * //WITH ISOLATION LEVELS
380          *
381          * db.supportsTransactionIsolationLevels = true;
382          * //BEGIN
383          * //SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED
384          * //DROP TABLE test1'
385          * //COMMIT
386          * DB.transaction({isolation:"uncommited"}, function(d) {
387          *      d.run("DROP TABLE test1");
388          * });
389          *
390          * //BEGIN
391          * //SET TRANSACTION ISOLATION LEVEL READ COMMITTED
392          * //DROP TABLE test1
393          * //COMMIT
394          *  DB.transaction({isolation:"committed"}, function(d) {
395          *      d.run("DROP TABLE test1");
396          *  });
397          *
398          * //BEGIN
399          * //SET TRANSACTION ISOLATION LEVEL REPEATABLE READ'
400          * //DROP TABLE test1
401          * //COMMIT
402          *  DB.transaction({isolation:"repeatable"}, function(d) {
403          *      d.run("DROP TABLE test1");
404          *  });
405          *
406          * //BEGIN
407          * //SET TRANSACTION ISOLATION LEVEL SERIALIZABLE
408          * //DROP TABLE test1
409          * //COMMIT
410          *  DB.transaction({isolation:"serializable"}, function(d) {
411          *      d.run("DROP TABLE test1");
412          *  });
413          *
414          * //With an Error
415          * //BEGIN
416          * //DROP TABLE test
417          * //ROLLBACK
418          * DB.transaction(function(d) {
419          *      d.execute('DROP TABLE test');
420          *      throw "Error";
421          * });
422          *
423          * @param {Object} [opts={}] options to use when performing the transaction.
424          * @param {String} [opts.isolation] The transaction isolation level to use for this transaction,
425          *               should be "uncommitted", "committed", "repeatable", or "serializable",
426          *               used if given and the database/adapter supports customizable
427          *               transaction isolation levels.
428          * @param {String} [opts.prepare] A string to use as the transaction identifier for a
429          *             prepared transaction (two-phase commit), if the database/adapter
430          *             supports prepared transactions.
431          * @param {Boolean} [opts.savepoint] Whether to create a new savepoint for this transaction,
432          *               only respected if the database/adapter supports savepoints.  By
433          *               default patio will reuse an existing transaction, so if you want to
434          *               use a savepoint you must use this option.
435          * @param {Function} cb a function used to perform the transaction. This function is
436          * called in the scope of the database by default so one can use this. The funciton is also
437          * called with the database as the first argument. If you have complex operations within your
438          * transactions you will want to return a promise to ensure proper transaction scope.
439          * <pre class="code">
440          * var ds = db.from("user");
441          * db.transaction(function(){
442          *      var ret = new comb.Promise();
443          *      ds.insert({
444          *          firstName:"Jane",
445          *          lastName:"Gorgenson",
446          *          password:"password",
447          *          dateOfBirth:new Date(1956, 1, 3)
448          *      }).then(function(){
449          *          ds.forEach(function(user){
450          *                  return ds.where({id:user.id}).update({firstName:user.firstName + 1});
451          *          }).then(comb.hitch(ret, "callback"), comb.hitch(ret, "errback"));
452          *      }, comb.hitch(ret, "errback"));
453          *     return ret;
454          * });
455          * </pre>
456          *
457          * @return {Promise} a promise that is resolved once the transaction is complete.
458          **/
459         transaction:function (opts, cb) {
460             if (comb.isFunction(opts)) {
461                 cb = opts;
462                 opts = {};
463             } else {
464                 opts = opts || {};
465             }
466             if (!this.__alreadyInTransaction) {
467                 this.__alreadyInTransaction = true;
468                 return this.__transaction(null, opts, cb);
469             } else {
470                 var ret =  new comb.Promise();
471                 var transaction = comb.hitch(this, function(){
472                     this.__transaction(null, opts, cb).then(hitch(ret, "callback"), hitch(ret, "errback"));
473                 });
474                 this.__transactionQueue.enqueue(transaction);
475                 return ret;
476             }
477         },

479         __transactionProxy:function (cb, conn) {
480             var promises = [];
481             var repl = [];
482             //set up our proxy methos
483             ["transaction", "execute"].forEach(function (n) {
484                 var orig = this[n];
485                 repl.push({name:n, orig:orig});
486                 this[n] = function (arg1, arg2) {
487                     try {
488                         var ret;
489                         if (n == "transaction") {
490                             //if its a transaction with no options then we just create a promise from what ever is returned
491                             if (comb.isFunction(arg1)) {
492                                 ret = comb.when(arg1.apply(this, [this]));
493                             } else {
494                                 if (this.supportsSavepoints && !comb.isUndefinedOrNull(arg1.savepoint)) {
495                                     //if we support save points there is a save point option then we
496                                     //use __transaction again with the previous connection
497                                     ret = this.__transaction(conn, arg1, arg2);
498                                 } else {
499                                     //other wise use the function passed in to get the returned promise
500                                     ret = comb.when(arg2.apply(this, [this]));
501                                 }
502                             }
503                         } else {
504                             ret = orig.apply(this, comb.argsToArray(arguments).concat(conn));
505 
506                         }
507                     } catch (e) {
508                         ret = new Promise().errback(e);
509                     }
510                     promises.push(comb.when(ret));
511                     return ret;
512                 }
513             }, this);
514             try {
515                 var cbRet = cb.apply(this, [this]);
516                 if (cbRet && comb.isInstanceOf(cbRet, Promise)) {
517                     promises.push(cbRet);
518                 }
519             } catch (e) {

521             }
522             if (promises.length == 0) {
523                 promises.push(new Promise().callback());
524             }
525             return new PromiseList(promises).both(hitch(this, function () {
526                 repl.forEach(function (o) {

585                     if (comb.isArray(e) && e.length == 2) {

614                     }

789          * Match the database's column type to a javascript type via a
790          * regular expression, and return the javascript type as a string
791          * such as "integer" or "string".
792          * @private
793          */
794         schemaColumnType:function (dbType) {
795             var ret = null, m;
796             if (dbType.match(/^interval$/i)) {
797                 ret = "interval";
798             } else if (dbType.match(/^(character( varying)?|n?(var)?char|n?text)/i)) {
799                 ret = "string";
800             } else if (dbType.match(/^int(eger)?|(big|small|tiny)int/i)) {
801                 ret = "integer";
802             } else if (dbType.match(/^date$/i)) {
803                 ret = "date";
804             } else if (dbType.match(/^year/i)) {
805                 ret = "year";
806             } else if (dbType.match(/^((small)?datetime|timestamp( with(out)? time zone)?)$/i)) {
807                 ret = "datetime";
808             } else if (dbType.match(/^time( with(out)? timezone)?$/i)) {
809                 ret = "time";
810             } else if (dbType.match(/^(bit|boolean)$/i)) {
811                 ret = "boolean";
812             } else if (dbType.match(/^(real|float|double( precision)?)$/i)) {
813                 ret = "float";
814             } else if ((m = dbType.match(/^(?:(?:(?:num(?:ber|eric)?|decimal|double)(?:\(\d+,\s*(\d+)\))?)|(?:small)?money)/i))) {
815                 ret = m[1] && m[1] == '0' ? "integer" : "decimal";
816             } else if (dbType.match(/^bytea|blob|image|(var)?binary/i)) {
817                 ret = "blob";
818             } else if (dbType.match(/^enum/i)) {
819                 ret = "enum";
820             } else if (dbType.match(/^set/i)) {
821                 ret = "set";
822             }
823             return ret;
