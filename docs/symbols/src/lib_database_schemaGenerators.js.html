 27 var Generator = comb.define(null, {
 28     instance : {

 30         __primaryKey : null,

 32         constructor : function(db, block) {
 33             this.db = db;
 34             this.columns = [];
 35             this.indexes = [];
 36             this.constraints = [];
 37             this.__primaryKey = null;
 38         },
 39 
 40         /*Add an unnamed constraint to the DDL, specified by the given block
 41          * or args:
 42          *
 43          *   check(:num=>1..5) # CHECK num >= 1 AND num <= 5
 44          *   check{num > 5} # CHECK num > 5
 45          *   */
 46         check : function() {
 47             return this.constraint.apply(this, [null].concat(comb.argsToArray(arguments)));
 48         },
 49 
 50         /* Add a column with the given name, type, and opts to the DDL.
 51          *
 52          *   column :num, :integer
 53          *   # num INTEGER
 54          *
 55          *   column :name, String, :null=>false, :default=>'a'
 56          *   # name varchar(255) NOT NULL DEFAULT 'a'
 57          *
 58          *   inet :ip
 59          *   # ip inet
 60          *
 61          * You can also create columns via method missing, so the following are
 62          * equivalent:
 63          *
 64          *   column :number, :integer
 65          *   integer :number
 66          *
 67          * The following options are supported:
 68          *
 69          * :default :: The default value for the column.
 70          * :deferrable :: This ensure Referential Integrity will work even if
 71          *                reference table will use for its foreign key a value that does not
 72          *                exists(yet) on referenced table. Basically it adds
 73          *                DEFERRABLE INITIALLY DEFERRED on key creation.
 74          * :index :: Create an index on this column.
 75          * :key :: For foreign key columns, the column in the associated table
 76          *         that this column references.  Unnecessary if this column
 77          *         references the primary key of the associated table.
 78          * :null :: Mark the column as allowing NULL values (if true),
 79          *          or not allowing NULL values (if false).  If unspecified, will default
 80          *          to whatever the database default is.
 81          * :on_delete :: Specify the behavior of this column when being deleted
 82          *               (:restrict, cascade, :set_null, :set_default, :no_action).
 83          * :on_update :: Specify the behavior of this column when being updated
 84          *               (:restrict, cascade, :set_null, :set_default, :no_action).
 85          * :primary_key :: Make the column as a single primary key column.  This should only
 86          *                 be used if you have a single, nonautoincrementing primary key column.
 87          * :size :: The size of the column, generally used with string
 88          *          columns to specify the maximum number of characters the column will hold.
 89          *          An array of two integers can be provided to set the size and the
 90          *          precision, respectively, of decimal columns.
 91          * :unique :: Mark the column as unique, generally has the same effect as
 92          *            creating a unique index on the column.
 93          * :unsigned :: Make the column type unsigned, only useful for integer
 94          *              columns.
 95          *              */
 96         column : function(name, type, opts) {
 97             opts = opts || {};
 98             this.columns.push(comb.merge({name : name, type : type}, opts));
 99             if (opts.index) {
100                 this.index(name);
101             }
102         },
103 
104         /* Adds a named constraint (or unnamed if name is nil) to the DDL,
105          * with the given block or args.
106          *
107          *   constraint(:blah, :num=>1..5) # CONSTRAINT blah CHECK num >= 1 AND num <= 5
108          *   check(:foo){num > 5} # CONSTRAINT foo CHECK num > 5
109          *   */
110         constraint : function(name) {
111             var args = comb.argsToArray(arguments);
112             var block = comb.isFunction(args[args.length - 1]) ? args.pop : null;
113             this.constraints.push({name : name, type : "check", check : block || args});
114         },
115 
116         /* for available options.
117          *
118          *   foreign_key(:artist_id) # artist_id INTEGER
119          *   foreign_key(:artist_id, :artists) # artist_id INTEGER REFERENCES artists
120          *   foreign_key(:artist_id, :artists, :key=>:id) # artist_id INTEGER REFERENCES artists(id)
121          *   */
122         foreignKey : function(name, table, opts) {
123             opts = opts || {};
124             opts = comb.isHash(table) ? comb.merge({}, table, opts) : comb.isString(table) ? comb.merge({table : table}, opts) : opts;
125             if (comb.isArray(name)) {
126                 return this.__compositeForeignKey(name, opts);
127             } else {
128                 return this.column(name, "integer", opts);
129             }
130         },
131 
132         /**
133          *Add a full text index on the given columns to the DDL.
134          */
135 
136         fullTextIndex : function(columns, opts) {
137             opts = opts || {};
138             return this.index(columns, comb.merge({type : "fullText"}, opts));
139         },
140 
141         /**
142          * True if the DDL includes the creation of a column with the given name.
143          */
144         hasColumn : function(name) {
145             return this.columns.some(function(c) {
146                 return c.name == name
147             });
148         },
149 
150         /** Add an index on the given column(s) with the given options to the DDL.
151          * The available options are:
152          *
153          * :type :: The type of index to use (only supported by some databases)
154          * :unique :: Make the index unique, so duplicate values are not allowed.
155          * :where :: Create a partial index (only supported by some databases)
156          *
157          *   index :name
158          *   # CREATE INDEX table_name_index ON table (name)
159          *
160          *   index [:artist_id, :name]
161          *   # CREATE INDEX table_artist_id_name_index ON table (artist_id, name)
162          *   */
163         index : function(columns, opts) {
164             this.indexes.push(comb.merge({columns : comb.array.toArray(columns)}, opts));
165         },
166 
167         /* Adds an autoincrementing primary key column or a primary key constraint to the DDL.
168          * To create a constraint, the first argument should be an array of column symbols
169          * specifying the primary key columns. To create an autoincrementing primary key
170          * column, a single symbol can be used. In both cases, an options hash can be used
171          * as the second argument.
172          *
173          * If you want to create a primary key column that is not autoincrementing, you
174          * should not use this method.  Instead, you should use the regular +column+ method
175          * with a <tt>:primary_key=>true</tt> option.
176          *
177          * Examples:
178          *   primary_key(:id)
179          *   primary_key([:street_number, :house_number])
180          * */
181         primaryKey : function(name) {
182             if (comb.isArray(name)) {
183                 return this.__compositePrimaryKey.apply(this, arguments);
184             } else {
185                 var args = comb.argsToArray(arguments).slice(1);
186                 var opts = args.pop();
187                 this.__primaryKey = comb.merge(opts, this.db.serialPrimaryKeyOptions, {name : name});
188                 opts = comb.isHash(opts) ? {type : opts} : opts;
189                 if ((type = args.pop()) != null) {
190                     comb.merge(opts, {type : type});
191                 }
192                 comb.merge(this.__primaryKey, opts);
193                 return this.__primaryKey;
194             }
195         },
196 
197         /*
198          Add a spatial index on the given columns to the DDL.
199          */
200         spatialIndex : function(columns, opts) {
201             opts = opts || {};
202             return this.index(columns, comb.merge({type : "spatial"}, opts));
203         },
204 
205         /* Add a unique constraint on the given columns to the DDL.
206          *
207          *   unique(:name) # UNIQUE (name)
208          *   */
209         unique : function(columns, opts) {
210             opts = opts || {};
211             this.constraints.push(comb.merge({type : "unique", columns : comb.array.toArray(columns)}, opts));
212         },
213 
214 
215 
216         /*
217          Add a composite primary key constraint
218          */
219         __compositePrimaryKey : function(columns) {
220             var args = comb.argsToArray(arguments).slice(1);
221             var opts = args.pop() || {};
222             this.constraints.push(comb.merge({type : "primaryKey", columns : columns}, opts));
223         },
224 
225         /*
226          Add a composite foreign key constraint
227          */
228         __compositeForeignKey : function(columns, opts) {
229             this.constraints.push(comb.merge({type : "foreignKey", columns : columns}, opts));
230         },
231 
232         getters : {
233             // The name of the primary key for this generator, if it has a primary key.
234             primaryKeyName : function() {
235                 return this.__primaryKey ? this.__primaryKey.name : null;
236             }
237 
238         }
239 
240     },
241     static : {
242         /*Add a method for each of the given types that creates a column
243          * with that type as a constant.  Types given should either already
244          * be constants/classes or a capitalized string/symbol with the same name
245          * as a constant/class.
246          * */
247         addTypeMethod : function(types) {
248             if (comb.isArray(types)) {
249                 types.forEach(function(type) {
250                     this.prototype[type] = function(name, opts) {
251                         opts = opts || {};
252                         return this.column(name, type, opts);
253                     }
254                 }, this);
255             } else {
256                 this.addTypeMethod(comb.argsToArray(arguments));
257             }
258         }
259     }

265 exports.SchemaGenerator = function(ds, block) {
266     var gen = new Generator(ds, block);
267     var prox = comb.methodMissing(gen, function(name) {
268         return function(type, opts) {
269             name == name || null,opts = opts || {};
270             if (name) {
271                 return this.column(name, type, opts);
272             } else {
273                 throw new TypeError();
274             }
275         }
276     }, Generator);
277     block.apply(prox, [prox]);
278     gen.columns = prox.columns;
279     if (gen.__primaryKey && !gen.hasColumn(gen.primaryKeyName)) {
280         gen.columns.unshift(gen.__primaryKey);
281     }
282     return gen;
283 }

290 var AlterTableGenerator = comb.define(null, {
291     instance : {
292 
293         constructor : function(db, block) {
294             this.db = db;
295             this.operations = [];
296             block.apply(this, [this]);
297 
298         },
299 
300         /* Add a column with the given name, type, and opts to the DDL for the table.
301          * See Generator#column for the available options.
302          *
303          *   add_column(:name, String) # ADD COLUMN name varchar(255)
304          *   */
305         addColumn : function(name, type, opts) {
306             opts = opts || {};
307             this.operations.push(comb.merge({op : "addColumn", name : name, type : type}, opts));
308         },
309 
310         /* Add a constraint with the given name and args to the DDL for the table.
311          * See Generator#constraint.
312          *
313          *   add_constraint(:valid_name, :name.like('A%'))
314          *   # ADD CONSTRAINT valid_name CHECK (name LIKE 'A%')
315          *   */
316         addConstraint : function(name) {
317             var args = comb.argsToArray(arguments);
318             var block = comb.isFunction(args[args.length - 1]) ? args.pop() : null;
319             this.operations.push({op : "addConstraint", name : name, type : "check", check : block || args});
320         },
321 
322         /** Add a unique constraint to the given column(s)
323          *
324          *   add_unique_constraint(:name) # ADD UNIQUE (name)
325          *   add_unique_constraint(:name, :name=>:unique_name) # ADD CONSTRAINT unique_name UNIQUE (name)
326          *   */
327         addUniqueConstraint : function(columns, opts) {
328             opts = opts || {};
329             this.operations.push(comb.merge({op : "addConstraint", type : "unique", columns : comb.array.toArray(columns)}, opts));
330         },
331 
332         /*
333          * Add a foreign key with the given name and referencing the given table
334          * to the DDL for the table.  See Generator#column for the available options.
335          *
336          * You can also pass an array of column names for creating composite foreign
337          * keys. In this case, it will assume the columns exists and will only add
338          * the constraint.
339          *
340          * NOTE: If you need to add a foreign key constraint to a single existing column
341          * use the composite key syntax even if it is only one column.
342          *
343          *   add_foreign_key(:artist_id, :table) # ADD COLUMN artist_id integer REFERENCES table
344          *   add_foreign_key([:name], :table) # ADD FOREIGN KEY (name) REFERENCES table
345          */
346         addForeignKey : function(name, table, opts) {
347             opts = opts;
348             if (comb.isArray(name)) {
349                 return this.__addCompositeForeignKey(name, table, opts);
350             } else {
351                 return this.addColumn(name, "integer", comb.merge({table : table}, opts));
352             }
353         },
354         /*
355          * Add a full text index on the given columns to the DDL for the table.
356          * See Generator#index for available options.
357          */
358         addFullTextIndex : function(columns, opts) {
359             opts = opts || {};
360             return this.addIndex(columns, comb.merge({type : "fullText"}, opts));
361         },
362 
363         /*
364          * Add an index on the given columns to the DDL for the table.  See
365          * Generator#index for available options.
366          *
367          *   add_index(:artist_id) # CREATE INDEX table_artist_id_index ON table (artist_id)
368          */
369         addIndex : function(columns, opts) {
370             opts = opts || {};
371             this.operations.push(comb.merge({op : "addIndex", columns : comb.array.toArray(columns)}, opts));
372         },
373 
374         /*
375          * Add a primary key to the DDL for the table.  See Generator#column
376          * for the available options.  Like +add_foreign_key+, if you specify
377          * the column name as an array, it just creates a constraint:
378          *
379          *   add_primary_key(:id) # ADD COLUMN id serial PRIMARY KEY
380          *   add_primary_key([:artist_id, :name]) # ADD PRIMARY KEY (artist_id, name)
381          */
382         addPrimaryKey : function(name, opts) {
383             opts = opts || {};
384             if (comb.isArray(name)) {
385                 return this.__addCompositePrimaryKey(name, opts);
386             } else {
387                 opts = comb.merge({}, this.db.serialPrimaryKeyOptions, opts);
388                 delete opts.type;
389                 return this.addColumn(name, opts, opts);
390             }
391         },
392 
393         /*
394          * Add a spatial index on the given columns to the DDL for the table.
395          * See Generator#index for available options.
396          * */
397         addSpatialIndex : function(columns, opts) {
398             opts = opts || {};
399             addIndex(columns, comb.merge({}, {type : "spatial"}, opts))
400         },
401         /*
402          * Remove a column from the DDL for the table.
403          *
404          *   drop_column(:artist_id) # DROP COLUMN artist_id
405          */
406         dropColumn : function(name) {
407             this.operations.push({op : "dropColumn", name : name});
408         },
409 
410         /*
411          * Remove a constraint from the DDL for the table.
412          *
413          *   drop_constraint(:unique_name) # DROP CONSTRAINT unique_name
414          */
415         dropConstraint : function(name) {
416             this.operations.push({op : "dropConstraint", name : name});
417         },
418 
419         /*
420          * Remove an index from the DDL for the table.
421          *
422          *   drop_index(:artist_id) # DROP INDEX table_artist_id_index
423          *   drop_index([:a, :b]) # DROP INDEX table_a_b_index
424          *   drop_index([:a, :b], :name=>:foo) # DROP INDEX foo
425          */
426         dropIndex : function(columns, opts) {
427             opts = opts || {};
428             this.operations.push(comb.merge({op : "dropIndex", columns : comb.array.toArray(columns)}, opts));
429         },
430 
431         /*
432          * Modify a column's name in the DDL for the table.
433          *
434          *   rename_column(:name, :artist_name) # RENAME COLUMN name TO artist_name
435          */
436         renameColumn : function(name, newName, opts) {
437             opts = opts || {};
438             this.operations.push(comb.merge({op : "renameColumn", name : name, newName : newName}, opts));
439         },
440 
441         /*
442          * Modify a column's default value in the DDL for the table.
443          *
444          *   set_column_default(:artist_name, 'a') # ALTER COLUMN artist_name SET DEFAULT 'a'
445          *   */
446         setColumnDefault : function(name, def) {
447             this.operations.push({op : "setColumnDefault", name : name, "default" : def});
448         },
449 
450         /*
451          * Modify a column's type in the DDL for the table.
452          *
453          *   set_column_type(:artist_name, 'char(10)') # ALTER COLUMN artist_name TYPE char(10)
454          */
455         setColumnType : function(name, type, opts) {
456             opts = opts || {}
457             this.operations.push(comb.merge({op : "setColumnType", name : name, type : type}, opts));
458         },
459 
460         /*
461          * Modify a column's NOT NULL constraint.
462          *
463          *   set_column_allow_null(:artist_name, false) # ALTER COLUMN artist_name SET NOT NULL
464          *   */
465         setAllowNull : function(name, allowNull) {
466             this.operations.push({op : "setColumnNull", name : name, "null" : allowNull});
467         },
468         /*
469          * Add a composite primary key constraint
470          * */
471         __addCompositePrimaryKey : function(columns, opts) {
472             this.operations.push(comb.merge({op : "addConstraint", type : "primaryKey", columns : columns}, opts));
473         },
474 
475         /*
476          * Add a composite foreign key constraint
477          * */
478         __addCompositeForeignKey : function(columns, table, opts) {
479             this.operations.push(comb.merge({op : "addConstraint", type : "foreignKey", columns : columns, table : table}, opts));
480         }
481     }
