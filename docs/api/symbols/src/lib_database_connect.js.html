 79     /**
 80      * Creates a connection to a Database see {@link patio#createConnection}.
 81      */
 82     connect:function(connectionString, opts) {
 83       opts = opts || {};
 84       if (comb.isString(connectionString)) {
 85         var url = URL.parse(connectionString, true);
 86         if (url.auth) {
 87           var parts = url.auth.split(":");
 88           !opts.user && (opts.user = parts[0]);
 89           !opts.password && (opts.password = parts[1]);
 90         }
 91         opts.type = url.protocol.replace(":", "");
 92         opts.host = url.hostname;
 93         if (url.pathname) {
 94           var path = url.pathname;
 95           var pathParts = path.split("/").slice(1);
 96           if (pathParts.length >= 1) {
 97             opts.database = pathParts[0];
 98           }
 99         }
100         opts = comb.merge(opts, url.query, {uri:connectionString});
101       } else {
102         opts = comb.merge({}, connectionString, opts);
103       }
104       if (opts && comb.isHash(opts) && (opts.adapter || opts.type)) {
105 
106         var type = (opts.type = opts.adapter || opts.type);
107         var Adapter = ADAPTERS[type];
108         if (Adapter) {
109           var adapter = new Adapter(opts);
110           this.DATABASES.push(adapter);
111           return adapter;
112         } else {
113           throw DatabaseError(type + " adapter was not found")
114         }
115       } else {
116         throw new DatabaseError("Options required when connecting.");
117       }
118     },
119 
120     setAdapterType:function(type) {
