 78     /**
 79      * Returns an INSERT SQL query string.  See {@link patio.dataset._Actions#insert}
 80      *
 81      * @example
 82      *
 83      *
 84      * DB.from("items").insertSql({a : 1});
 85      *      //=> INSERT INTO items (a) VALUES (1)
 86      *
 87      * var ds = DB.from("test");
 88      *
 89      * //default values
 90      * ds.insertSql();
 91      *      //=> INSERT INTO test DEFAULT VALUES
 92      *
 93      * //with hash
 94      * ds.insertSql({name:'wxyz', price:342});
 95      *      //=> INSERT INTO test (name, price) VALUES ('wxyz', 342)
 96      * ds.insertSql({});
 97      *      //=> INSERT INTO test DEFAULT VALUES
 98      *
 99      * //object that has a values property
100      * ds.insertSql({values:{a:1}});
101      *      //=> INSERT INTO test (a) VALUES (1)
102      *
103      * //arbitrary value
104      * ds.insertSql(123);
105      *      //=> INSERT INTO test VALUES (123)
106      *
107      * //with dataset
108      * ds.insertSql(DB.from("something").filter({x:2}));
109      *      //=> INSERT INTO test SELECT * FROM something WHERE (x = 2)
110      *
111      * //with array
112      * ds.insertSql('a', 2, 6.5);
113      *      //=> INSERT INTO test VALUES ('a', 2, 6.5)
114      *
115      * @throws {patio.QueryError} if there are Different number of values and columns given to insertSql or
116      *                                  if an invalid BooleanExpresion is given.
117      *
118      * @param {patio.Dataset|patio.sql.LiteralString|Array|Object|patio.sql.BooleanExpression|...} values  values to
119      *      insert into the database. The INSERT statement generated depends on the type.
120      *      <ul>
121      *          <li>Empty object| Or no arugments: then DEFAULT VALUES is used.</li>
122      *          <li>Object: the keys will be used as the columns, and values will be the values inserted.</li>
123      *          <li>Single {@link patio.Dataset} : an insert with subselect will be performed.</li>
124      *          <li>Array with {@link patio.Dataset} : The array will be used for columns and a subselect will performed with the dataset for the values.</li>
125      *          <li>{@link patio.sql.LiteralString} : the literal value will be used.</li>
126      *          <li>Single Array : the values in the array will be used as the VALUES clause.</li>
127      *          <li>Two Arrays: the first array is the columns the second array is the values.</li>
128      *          <li>{@link patio.sql.BooleanExpression} : the expression will be used as the values.
129      *          <li>An arbitrary number of arguments : the {@link patio.Dataset#literal} version of the values will be used</li>
130      *      </ul>
131      *
132      *
133      * @return {String} a INSERT SQL query string
134      */
135     insertSql:function(values) {
136       values = comb.argsToArray(arguments);
137       var opts = this.__opts;
138       if (opts.sql) {
139         return this._staticSql(opts.sql);
140       }
141       this.__checkModificationAllowed();
142 
143       var columns = [];
144 
145       switch (values.length) {
146         case 0 :
147           //we have no values
148           return this.insertSql({});
149         case 1 :
150           var vals = values[0], v;
151           if (comb.isInstanceOf(vals, Dataset, LiteralString) || comb.isArray(vals)) {
152             values = vals;
153           } else if (vals.hasOwnProperty("values") && comb.isObject((v = vals.values))) {
154             return this.insertSql(v);
155           } else if (comb.isHash(vals)) {
156             vals = comb.merge({}, opts.defaults || {}, vals);
157             vals = comb.merge({}, vals, opts.overrides || {});
158             values = [];
159             for (var i in vals) {
160               columns.push(i);
161               values.push(vals[i]);
162             }
163           } else if (comb.isInstanceOf(vals, BooleanExpression)) {
164             var op = vals.op;
165             values = [];
166             if (!comb.isUndefinedOrNull(this._static.TWO_ARITY_OPERATORS[op])) {
167               var args = vals.args;
168               columns.push(args[0]);
169               values.push(args[1]);
170             } else {
171               throw new QueryError("Invalid Expression op: " + op);
172             }
173           }
174           break;
175         case 2 :
176           var v0 = values[0], v1 = values[1]
177           if (comb.isArray(v0) && comb.isArray(v1) || comb.isInstanceOf(v1, Dataset, LiteralString)) {
178             columns = v0, values = v1;
179             if (comb.isArray(values) && columns.length != values.length) {
180               throw new QueryError("Different number of values and columns given to insertSql");
181             }
182           }
183           break;
184       }
185       columns = columns.map(function(k) {
186         return comb.isString(k) ? new Identifier(k) : k;
187       }, this);
188       return this.mergeOptions({columns:columns, values:values})._insertSql();
189     },

244      * Returns a qualified column name (including a table name) if the column
245      * name isn't already qualified.
246      *
247      * @example
248      *
249      * dataset.qualifiedColumnName("b1", "items");
250      *      //=> items.b1
251      *
252      * dataset.qualifiedColumnName("ccc__b"));
253      *      //=> 'ccc.b'
254      *
255      * dataset.qualifiedColumnName("ccc__b", "items"));
256      *      //=> 'ccc.b'
257      *
258      * @param {String} column the column to qualify. If the column is already qualified (e.g. ccc__b) then the
259      *                        table name (e.g. ccc) will override the provided table.
260      *
261      * @param {String} table the name of the table to qualify the column to.
262      *
263      * @return {String} the qualified column name..
264      *
265      */
266     qualifiedColumnName:function(column, table) {
267       if (comb.isString(column)) {
268         var parts = this._splitString(column);
269         var columnTable = parts[0], alias = parts[2], tableAlias;
270         column = parts[1];
271         if (!columnTable) {
272           if (comb.isInstanceOf(table, Identifier)) {
273             table = table.value;
274           }
275           if (comb.isInstanceOf(table, AliasedExpression)) {
276             tableAlias = table.alias;
277           } else if (comb.isInstanceOf(table, QualifiedIdentifier)) {
278             tableAlias = table;
279           } else {
280             parts = this._splitString(table);
281             var schema = parts[0], tableAlias = parts[2];
282             table = parts[1];
283             if (schema) {
284               tableAlias = new Identifier(tableAlias) || new QualifiedIdentifier(schema, table);
285             }
286           }
287           columnTable = tableAlias || table;
288         }
289         return new QualifiedIdentifier(columnTable, column);
290       } else {
291         return column;
292       }

296      * Creates a unique table alias that hasn't already been used in this dataset.
297      *
298      * @example
299      *
300      * DB.from("table").unusedTableAlias("t");
301      *       //=> "t"
302      *
303      * DB.from("table").unusedTableAlias("table");
304      *   //=> "table0"
305      *
306      * DB.from("table", "table0"]).unusedTableAlias("table");
307      *   //=> "table1"
308      *
309      * @param {String|patio.sql.Identifier} tableAlias the table to get an unused alias for.
310      *
311      * @return {String} the implicit alias that is in tableAlias with a possible "N"
312      *                  if the alias has already been used, where N is an integer starting at 0.
313      */
314     unusedTableAlias:function(tableAlias) {
315       tableAlias = this._toTableName(tableAlias);
316       var usedAliases = [], from, join;
317       if ((from = this.__opts.from) != null) {
318         usedAliases = usedAliases.concat(from.map(function(n) {
319           return this._toTableName(n)
320         }, this));
321       }
322       if ((join = this.__opts.join) != null) {
323         usedAliases = usedAliases.concat(join.map(function(join) {
324           if (join.tableAlias) {
325             return this.__toAliasedTableName(join.tableAlias);
326           } else {
327             return this._toTableName(join.table);
328           }
329         }, this));
330       }
331       if (usedAliases.indexOf(tableAlias) != -1) {
332         var base = tableAlias, i = 0;
333         do {
334           tableAlias = comb.string.format("%s%d", base, i++);
335         } while (usedAliases.indexOf(tableAlias) != -1)
336       }
337       return tableAlias;
338     },

340     /**
341      * Returns a literal representation of a value to be used as part
342      * of an SQL expression.
343      *
344      * @example
345      *
346      *  DB.from("items").literal("abc'def\\") //=> "'abc''def\\\\'"
347      *  DB.from("items").literal("items__id") //=> "items.id"
348      *  DB.from("items").literal([1, 2, 3]) //=> "(1, 2, 3)"
349      *  DB.from("items").literal(DB.from("items")) //=> "(SELECT * FROM items)"
350      *  DB.from("items").literal(sql.x.plus(1).gt("y")); //=> "((x + 1) > y)"
351      *
352      * @throws {patio.QueryError} If an unsupported object is given.
353      * @param {*} v the value to convert the the SQL literal representation
354      *
355      * @return {String} a literal representation of the value.
356      */
357     literal:function(v) {
358       if (comb.isInstanceOf(v, LiteralString)) {
359         return "" + v;
360       } else if (comb.isString(v)) {
361         return this._literalString(v);
362       } else if (comb.isNumber(v)) {
363         return this._literalNumber(v);
364       }
365       else if (comb.isInstanceOf(v, Expression)) {
366         return this._literalExpression(v);
367       }
368       else if (comb.isInstanceOf(v, Dataset)) {
369         return this._literalDataset(v);
370       }
371       else if (comb.isArray(v)) {
372         return this._literalArray(v);
373       } else if (comb.isInstanceOf(v, sql.Year)) {
374         return this._literalYear(v);
375       } else if (comb.isInstanceOf(v, sql.TimeStamp, sql.DateTime)) {
376         return this._literalTimestamp(v);
377       } else if (comb.isDate(v)) {
378         return this._literalDate(v);
379       } else if (comb.isInstanceOf(v, sql.Time)) {
380         return this._literalTime(v);
381       }
382       else if (comb.isNull(v)) {
383         return this._literalNull();
384       }
385       else if (comb.isBoolean(v)) {
386         return this._literalBoolean(v);
387       }
388       else if (comb.isHash(v)) {
389         return this._literalObject(v);
390       } else {
391         return this._literalOther(v);
392       }
393     },

398     /**
399      *
400      * Qualify the given expression to the given table.
401      * @param {patio.sql.Expression} column the expression to qualify
402      * @param table the table to qualify the expression to
403      */
404     _qualifiedExpression:function(e, table) {
405       var h, i, args;
406       if (comb.isString(e)) {
407         //this should not be hit but here just for completeness
408         return this.stringToIdentifier(e);
409       } else if (comb.isArray(e)) {
410         return e.map(function(exp) {
411           return this._qualifiedExpression(exp, table);
412         }, this);
413       } else if (comb.isInstanceOf(e, Identifier)) {
414         return new QualifiedIdentifier(table, e);
415       } else if (comb.isInstanceOf(e, OrderedExpression)) {
416         return new OrderedExpression(this._qualifiedExpression(e.expression, table), e.descending, {nulls:e.nulls});
417       } else if (comb.isInstanceOf(e, AliasedExpression)) {
418         return new AliasedExpression(this._qualifiedExpression(e.expression, table), e.alias);
419       } else if (comb.isInstanceOf(e, CaseExpression)) {
420         args = [this._qualifiedExpression(e.conditions, table), this._qualifiedExpression(e.def, table)];
421         if (e.hasExpression) {
422           args.push(this._qualifiedExpression(e.expression, table));
423         }
424         return CaseExpression.fromArgs(args);
425       } else if (comb.isInstanceOf(e, Cast)) {
426         return new Cast(this._qualifiedExpression(e.expr, table), e.type);
427       } else if (comb.isInstanceOf(e, SQLFunction)) {
428         return SQLFunction.fromArgs([e.f].concat(this._qualifiedExpression(e.args, table)));
429       } else if (comb.isInstanceOf(e, ComplexExpression)) {
430         return ComplexExpression.fromArgs([e.op].concat(this._qualifiedExpression(e.args, table)));
431       } else if (comb.isInstanceOf(e, SubScript)) {
432         return new SubScript(this._qualifiedExpression(e.f, table), this._qualifiedExpression(e.sub, table));
433       } else if (comb.isInstanceOf(e, PlaceHolderLiteralString)) {
434         args = [];
435         var eArgs = e.args;
436         if (comb.isHash(eArgs)) {
437           h = {};
438           for (i in eArgs) {
439             h[i] = this._qualifiedExpression(eArgs[i], table);
440           }
441           args = h;
442         } else {
443           args = this._qualifiedExpression(eArgs, table);
444         }
445         return new PlaceHolderLiteralString(e.str, args, e.parens);
446 
447       } else if (comb.isHash(e)) {
448         h = {};
449         for (i in e) {
450           h[this._qualifiedExpression(i, table) + ""] = this._qualifiedExpression(e[i], table);
451         }
452         return h;
453       } else {
454         return e;
455       }
456     },

676     /**
677      * @private For internal use by patio
678      * SQL fragment for complex expressions
679      **/
680     complexExpressionSql:function(op, args) {
681       var newOp;
682       var isOperators = this._static.IS_OPERATORS, isLiterals = this._static.IS_LITERALS;
683       if ((newOp = isOperators[op]) != null) {
684         var r = args[1], v = comb.isNull(r) ? isLiterals.NULL : isLiterals[r];
685         if (r == null || this.supportsIsTrue) {
686           if (comb.isUndefined(v)) {
687             throw new QueryError(string.format("Invalid argument('%s') used for IS operator", r));
688           }
689           l = args[0];
690           return string.format("(%s %s %s)", comb.isString(l) ? l : this.literal(l), newOp, v);
691         } else if (op == "IS") {
692           return this.complexExpressionSql("EQ", args);
693         } else {
694           return this.complexExpressionSql("OR", [BooleanExpression.fromArgs(["NEQ"].concat(args)), new BooleanExpression("IS", args[0], null)]);
695         }
696 
697       } else if (["IN", "NOTIN"].indexOf(op) != -1) {
698         var cols = args[0], vals = args[1], colArray = comb.isArray(cols), valArray = false, emptyValArray = false;
699 
700         if (comb.isArray(vals)) {
701           valArray = true;
702           emptyValArray = vals.length == 0;
703         }
704         if (colArray) {
705           if (emptyValArray) {
706             if (op == "IN") {
707               return this.literal(BooleanExpression.fromValuePairs(cols.map(function(x) {
708                 return  [x, x];
709               }), "AND", true));
710             } else {
711               return this.literal({1:1});
712             }
713           } else if (!this.supportsMultipleColumnIn) {
714             if (valArray) {
715               var expr = BooleanExpression.fromArgs(["OR"].concat(vals.map(function(vs) {
716                 return BooleanExpression.fromValuePairs(array.zip(cols, vs))
717               })));
718               return this.literal(op == "IN" ? expr : expr.invert())
719             }
720           } else {
721             //If the columns and values are both arrays, use _arraySql instead of
722             //literal so that if values is an array of two element arrays, it
723             //will be treated as a value list instead of a condition specifier.
724             return string.format("(%s %s %s)", comb.isString(cols) ? cols : this.literal(cols), ComplexExpression.IN_OPERATORS[op], valArray ? this._arraySql(vals) : this.literal(vals));
725           }
726         }
727         else {
728           if (emptyValArray) {
729             if (op == "IN") {
730               return this.literal(BooleanExpression.fromValuePairs([
731                 [cols, cols]
732               ], "AND", true));
733             } else {
734               return this.literal({1:1});
735             }
736           } else {
737             return string.format("(%s %s %s)", comb.isString(cols) ? cols : this.literal(cols), ComplexExpression.IN_OPERATORS[op], this.literal(vals));
738           }
739         }
740       } else if ((newOp = this._static.TWO_ARITY_OPERATORS[op]) != null) {
741         var l = args[0];
742         return string.format("(%s %s %s)", comb.isString(l) ? l : this.literal(l), newOp, this.literal(args[1]));
743       } else if ((newOp = this._static.N_ARITY_OPERATORS[op]) != null) {
744         return string.format("(%s)", args.map(this.literal, this).join(" " + newOp + " "));
745       } else if (op == "NOT") {
746         return string.format("NOT %s", this.literal(args[0]));
747       } else if (op == "NOOP") {
748         return this.literal(args[0]);
749       } else {
750         throw new QueryError("Invalid operator " + op);
751       }
752     },

1335     /**
1336      * @returns The SQL fragment specifying the columns and values to SET.
1337      * */
1338     _updateSetSql:function() {
1339       var values = this.__opts.values, defs = this.__opts.defaults, overrides = this.__opts.overrides;
1340       var st = "";
1341       if (comb.isArray(values)) {
1342         var v = [], mergedDefsAndOverrides = false;
1343         values.forEach(function(val) {
1344           if (comb.isHash(val)) {
1345             mergedDefsAndOverrides = true;
1346             val = comb.merge({}, defs || {}, val);
1347             val = comb.merge({}, val, overrides || {});
1348             for (var i in val) {
1349               var ident = this.stringToIdentifier(i);
1350               v.push(this.quoteIdentifier(ident) + " = " + this.literal(val[i]));
1351             }
1352           } else if (comb.isInstanceOf(val, Expression)) {
1353             v.push(this._literalExpression(val).replace(/^\(|\)$/g, ""));
1354           } else {
1355             v.push(val);
1356           }
1357         }, this);
1358         if (!mergedDefsAndOverrides) {
1359           var val = comb.merge({}, defs || {});
1360           val = comb.merge({}, val, overrides || {});
1361           for (var i in val) {
1362             var ident = this.stringToIdentifier(i);
1363             v.push(this.quoteIdentifier(ident) + " = " + this.literal(val[i]));
1364           }
1365         }
1366         st += v.join(this._static.COMMA_SEPARATOR);
1367       } else {
1368         st = values;
1369       }
1370 
1371       return " SET " + st;
