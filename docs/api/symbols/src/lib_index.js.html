 25         /**
 26          * A singleton class that acts as the entry point for all actions performed in patio.
 27          *
 28          * @example
 29          *
 30          * var patio = require("patio");
 31          *
 32          * patio.createConnection(....);
 33          *
 34          * patio.camelize = true;
 35          * patio.quoteIdentifiers=false;
 36          *
 37          * patio.createModel("my_table");
 38          *
 39          *
 40          *  //CHANGING IDENTIFIER INPUT METHOD
 41          *
 42          *
 43          *  //use whatever is passed in
 44          *   patio.identifierInputMethod = null;
 45          *  //convert to uppercase
 46          *  patio.identifierInputMethod = "toUpperCase";
 47          *  //convert to camelCase
 48          *  patio.identifierInputMethod = "camelize";
 49          *  //convert to underscore
 50          *  patio.identifierInputMethod = "underscore";
 51          *
 52          *
 53          *  //CHANGING IDENTIFIER OUTPUT METHOD
 54          *
 55          *  //use whatever the db returns
 56          *   patio.identifierOutputMethod = null;
 57          *  //convert to uppercase
 58          *  patio.identifierOutputMethod = "toUpperCase";
 59          *  //convert to camelCase
 60          *  patio.identifierOutputMethod = "camelize";
 61          *  //convert to underscore
 62          *  patio.identifierOutputMethod = "underscore";
 63          *
 64          *  //TURN QUOTING OFF
 65          *   patio.quoteIdentifiers = false
 66          *
 67          * @constructs
 68          * @augments patio.Time
 69          * @param options
 70          *
 71          * @property {patio.Database} defaultDatabase the default database to use, this property can only be used after the conneciton has
 72          *                             initialized.
 73          * @property {Object[]} adapters that are available to use.
 74          *
 75          *
 76          * @property {patio.Database[]} DATABASES An array of databases that are currently connected.
 77          *
 78          *
 79          * @property {String} identifierInputMethod  Set the method to call on identifiers going into the database.  This affects
 80          *  how identifiers are sent to the database. So if you use camelCased and the db identifiers are all underscored
 81          *  use camelize. The method can include
 82          * <ul>
 83          *     <li>toUpperCase</li>
 84          *     <li>toLowerCase</li>
 85          *     <li>camelize</li>
 86          *     <li>underscore</li>
 87          *     <li>Other String instance method names.</li>
 88          * </ul>
 89          *
 90          * patio uses toUpperCase identifiers in all SQL strings for most databases.
 91          *
 92          *
 93          * @property {String} identifierOutputMethod Set the method to call on identifiers coming out of the database.  This affects
 94          * the how identifiers are represented by calling the method on them.
 95          * The method can include
 96          * <ul>
 97          *     <li>toUpperCase</li>
 98          *     <li>toLowerCase</li>
 99          *     <li>camelize</li>
100          *     <li>underscore</li>
101          *     <li>Other String instance method names.</li>
102          * </ul>
103          * most database implementations in patio use toLowerCase
104          *
105          * @property {Boolean} quoteIdentifiers Set whether to quote identifiers for all databases by default. By default,
106          * patio quotes identifiers in all SQL strings.
107          *
108          */
109         constructor:function () {
110             this._super(arguments);
111             var constants = SQL.Constants;

113                 this[i] = constants[i];
114             }

117         /**
118          * Returns a {@link patio.Database} object that can be used to for querying.
119          *
120          * <p>This method is the entry point for all interactions with a database including getting
121          * {@link patio.Dataset}s for creating queries(see {@link patio.Database#from}).
122          * </p>
123          *
124          * <p>The {@link patio.Database} returned can also be used to create({@link patio.Database#createTable}),
125          * alter(@link patio.Database#alterTable}), rename({@link patio.Database#renameTable}), and
126          * drop({@link patio.Database#dropTable}) as well as many other {@link patio.Database} actions.
127          * </p>
128          *
129          * @example
130          *
131          * //connect using an object
132          * var DB = patio.createConnection({
133          *              host : "127.0.0.1",
134          *              port : 3306,
135          *              type : "mysql",
136          *              maxConnections : 1,
137          *              minConnections : 1,
138          *              user : "test",
139          *              password : "testpass",
140          *              database : 'test'
141          * });
142          * //connect using a connection string
143          * var CONNECT_STRING = "mysql://test:testpass@localhost:3306/test?maxConnections=1&minConnections=1";
144          * var DB = patio.createConnection(CONNECT_STRING);
145          *
146          * //...do something
147          * DB.createTable("myTable", function(){
148          *    this.name("text");
149          *    this.value("integer");
150          * }).then(function(){
151          *     //tables created!!!
152          * });
153          *
154          * @param {String|Object} options the options used to initialize the database connection.
155          *                        This may be a database connetion string or object.
156          * @params {Number} [options.maxConnections = 10] the number of connections to pool.
157          * @params {Number} [options.minConnections = 3] the number of connections to pool.
158          * @param {String} [options.type = "mysql"] the type of database to communicate with.
159          * @params {String} options.user the user to authenticate as.
160          * @params {String} options.password the password of the user.
161          * @params {String} options.database the name of the database to use, the database
162          *                                   specified here is the default database for all connections.
163          */
164         createConnection:function (options) {
165             return Database.connect(options);
166         },
167 
168         /**
169          *  @see patio#createConnection
170          */
171         connect:function () {
172             return this.createConnection.apply(this, arguments);
173         },
174 
175         /**
176          * This method allows one to connect to a database and immediately execute code.
177          * For connection options @see patio#createConnection
178          *
179          * @example
180          *
181          *
182          * var DB;
183          * var CONNECT_STRING = "dummyDB://test:testpass@localhost/dummySchema";
184          * var connectPromise = patio.connectAndExecute(CONNECT_STRING, function (db) {
185          *      db.dropTable("test");
186          *      db.createTable("test", function () {
187          *          this.primaryKey("id");
188          *          this.name(String);
189          *          this.age(Number);
190          *      });
191          * });
192          *
193          * connectPromise.then(function (db) {
194          *      DB = db;
195          *      //do more stuff!
196          *  });
197          *
198          * @param {String|Object} options @see patio#createConnection
199          * @param {Function} cb the function to callback once connected.
200          *
201          * @returns {comb.Promise} a promise that is resolved once the database execution has finished.
202          */
203         connectAndExecute:function (options, cb) {
204             if (!comb.isFunction(cb)) throw new PatioError("callback must be a function");
205             var db = this.createConnection.apply(this, arguments);
206             return comb.executeInOrder(db, patio, function (db, patio) {
207                 cb(db, patio);
208                 return db;
209             });
210         },
211 
212         /**
213          * Disconnects all databases in use.
214          *
215          * @return {comb.Promise} a promise that is resolved once all databases have disconnected.
216          */
217         disconnect:function () {
218             return new PromiseList(this.DATABASES.map(function (d) {
219                 return d.disconnect()
220             }), true);
221         },
