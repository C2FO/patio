 88         /**
 89          * Returns a {@link patio.Database} object that can be used to for querying.
 90          *
 91          * <p>This method is the entry point for all interactions with a database including getting
 92          * {@link patio.Dataset}s for creating queries(see {@link patio.Database#from}).
 93          * </p>
 94          *
 95          * <p>The {@link patio.Database} returned can also be used to create({@link patio.Database#createTable}),
 96          * alter(@link patio.Database#alterTable}), rename({@link patio.Database#renameTable}), and
 97          * drop({@link patio.Database#dropTable}) as well as many other {@link patio.Database} actions.
 98          * </p>
 99          *
100          * @example
101          *
102          * //connect using an object
103          * var DB = patio.createConnection({
104          *              host : "127.0.0.1",
105          *              port : 3306,
106          *              type : "mysql",
107          *              maxConnections : 1,
108          *              minConnections : 1,
109          *              user : "test",
110          *              password : "testpass",
111          *              database : 'test'
112          * });
113          * //connect using a connection string
114          * var CONNECT_STRING = "mysql://test:testpass@localhost:3306/test?maxConnections=1&minConnections=1";
115          * var DB = patio.createConnection(CONNECT_STRING);
116          *
117          * //...do something
118          * DB.createTable("myTable", function(){
119          *    this.name("text");
120          *    this.value("integer");
121          * }).then(function(){
122          *     //tables created!!!
123          * });
124          *
125          * @param {String|Object} options the options used to initialize the database connection.
126          *                        This may be a database connetion string or object.
127          * @params {Number} [options.maxConnections = 10] the number of connections to pool.
128          * @params {Number} [options.minConnections = 3] the number of connections to pool.
129          * @param {String} [options.type = "mysql"] the type of database to communicate with.
130          * @params {String} options.user the user to authenticate as.
131          * @params {String} options.password the password of the user.
132          * @params {String} options.database the name of the database to use, the database
133          *                                   specified here is the default database for all connections.
134          */
135         createConnection:function(options){
136             return Database.connect(options);
137         },

146         /**
147          * This method allows one to connect to a database and immediately execute code.
148          * For connection options @see patio#createConnection
149          *
150          * @example
151          *
152          *
153          * var DB;
154          * var CONNECT_STRING = "dummyDB://test:testpass@localhost/dummySchema";
155          * var connectPromise = patio.connectAndExecute(CONNECT_STRING, function (db) {
156          *      db.dropTable("test");
157          *      db.createTable("test", function () {
158          *          this.primaryKey("id");
159          *          this.name(String);
160          *          this.age(Number);
161          *      });
162          * });
163          *
164          * connectPromise.then(function (db) {
165          *      DB = db;
166          *      //do more stuff!
167          *  });
168          *
169          * @param {String|Object} options @see patio#createConnection
170          * @param {Function} cb the function to callback once connected.
171          *
172          * @returns {comb.Promise} a promise that is resolved once the database execution has finished.
173          */
174         connectAndExecute:function(options, cb){
175             if (!comb.isFunction(cb)) {
176                 throw new PatioError("callback must be a function");
177             }
178             var db = this.createConnection.apply(this, arguments);
179             return comb.executeInOrder(db, patio, function(db, patio){
180                 cb(db, patio);
181                 return db;
182             });
183         },

194         /**
195          * Allows for the importing of multiple models so you do not have to worry about the promise that is returned from create model,
196          * or a directory if an index.js file is used.
197          *
198          * <p>
199          *     To import a group of model files you can do the following:
200          *      <pre class="code">
201          * patio.import(__dirname + "/models/Flight.js",
202          *              __dirname + "/models/Airport.js",
203          *              __dirname + "/models/Airplane.js").then(function(Flight, Airport, Airplane){
204          *                  //...
205          *              });
206          *
207          * patio.import(__dirname + "/models/Flight.js",
208          *              __dirname + "/models/Airport.js",
209          *              __dirname + "/models/Airplane.js").then(function(){
210          *                  var Flight = patio.getModel("flight"),
211          *                      Airport = patio.getModel("airport"),
212          *                      Airplane = patio.getModel("airplane");
213          *              });
214          *      </pre>
215          *     Another approach is to create an index.js file inside of a directory that requires all of the models needed.
216          *     then use {@link patio#import} on that file.
217          *     <pre class="code">
218          * patio.import(__dirname + "/models").then(function(){
219          *      var Flight = patio.getModel("flight"),
220          *          Airport = patio.getModel("airport"),
221          *          Airplane = patio.getModel("airplane");
222          * });
223          *     </pre>
224          * </p>
225          *
226          * @param {...} files a single or list of files to import.
227          *
228          * @return {Promise} returns a promise that will be resolved when all models in the imported files
229          * have been loaded.
230          */
231         import:function(files){
232             files = comb.argsToArray(arguments);
233             return new PromiseList(files.map(hitch(this, "__addModelProxy")), true);
234         },

236         /**
237          * This method is used to create a {@link patio.Model} object. This method returns a Promise which will
238          * be resolved once the model has been loaded.
239          *
240          * @example
241          * patio.addModel("flight", {
242          *      instance:{
243          *              toObject:function () {
244          *                  var obj = this._super(arguments);
245          *                  obj.weekdays = this.weekdaysArray;
246          *                  obj.legs = this.legs.map(function (l) {
247          *                      return l.toObject();
248          *                  });
249          *                  return obj;
250          *              },
251          *
252          *              _setWeekdays:function (weekdays) {
253          *                  this.weekdaysArray = weekdays.split(",");
254          *                  return weekdays;
255          *              }
256          *      },
257          *
258          *      static:{
259          *
260          *          init:function () {
261          *              this.oneToMany("legs", {
262          *                  model:"flightLeg",
263          *                  orderBy:"scheduledDepartureTime",
264          *                  fetchType:this.fetchType.EAGER
265          *              });
266          *          },
267          *
268          *          byAirline:function (airline) {
269          *              return this.filter({airline:airline}).all();
270          *          },
271          *
272          *          arrivesAt:function (airportCode) {
273          *              return this.join(this.flightLeg.select("flightId").filter({arrivalCode:airportCode}).distinct(), {flightId:"id"}).all();
274          *          },
275          *
276          *          departsFrom:function (airportCode) {
277          *              return this.join(this.flightLeg.select("flightId").filter({departureCode:airportCode}).distinct(), {flightId:"id"}).all();
278          *          },
279          *
280          *          getters:{
281          *              flightLeg:function () {
282          *                  if (!this.__flightLeg) {
283          *                      this.__flightLeg = this.patio.getModel("flightLeg");
284          *                  }
285          *                  return this.__flightLeg;
286          *              }
287          *          }
288          *      }
289          *  });
290          *
291          *
292          * @param {String|patio.Dataset} table the table to use as the base for the model.
293          * @param {Object} proto an object to be used as the prototype for the model. See
294          * <a href="http://pollenware.github.com/comb/symbols/comb.html#.define">comb.define</a>.
295          * @param [Object[]] [proto.plugins] this can be used to specify additional plugins to use such as.
296          * <ul>
297          *     <li>{@link patio.plugins.TimeStampPlugin</li>
298          *     <li>{@link patio.plugins.CachePlugin</li>
299          * </ul>
300          *
301          *
302          *
303          */
304         addModel:function(table, proto){
305             var ret = new Promise();
306             model.create(table, proto).then(hitch(this, function(model){
307                 ret.callback(model);
308             }), hitch(ret, "errback"));
309             return ret;
310         },

344         /**
345          * @private
346          *
347          * Proxies all {@link patio#addModel} calls to allow for the easy import of a group of models that
348          * are in a directory.
349          * @param file
350          */
351         __addModelProxy:function(file){
352             var promises = [];
353             var repl = [];
354 
355             var orig = this["addModel"];
356             repl.push({name:"addModel", orig:orig});
357             this["addModel"] = function(arg1, arg2){
358                 try {
359                     var ret;
360                     ret = orig.apply(this, arguments);
361                     promises.push(comb.when(ret));
362                 } catch (e) {
363                     promises.push(new Promise().errback(e));
364                 }
365                 return ret;
366             };
367             require(file);
368             if (promises.length == 0) {
369                 promises.push(new Promise().callback());
370             }
371             return new PromiseList(promises, true).both(hitch(this, function(){
372                 repl.forEach(function(o){
373                     this[o.name] = o.orig;
374                 }, this)
375             }));
376         },

606              * identifiers when sending identifiers to the database. Setting this property to true has the same effect
607              * as:
608              * <pre class="code">
609              *     patio.identifierOutputMethod = "camelize";

621              *          this.state(String, {size:2, allowNull:false});
622              *      });

624              *      //    id integer PRIMARY KEY AUTO_INCREMENT,
625              *      //    airport_code varchar(4) UNIQUE NOT NULL,

628              *      //    state varchar(2) NOT NULL
