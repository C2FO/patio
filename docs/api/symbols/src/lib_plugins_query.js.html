 30         /**
 31          * Forces the reload of the data for a particular model instance.  The Promise returned is resolved with the
 32          * model.
 33          *
 34          * @example
 35          *
 36          * myModel.reload().then(function(model){
 37          *    //model === myModel
 38          * });
 39          *
 40          * @return {comb.Promise} resolved with the model instance.
 41          */
 42         reload:function(){
 43             var ret = new Promise();
 44             if (!this.__isNew) {
 45                 this.dataset.naked().filter(this._getPrimaryKeyQuery()).one().then(hitch(this, function(values){
 46                     this.__set(values, true);
 47                     ret.callback(this);
 48                 }), hitch(ret, "errback"));
 49             } else {
 50                 ret.callback(this);
 51             }
 52             return ret;
 53         },

 55         /**
 56          * This method removes the instance of the model. If the model {@link patio.Model#isNew} then the promise is
 57          * resolved with a 0 indicating no rows were affected. Otherwise the model is removed, primary keys are cleared
 58          * and the model's isNew flag is set to true.
 59          *
 60          * @example
 61          * myModel.remove().then(function(){
 62          *     //model is deleted
 63          *     assert.isTrue(myModel.isNew);
 64          * });
 65          *
 66          * //dont use a transaction to remove this model
 67          * myModel.remove({transaction : false}).then(function(){
 68          *     //model is deleted
 69          *     assert.isTrue(myModel.isNew);
 70          * });
 71          *
 72          * @param {Object} [options] additional options.
 73          * @param {Boolean} [options.transaction] boolean indicating if a transaction should be used when
 74          * removing the model.
 75          *
 76          * @return {comb.Promise} called back after the deletion is successful.
 77          */
 78         remove:function(options){
 79             if (!this.__isNew) {
 80                 return this._checkTransaction(options, comb.hitch(this, function(){
 81                     return comb.executeInOrder(this, this.dataset, function(m, ds){
 82                         m._hook("pre", "remove");
 83                         var ret = ds.filter(m._getPrimaryKeyQuery()).remove();
 84                         m._hook("post", "remove");
 85                         m._clearPrimaryKeys();
 86                         m.__isNew = true;
 87                         return ret;
 88                     });
 89                 }));
 90             } else {
 91                 return new comb.Promise().callback(0);
 92             }
 93         },

113         /**
114          * Updates a model. This action checks if the model is not new and values have changed.
115          * If the model is new then the {@link patio.Model#save} action is called.
116          *
117          * @example
118          *
119          * //set values before saving
120          * someModel.update({
121          *      myVal1 : "newValue1",
122          *      myVal2 : "newValue2",
123          *      myVal3 : "newValue3"
124          *      }).then(function(){
125          *     //do something
126          * });
127          *
128          * //set vals before saving and don't use a transaction
129          * someModel.update({
130          *      myVal1 : "newValue1",
131          *      myVal2 : "newValue2",
132          *      myVal3 : "newValue3"
133          *      }, {transaction : false}).then(function(){
134          *     //do something
135          * });
136          *
137          * //or
138          *
139          * someModel.myVal1 = "newValue1";
140          * someModel.myVal2 = "newValue2";
141          * someModel.myVal3 = "newValue3";
142          *
143          * //update model with current values
144          * someModel.update().then(function(){
145          *     //do something
146          * });
147          *
148          *
149          * //don't use a transaction
150          * someModel.update(null, {transaction : false});
151          *
152          * @param {Object} [vals] optional values hash to set on the model before saving.
153          * @param {Object} [options] additional options.
154          * @param {Boolean} [options.transaction] boolean indicating if a transaction should be used when
155          * updating the model.
156          *
157          * @return {comb.Promise} resolved when the update action has completed.
158          */
159         update:function(vals, options){
160             if (!this.__isNew && this.__isChanged) {
161                 return this._checkTransaction(options, comb.hitch(this, function(){
162                     comb.isHash(vals) && this.__set(vals);
163                     var saveChanged = !comb.isEmpty(this.__changed);
164                     return comb.executeInOrder(this, this.dataset.filter(this._getPrimaryKeyQuery()), function(m, ds){
165                         m._hook("pre", "update");
166                         saveChanged && ds.update(m.__changed);
167                         m._updateReload();
168                         m._hook("post", "update");
169                         return m;
170                     });
171                 }));
172             } else if (this.__isNew && this.__isChanged) {
173                 return this.save(vals, options);
174             } else {
175                 return new comb.Promise().callback(this);
176             }
177         },

179         /**
180          * Updates a model. This action checks if the model is new and values have changed.
181          * If the model is not new then the {@link patio.Model#update} action is called.
182          *
183          * @example
184          *
185          * //set new values and save the model
186          * someModel.save({
187          *      myVal1 : "newValue1",
188          *      myVal2 : "newValue2",
189          *      myVal3 : "newValue3"
190          *      }).then(function(){
191          *          //do something
192          *      });
193          *
194          * //or
195          *
196          * //set new values and save the model and DO NOT use a transaction
197          * someModel.save({
198          *      myVal1 : "newValue1",
199          *      myVal2 : "newValue2",
200          *      myVal3 : "newValue3"
201          *      }, {transaction : false}).then(function(){
202          *          //do something
203          *      });
204          *
205          * //or
206          * someModel.myVal1 = "newValue1";
207          * someModel.myVal2 = "newValue2";
208          * someModel.myVal3 = "newValue3";
209          *
210          * someModel.save().then(function(){
211          *     //do something
212          * });
213          *
214          * //or
215          *
216          * //dont use a transaction
217          * someModel.save(null, {transaction : false}).then(function(){
218          *     //do something
219          * });
220          *
221          * @param {Object} [vals] optional values hash to set on the model before saving.
222          * @param {Object} [options] additional options.
223          * @param {Boolean} [options.transaction] boolean indicating if a transaction should be used when
224          * saving the model.
225          *
226          * @return {comb.Promise} resolved when the save action has completed.
227          */
228         save:function(vals, options){
229             if (this.__isNew) {
230                 return this._checkTransaction(options, comb.hitch(this, function(){
231                     comb.isHash(vals) && this.__set(vals);
232                     var pk = this._static.primaryKey[0];
233                     return comb.executeInOrder(this, this.dataset, function(m, ds){
234                         m._hook("pre", "save");
235                         var id = ds.insert(m._toObject());
236                         m.__ignore = true;
237                         m[pk] = id;
238                         m.__ignore = false;
239                         m.__isNew = false;
240                         m.__isChanged = false;
241                         m._saveReload();
242                         m._hook("post", "save");
243                         return m;
244                     });
245                 }));
246             } else {
247                 return this.update(vals, options);
248             }
249         }

257         /**
258          * Retrieves a record by the primaryKey/s of a table.
259          *
260          * @example
261          *
262          * var User = patio.getModel("user");
263          *
264          * User.findById(1).then(function(userOne){
265          *
266          * });
267          *
268          * //assume the primary key is a compostie of first_name and last_name
269          * User.findById(["greg", "yukon"]).then(function(userOne){
270          *
271          * });
272          *
273          *
274          * @param {*} id the primary key of the record to find.
275          *
276          * @return {comb.Promise} called back with the record or null if one is not found.
277          */
278         findById:function(id){
279             var pk = this.primaryKey;
280             pk = pk.length == 1 ? pk[0] : pk;
281             var q = {};
282             if (comb.isArray(id) && comb.isArray(pk)) {
283                 if (id.length === pk.length) {
284                     pk.forEach(function(k, i){
285                         q[k] = id[i]
286                     });
287                 } else {
288                     throw new ModelError("findById : ids length does not equal the primaryKeys length.");
289                 }
290             } else {
291                 q[pk] = id;
292             }
293             return this.filter(q).one();
294         },

308         /**
309          * Update multiple rows with a set of values.
310          *
311          * @example
312          * var User = patio.getModel("user");
313          *
314          * //BEGIN
315          * //UPDATE `user` SET `password` = NULL WHERE (`last_accessed` <= '2011-01-27')
316          * //COMMIT
317          * User.update({password : null}, function(){
318          *  return this.lastAccessed.lte(comb.date.add(new Date(), "year", -1));
319          * });
320          * //same as
321          * User.update({password : null}, {lastAccess : {lte : comb.date.add(new Date(), "year", -1)}});
322          *
323          * //UPDATE `user` SET `password` = NULL WHERE (`last_accessed` <= '2011-01-27')
324          * User.update({password : null}, function(){
325          *  return this.lastAccessed.lte(comb.date.add(new Date(), "year", -1));
326          * }, {transaction : false});
327          *
328          * @param {Object} vals a hash of values to update. See {@link patio.Dataset#update}.
329          * @param query a filter to apply to the UPDATE.  See {@link patio.Dataset#filter}.
330          *
331          * @param {Object} [options] additional options.
332          * @param {Boolean} [options.transaction] boolean indicating if a transaction should be used when
333          * updating the models.
334          *
335          * @return {Promise} a promise that is resolved once the update statement has completed.
336          */
337         update:function(vals, /*?object*/query, options){
338             options = options || {};
339             var args = comb.argsToArray(arguments);
340             return this._checkTransaction(options, comb.hitch(this, function(){
341                 var dataset = this.dataset;
342                 if (!comb.isUndefined(query)) {
343                     dataset = dataset.filter(query);
344                 }
345                 return dataset.update(vals);
346             }));

349         /**
350          * Remove(delete) models. This can be used to do a mass delete of models.
351          *
352          * @example
353          * var User = patio.getModel("user");
354          *
355          * //remove all users
356          * User.remove();
357          *
358          * //remove all users who's names start with a.
359          * User.remove({name : /A%/i});
360          *
361          * //remove all users who's names start with a, without a transaction.
362          * User.remove({name : /A%/i}, {transaction : false});
363          *
364          * @param {Object} [q] query to filter the rows to remove. See {@link patio.Dataset#filter}.
365          * @param {Object} [options] additional options.
366          * @param {Boolean} [options.transaction] boolean indicating if a transaction should be used when
367          * removing the models.
368          * @param {Boolean} [options.load=true] boolean set to prevent the loading of each model. This is more efficient
369          * but the pre/post remove hooks not be notified of the deletion.
370          *
371          * @return {comb.Promise} called back when the removal completes.
372          */
373         remove:function(q, options){
374             options = options || {};
375             var loadEach = comb.isBoolean(options.load) ? options.load : true;
376             //first find all records so we call alert the middleware for each model
377             return this._checkTransaction(options, comb.hitch(this, function(){
378                 var ds = this.dataset;
379                 ds = ds.filter.call(ds, q);
380                 if (loadEach) {
381                     return ds.map(function(r){
382                         //todo this sucks find a better way!
383                         return r.remove(options);
384                     });
385                 } else {
386                     return ds.remove();
387                 }
388             }));
389         },

420          * Save either a new model or list of models to the database.
421          *
422          * @example
423          * var Student = patio.getModel("student");
424          * Student.save([
425          *      {
426          *          firstName:"Bob",
427          *          lastName:"Yukon",
428          *          gpa:3.689,
429          *          classYear:"Senior"
430          *      },
431          *      {
432          *          firstName:"Greg",
433          *          lastName:"Horn",
434          *          gpa:3.689,
435          *          classYear:"Sohpmore"
436          *      },
437          *      {
438          *          firstName:"Sara",
439          *          lastName:"Malloc",
440          *          gpa:4.0,
441          *          classYear:"Junior"
442          *      },
443          *      {
444          *          firstName:"John",
445          *          lastName:"Favre",
446          *          gpa:2.867,
447          *          classYear:"Junior"
448          *      },
449          *      {
450          *          firstName:"Kim",
451          *          lastName:"Bim",
452          *          gpa:2.24,
453          *          classYear:"Senior"
454          *      },
455          *      {
456          *          firstName:"Alex",
457          *          lastName:"Young",
458          *          gpa:1.9,
459          *          classYear:"Freshman"
460          *      }
461          * ]).then(function(users){
462          *     //work with the users
463          * });
464          *
465          * Save a single record
466          * MyModel.save(m1);
467          *
468          * @param {patio.Model|Object|patio.Model[]|Object[]} record the record/s to save.
469          * @param {Object} [options] additional options.
470          * @param {Boolean} [options.transaction] boolean indicating if a transaction should be used when
471          * saving the models.
472          *
473          * @return {comb.Promise} called back with the saved record/s.
474          */
475         save:function(items, options){
476             options = options || {};
477             return this._checkTransaction(options, hitch(this, function(){
478                 var ps;
479                 if (comb.isArray(items)) {
480                     ps = items.map(function(o){
481                         if (!comb.isInstanceOf(o, this)) {
482                             o = new this(o);
483                         }
484                         return o.save(null, options);
485                     }, this);
486                     var ret = new comb.Promise();
487                     return new PromiseList(ps, true).then(hitch(ret, "callback"), hitch(ret, "errback"));
488                 } else {
489                     var ret = new comb.Promise();
490                     try {
491                         if (!comb.isInstanceOf(items, this)) {
492                             items = new this(items);
493                         }
494                         ret = items.save(null, options);
495                     } catch (e) {
496                         ret.errback(e);
497                     }
498                     return ret;
499                 }
500             }));
501         }
