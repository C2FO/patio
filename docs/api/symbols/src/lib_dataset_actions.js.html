 32          * If a block is given, the array is iterated over after all items have been loaded.
 33          *
 34          * @example
 35          *
 36          * // SELECT * FROM table
 37          * DB.from("table").all().then(function(res){
 38          *      //res === [{id : 1, ...}, {id : 2, ...}, ...];
 39          * });

 44          *    //=> rows == myArr;
 45          * });

 47          * @param {Function} block a block to be called with each item. The return value of the block is ignored.

 52             var a = [];
 53             var ret = new Promise();
 54             this.forEach(hitch(this, function(r){
 55                 a.push(r);
 56             })).then(hitch(this, function(){
 57                 this.postLoad(a);
 58                 block && a.forEach(block, this);
 59                 ret.callback(a);
 60             }), hitch(ret, "errback"));
 61             return ret;

184         /**
185          * If a integer argument is given, it is interpreted as a limit, and then returns all
186          * matching records up to that limit.  If no argument is passed,
187          * it returns the first matching record.  If any other type of
188          * argument(s) is passed, it is given to {@link patio.Dataset#filter} and the
189          * first matching record is returned.  Examples:
190          *
191          * @example
192          *
193          * comb.executeInOrder(DB.from("table"), function(ds){
194          *   // SELECT * FROM table LIMIT 1
195          *   ds.first(); // => {id : 7}
196          *
197          *   // SELECT * FROM table LIMIT 2
198          *   ds.first(2); // => [{id : 6}, {id : 4}]
199          *
200          *   // SELECT * FROM table WHERE (id = 2) LIMIT 1
201          *   ds.first({id : 2}) // => {id : 2}
202          *
203          *
204          *  // SELECT * FROM table WHERE (id = 3) LIMIT 1
205          *   ds.first("id = 3"); // => {id : 3}
206          *
207          *   // SELECT * FROM table WHERE (id = 4) LIMIT 1
208          *   ds.first("id = ?", 4); // => {id : 4}
209          *
210          *   // SELECT * FROM table WHERE (id > 2) LIMIT 1
211          *   ds.first(function(){return this.id.gt(2);}); // => {id : 5}
212          *
213          *
214          *   // SELECT * FROM table WHERE ((id > 4) AND (id < 6)) LIMIT 1
215          *   ds.first("id > ?", 4, function(){
216          *          return this.id.lt(6);
217          *   }); // => {id : 5}
218          *
219          *    // SELECT * FROM table WHERE (id < 2) LIMIT 2
220          *   ds.first(2, function(){
221          *          return this.id.lt(2)
222          *   }); // => [{id:1}]
223          * });
224          *
225          * @param {*} args varargs to be used to limit/filter the result set.
226          *
227          * @return {comb.Promise} a promise that is resolved with the either the first matching record.
228          *                        Or an array of items if a limit was provided as the first argument.
229          */
230         first:function(args){
231             args = comb.argsToArray(arguments);
232             var block = comb.isFunction(args[args.length - 1]) ? args.pop() : null;
233             var ds = block ? this.filter(block) : this;
234             if (!args.length) {
235                 return ds.singleRecord();
236             } else {
237                 args = (args.length == 1) ? args.shift() : args;
238                 if (comb.isNumber(args)) {
239                     return ds.limit(args).all();
240                 } else {
241                     return ds.filter(args).singleRecord();
242                 }
243             }
244         },

277         /**
278          * Inserts multiple records into the associated table. This method can be
279          * used to efficiently insert a large number of records into a table in a
280          * single query if the database supports it. Inserts
281          * are automatically wrapped in a transaction.
282          *
283          * This method is called with a columns array and an array of value arrays:
284          * <pre class="code">
285          *   // INSERT INTO table (x, y) VALUES (1, 2)
286          *   // INSERT INTO table (x, y) VALUES (3, 4)
287          *   DB.from("table").import(["x", "y"], [[1, 2], [3, 4]]).
288          * </pre>
289          *
290          * This method also accepts a dataset instead of an array of value arrays:
291          *
292          * <pre class="code">
293          *  // INSERT INTO table (x, y) SELECT a, b FROM table2
294          *  DB.from("table").import(["x", "y"], DB.from("table2").select("a", "b"));
295          * </pre>
296          *
297          * The method also accepts a commitEvery option that specifies
298          * the number of records to insert per transaction. This is useful especially
299          * when inserting a large number of records, e.g.:
300          *
301          * <pre class="code">
302          *   // this will commit every 50 records
303          *  DB.from("table").import(["x", "y"], [[1, 2], [3, 4], ...], {commitEvery : 50});
304          * </pre>
305          *
306          * @param {Array} columns The columns to insert values for.
307          *                  This array will be used as the base for each values item in the values array.
308          * @param {Array[Array]} values Array of arrays of values to insert into the columns.
309          * @param {Object} [opts] options
310          * @param {Number} [opts.commitEvery] the number of records to insert per transaction.
311          *
312          * @return {comb.Promise} a promise that is resolved once all records have been inserted.
313          */
314         import:function(columns, values, opts){
315             opts = opts || {};
316             if (comb.isInstanceOf(values, Dataset)) {
317                 return this.db.transaction(hitch(this, function(){
318                     return this.insert(columns, values)
319                 }));
320             }
321             var ret = new comb.Promise();
322             if (!values.length) {
323                 ret.callback();
324             }
325             if (!columns.length) {
326                 throw new QueryError("Invalid columns in import");
327             }
328 
329             var sliceSize = opts.commitEvery || opts.slice;
330             if (sliceSize) {
331                 var offset = 0;
332                 var ret = new Promise();
333                 var execute = hitch(this, function(){
334                     if (offset >= values.length) {
335                         ret.callback();
336                         return;
337                     }
338                     this.db.transaction(opts, hitch(this, function(){
339                         this.multiInsertSql(columns, values.slice(offset, sliceSize + offset)).forEach(hitch(this, function(st){
340                             return this.executeDui(st);
341                         }));
342                     })).then(hitch(this, function(){
343                         offset += sliceSize;
344                         execute();
345                     }), hitch(ret, "errback"));
346                 });
347                 execute();
348                 return ret;
349             } else {
350                 var statements = this.multiInsertSql(columns, values);
351                 return this.db.transaction(hitch(this, function(){
352                     statements.forEach(function(st){
353                         this.executeDui(st);
354                     }, this);
355                 }));
356             }
357         },

360          * This is the recommended function to do the insert of multiple items into the
361          * database. This acts as a proxy to the {@link patio.Dataset#import} method so
362          * one can use an array of hashes rather than an array of columns and an array of values.
363          * See {@link patio.Dataset#import} for more information regarding the method of inserting.
364          * <p>
365          *     <b>NOTE:</b>All hashes should have the same keys other wise some values could be missed</b>

367          *
368          * @example
369          *
370          * // INSERT INTO table (x) VALUES (1)
371          * // INSERT INTO table (x) VALUES (2)
372          * DB.from("table").multiInsert([{x : 1}, {x : 2}]).then(function(){
373          *     //...do something
374          * })
375          *
376          * //commit every 50 inserts
377          * DB.from("table").multiInsert([{x : 1}, {x : 2},....], {commitEvery : 50}).then(function(){
378          *     //...do something
379          * });
380          *
381          * @param {Object[]} hashes an array of objects to insert into the database. The keys of
382          * the first item in the array will be used to look up columns in all subsequent objects. If the
383          * array is empty then the promise is resolved immediatly.
384          *
385          * @param {Object} opts See {@link patio.Dataset#import}.
386          *
387          * @return {comb.Promise} See {@link patio.Dataset#import} for return functionality.
388          */
389         multiInsert:function(hashes, opts){
390             opts = opts || {};
391             hashes = hashes || [];
392             var ret = new Promise();
393             if (!hashes.length) {
394                 ret.callback();
395             } else {
396                 var columns = Object.keys(hashes[0]);
397                 return this.import(columns, hashes.map(function(h){
398                     return columns.map(function(c){
399                         return h[c]
400                     })
401                 }), opts);
402             }
403             return ret;
404         },

832         /**
833          * Returns a promise resolved with a string in CSV format containing the dataset records. By
834          * default the CSV representation includes the column titles in the
835          * first line. You can turn that off by passing false as the
836          * includeColumnTitles argument.
837          *
838          * <p>
839          *    <b>NOTE:</b> This does not use a CSV library or handle quoting of values in
840          *          any way.  If any values in any of the rows could include commas or line
841          *          endings, you shouldn't use this.
842          * </p>
843          *
844          * @example
845          *  // SELECT * FROM table
846          *  DB.from("table").toCsv().then(function(csv){
847          *      console.log(csv);

849          *        id,name

856          *      console.log(csv);

860          *  });

862          *  @param {Boolean} [includeColumnTitles=true] Set to false to prevent the printing of the column
863          *  titles as the first line.
864          *
865          *  @return {comb.Promise} a promise that will be resolved with the CSV string of the results of the

867          */
868         toCsv:function(includeColumnTitles){

871             var csv = null, cols = [];

873             n.columns.then(hitch(this, function(cols){
874                 var vals = [];
875                 includeColumnTitles && vals.push(cols.join(", "));
876                 n.forEach(
877                     function(r){
878                         vals.push(cols.map(
879                             function(c){
880                                 return r[c] || ""
881                             }).join(", "));
882                     }).then(function(){

884                     }, hitch(ret, "errback"));
885             }), hitch(ret, "errback"));
