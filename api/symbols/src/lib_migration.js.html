 51         getFileNames:function(){
 52             if (!this.__files) {
 53                 return this._static.getFileNames(this.directory).addCallback(hitch(this, function(files){

 67         /**@lends patio.migrations.Migrator*/

 69         MIGRATION_FILE_PATTERN:/^.+\.\d+\.js$/i,

 73         getFileNames:function(directory){
 74             var ret = new Promise();
 75             fs.readdir(directory, hitch(this, function(err, files){
 76                 if (err) {
 77                     ret.errback(err);
 78                 } else {
 79                     files = files.filter(function(file){
 80                         return file.match(this.MIGRATION_FILE_PATTERN) != null;
 81                     });
 82                     ret.callback(files.map(function(file){
 83                         return path.resolve(directory, file)
 84                     }));
 85                 }
 86             }));
 87             return ret;

 90         /**
 91          * Migrates the database using migration files found in the supplied directory.
 92          * See {@link patio#migrate}
 93          *
 94          * @example
 95          * var DB = patio.connect("my://connection/string");
 96          * patio. migrate(DB, __dirname + "/timestamp_migration").then(function(){
 97          *     console.log("done migrating!");
 98          * });
 99          *
100          * patio. migrate(DB, __dirname + "/timestamp_migration", {target : 0}).then(function(){
101          *     console.log("done migrating down!");
102          * });
103          *
104          *
105          * @param {patio.Database} db the database to migrate
106          * @param {String} directory directory that the migration files reside in
107          * @param {Object} [opts={}] optional parameters.
108          * @param {String} [opts.column] the column in the table that version information should be stored.
109          * @param {String} [opts.table] the table that version information should be stored.
110          * @param {Number} [opts.target] the target migration(i.e the migration to migrate up/down to).
111          * @param {String} [opts.current] the version that the database is currently at if the current version
112          * is not provided it is retrieved from the database.
113          *
114          * @return {Promise} a promise that is resolved once the migration is complete.
115          */
116         run:function(db, directory, opts){
117             opts = opts || {};
118             var ret = new Promise();
119             this.__getMigrator(directory).then(function(migrator){
120                 new migrator(db, directory, opts).run().then(hitch(ret, "callback"), hitch(ret, "errback"));
121             }, hitch(ret, "errback"));
122             return ret;
123         },

125         // Choose the Migrator subclass to use.  Uses the TimestampMigrator
126         // // if the version number appears to be a unix time integer for a year
127         // after 2005, otherwise uses the IntegerMigrator.
128         __getMigrator:function(directory){
129             var ret = new Promise();
130             var retClass = IntegerMigrator;
131             this.getFileNames(directory).then(hitch(this, function(files){
132                 var l = files.length;
133                 if (l) {
134                     for (var i = 0; i < l; i++) {
135                         var file = files[i];
136                         if (parseInt(file.split(this.MIGRATION_SPLITTER)[1], 10) > this.MINIMUM_TIMESTAMP) {
137                             retClass = TimestampMigrator;
138                             break;
139                         }
140                     }
141                 }
142                 ret.callback(retClass);
143 
144             }), hitch(ret, "errback"));
145             return ret;
146         }
147     }

151 /**
152  * @class Migrator that uses the file format {migrationName}.{version}.js, where version starts at 0.
153  * <b>Missing migrations are not allowed</b>
154  *
155  * @augments patio.migrations.Migrator
156  * @name IntegerMigrator
157  * @memberOf patio.migrations
158  */
159 var IntegerMigrator = comb.define(Migrator, {
160     instance:{
161         /**@lends patio.migrations.IntegerMigrator.prototype*/
162         current:null,
163         direction:null,
164         migrations:null,
165 
166         _migrationFiles:null,
167 
168         run:function(){
169             var ret = new Promise(), DB = this.db;
170             comb.when(this._getLatestMigrationVersion(), this._getCurrentMigrationVersion(), hitch(this, function(res){
171                 var target = res[0], current = res[1];
172                 if (current != target) {
173                     var direction = this.direction = current < target ? "up" : "down", isUp = direction === "up";
174                     this._getMigrations(current, target, direction).then(hitch(this, function(migrations){
175                         var runMigration = hitch(this, function(index){
176                             if (index >= migrations.length) {
177                                 ret.callback(migrations[index - 1][1]);
178                             } else {
179                                 var curr = migrations[index], migration = curr[0], version = curr[1];
180                                 var now = new Date();
181                                 var lv = isUp ? version : version + 1;
182                                 DB.logInfo(format("Begin applying migration version %d, direction: %s", lv, direction));
183                                 DB.transaction(hitch(this, function(){
184                                     var ret = new Promise();
185                                     comb.when(migration[direction].apply(DB, [DB])).then(hitch(this, function(args){
186                                         this._setMigrationVersion(version).then(hitch(ret, "callback"), hitch(ret, "errback"));
187                                     }), hitch(ret, "errback"));
188                                     return ret;
189                                 })).then(function(){
190                                         DB.logInfo(format("Finished applying migration version %d, direction: %s, took % 4dms seconds", lv, direction, new Date() - now));
191                                         runMigration(index + 1);
192                                     }, hitch(ret, "errback"));
193                             }
194 
195                         });
196                         runMigration(0);
197                     }), hitch(ret, "errback"));
198                 } else {
199                     ret.callback(target);
200                 }
201 
202             }), hitch(ret, "errback"));
203             return ret;
204         },
205 
206         _getMigrations:function(current, target, direction){
207             var ret = new Promise(), isUp = direction === "up", migrations = [];
208             comb.when(this._getMigrationFiles(), function(files){
209                 try {
210                     for (; isUp ? current <= target : current >= target; isUp ? current++ : current--) {
211                         migrations.push([require(files[current]), current]);
212                     }
213                 } catch (e) {
214                     return ret.errback(e);
215                 }
216                 ret.callback(migrations);
217             }, hitch(ret, "errback"));
218             return ret;
219         },
220 
221 
222         _getMigrationFiles:function(){
223             var ret = new Promise();
224             if (!this._migrationFiles) {
225                 var retFiles = [];
226                 var directory = this.directory;
227                 this.getFileNames().then(hitch(this, function(files){
228                     var l = files.length;
229                     if (l) {
230                         for (var i = 0; i < l; i++) {
231                             var file = files[i];
232                             var version = this.getMigrationVersionFromFile(file);
233                             if (comb.isUndefined(retFiles[version])) {
234                                 retFiles[version] = file;
235                             } else {
236                                 return ret.errback(new MigrationError("Duplicate migration number " + version));
237                             }
238                         }
239                         for (var i = 0; i < l; i++) {
240                             if (comb.isUndefined(retFiles[i])) {
241                                 return ret.errback(new MigrationError("Missing migration for " + i));
242                             }
243                         }
244                     }
245                     this._migrationFiles = retFiles;
246                     ret.callback(retFiles);
247                 }), hitch(ret, "errback"));
248             } else {
249                 ret.callback(this._migrationFiles);
250             }
251             return ret;
252         },
253 
254         _getLatestMigrationVersion:function(){
255             var ret = new Promise();
256             if (!comb.isUndefined(this._opts.target)) {
257                 ret.callback(this._opts.target);
258             } else {
259                 this._getMigrationFiles().then(hitch(this, function(files){
260                     var l = files[files.length - 1];
261                     ret.callback(l ? this.getMigrationVersionFromFile(path.basename(l)) : null);
262                 }), hitch(ret, "errback"));
263             }
264             return ret;
265         },
266 
267         _getCurrentMigrationVersion:function(){
268             var ret = new Promise();
269             if (!comb.isUndefined(this._opts.current)) {
270                 ret.callback(this._opts.current);
271             } else {
272                 comb.when(this._getSchemaDataset(), hitch(this, function(ds){
273                     ds.get(this.column).then(hitch(ret, "callback"), hitch(ret, "errback"));
274                 }), hitch(ret, "errback"));
275             }
276             return ret;
277         },
278 
279         _setMigrationVersion:function(version){
280             var ret = new Promise(), c = this.column;
281             this._getSchemaDataset().then(function(ds){
282                 var item = {};
283                 item[c] = version;
284                 ds.update(item).then(hitch(ret, "callback"), hitch(ret, "callback"));
285             }, hitch(ret, "errback"));
286 
287             return ret;
288         },
289 
290         _getSchemaDataset:function(){
291             var c = this.column, table = this.table;
292             var ret = new Promise();
293             if (!this.__schemaDataset) {
294                 var ds = this.db.from(table);
295                 this.__createOrAlterMigrationTable().then(hitch(this, function(){
296                     ds.isEmpty().then(hitch(this, function(empty){
297                         if (empty) {
298                             var item = {};
299                             item[c] = 0;
300                             this.__schemaDataset = ds;
301                             ds.insert(item).then(hitch(ret, "callback", ds), hitch(ret, "errback"));
302                         } else {
303                             ds.count().then(hitch(this, function(count){
304                                 if (count > 1) {
305                                     ret.errback(new Error("More than one row in migrator table"));
306                                 } else {
307                                     this.__schemaDataset = ds;
308                                     ret.callback(ds);
309                                 }
310                             }), hitch(ret, "errback"));
311                         }
312                     }), hitch(ret, "errback"));
313                 }), hitch(ret, "errback"));
314             } else {
315                 ret.callback(this.__schemaDataset);
316             }
317             return ret;
318         },
319 
320         __createOrAlterMigrationTable:function(){
321             var c = this.column, table = this.table, db = this.db;
322             var ds = this.db.from(table);
323             var ret = new Promise();
324             db.tableExists(table).then(hitch(this, function(exists){
325                 if (!exists) {
326                     db.createTable(table,
327                         function(){
328                             this.column(c, "integer", {"default":0, allowNull:false});
329                         }).then(hitch(ret, "callback"), hitch(ret, "errback"));
330                 } else {
331                     ds.columns.then(function(columns){
332                         if (columns.indexOf(c) == -1) {
333                             db.addColumn(table, c, "integer", {"default":0, allowNull:false})
334                                 .then(hitch(ret, "callback"), hitch(ret, "errback"))
335                         } else {
336                             ret.callback();
337                         }
338                     })
339                 }
340             }), hitch(ret, "errback"));
341             return ret;
342         }
343 
344     },

347         DEFAULT_SCHEMA_COLUMN:"version",
348         DEFAULT_SCHEMA_TABLE:"schema_info"
349     }

354 /**
355  * @class Migrator that uses the file format {migrationName}.{timestamp}.js, where the timestamp
356  * can be anything greater than 20000101.
357  *
358  * @name TimestampMigrator
359  * @augments patio.migrations.Migrator
360  * @memberOf patio.migrations
361  */
362 var TimestampMigrator = comb.define(Migrator, {
363     instance:{
364 
365         constructor:function(db, directory, opts){
366             this._super(arguments);
367             opts = opts || {};
368             this.target = opts.target;
369         },
370 
371         run:function(){
372             var ret = new Promise(), DB = this.db, column = this.column;
373             comb.when(this.__getMirationFiles(), this._getSchemaDataset(), function(res){
374                 var migrations = res[0], ds = res[1];
375                 var runMigration = hitch(this, function(index){
376                     if (index >= migrations.length) {
377                         ret.callback();
378                     } else {
379                         var curr = migrations[index], file = curr[0], migration = curr[1], direction = curr[2];
380                         var now = new Date();
381                         DB.logInfo(format("Begin applying migration file %s, direction: %s", file, direction));
382                         DB.transaction(hitch(this, function(){
383                             var ret = new Promise();
384                             comb.when(migration[direction].apply(DB, [DB])).then(hitch(this, function(args){
385                                 var fileLowerCase = file.toLowerCase();
386                                 var query = {};
387                                 query[column] = fileLowerCase;
388                                 (direction === "up" ? ds.insert(query) : ds.filter(query).remove()).then(hitch(ret, "callback"), hitch(ret, "errback"));
389                             }), hitch(ret, "errback"));
390                             return ret;
391                         })).then(function(){
392                                 DB.logInfo(format("Finished applying migration file %s, direction: %s, took % 4dms seconds", file, direction, new Date() - now));
393                                 runMigration(index + 1);
394                             }, hitch(ret, "errback"));
395                     }
396 
397                 });
398                 runMigration(0);
399             }, hitch(ret, "errback"));
400             return ret;
401         },
402 
403         getFileNames:function(){
404             var ret = new Promise();
405             var sup = this._super(arguments);
406             sup.then(hitch(this, function(files){
407                 ret.callback(files.sort(hitch(this, function(f1, f2){
408                     return this.getMigrationVersionFromFile(f1) - this.getMigrationVersionFromFile(f2);
409                 })));
410             }), hitch(ret, "errback"));
411             return ret;
412         },
413 
414         __getAppliedMigrations:function(){
415             var ret = new Promise();
416             if (!this.__appliedMigrations) {
417                 this._getSchemaDataset().then(hitch(this, function(ds){
418                     comb.when(ds.selectOrderMap(this.column), this.getFileNames(), hitch(this, function(res){
419                         var appliedMigrations = res[0], files = res[1].map(function(f){
420                             return path.basename(f);
421                         });
422                         var l = appliedMigrations.length;
423                         if (l) {
424                             for (var i = 0; i < l; i++) {
425                                 if (files.indexOf(appliedMigrations[i]) == -1) {
426                                     return ret.errback("Applied migrations file not found in directory " + appliedMigrations[i]);
427                                 }
428                             }
429                             this.__appliedMigrations = appliedMigrations;
430                             ret.callback(appliedMigrations);
431                         } else {
432                             this.__appliedMigrations = [];
433                             ret.callback([]);
434                         }
435                     }), hitch(ret, "errback"));
436                 }), hitch(ret, "errback"));
437             } else {
438                 ret.callback(this.__appliedMigrations);
439             }
440             return ret;
441         },
442 
443         __getMirationFiles:function(){
444             var ret = new Promise();
445             var upMigrations = [], downMigrations = [], target = this.target;
446             if (!this.__migrationFiles) {
447                 comb.when(this.getFileNames(), this.__getAppliedMigrations(), hitch(this, function(res){
448                     var files = res[0], appliedMigrations = res[1];
449                     var l = files.length, inserts = [];
450                     if (l > 0) {
451                         try {
452                             for (var i = 0; i < l; i++) {
453                                 var file = files[i], f = path.basename(file), fLowerCase = f.toLowerCase();
454                                 if (!comb.isUndefined(target)) {
455                                     var version = this.getMigrationVersionFromFile(f);
456                                     if (version > target || (version === 0 && target === version)) {
457                                         if (appliedMigrations.indexOf(fLowerCase) != -1) {
458                                             downMigrations.push([f, require(file), "down"]);
459                                         }
460                                     } else if (appliedMigrations.indexOf(fLowerCase) == -1) {
461                                         upMigrations.push([f, require(file), "up"]);
462                                     }
463                                 } else if (appliedMigrations.indexOf(fLowerCase) == -1) {
464                                     upMigrations.push([f, require(file), "up"]);
465                                 }
466                             }
467                         } catch (e) {
468                             return ret.errback(e)
469                         }
470                         this.__migrationFiles = upMigrations.concat(downMigrations.reverse())
471                         ret.callback(this.__migrationFiles);
472                     } else {
473                         return ret.callback();
474                     }
475                 }), hitch(ret, "errback"));
476             } else {
477                 ret.callback(this.__migrationFiles);
478             }
479             return ret;
480         },

483         // Returns the dataset for the schema_migrations table. If no such table
484         // exists, it is automatically created.
485         _getSchemaDataset:function(){
486             var ret = new Promise();
487             if (!this.__schemaDataset) {
488                 var ds = this.db.from(this.table);
489                 this.__createTable().then(hitch(this, function(){
490                     this.__schemaDataset = ds;
491                     ret.callback(ds);
492                 }), hitch(ret, "errback"));
493             } else {
494                 ret.callback(this.__schemaDataset);
495             }
496             return ret;
497         },

499         __convertSchemaInfo:function(){
500             var ret = new Promise(), c = this.column;
501             var ds = this.db.from(this.table);
502             this.db.from(IntegerMigrator.DEFAULT_SCHEMA_TABLE).get(IntegerMigrator.DEFAULT_SCHEMA_COLUMN).then(hitch(this, function(version){
503                 this.getFileNames().then(hitch(this, function(files){
504                     var l = files.length, inserts = [];
505                     if (l > 0) {
506                         for (var i = 0; i < l; i++) {
507                             var f = path.basename(files[i]);
508                             if (this.getMigrationVersionFromFile(f) <= version) {
509                                 var insert = {};
510                                 insert[c] = f;
511                                 inserts.push(ds.insert(insert));
512                             }
513                         }
514                     }
515                     if (inserts.length) {
516                         comb.when.apply(comb, [inserts, hitch(ret, "callback"), hitch(ret, "errback")]);
517                     } else {
518                         ret.callback();
519                     }
520                 }), hitch(ret, "errback"));
521             }), hitch(ret, "errback"));
522             return ret;
523 
524         },
525 
526         __createTable:function(){
527             var c = this.column, table = this.table, db = this.db, intMigrationTable = IntegerMigrator.DEFAULT_SCHEMA_TABLE;
528             var ds = this.db.from(table);
529             var ret = new Promise();
530             comb.when(db.tableExists(table), db.tableExists(intMigrationTable), hitch(this, function(res){
531                 var exists = res[0], intMigratorExists = res[1];
532                 if (!exists) {
533                     db.createTable(table,
534                         function(){
535                             this.column(c, String, {primaryKey:true});
536                         }).addErrback(hitch(ret, "errback"));
537                     if (intMigratorExists) {
538                         db.from(intMigrationTable).all().then(hitch(this, function(versions){
539                             var version;
540                             if (versions.length === 1 && (version = versions[0]) && comb.isNumber(version[Object.keys(version)[0]])) {
541                                 this.__convertSchemaInfo().then(hitch(ret, "callback"), hitch(ret, "errback"));
542                             } else {
543                                 ret.callback();
544                             }
545                         }));
546                     } else {
547                         ret.callback();
548                     }
549                 } else {
550                     ds.columns.then(hitch(this, function(columns){
551                         if (columns.indexOf(c) == -1) {
552                             ret.errback(new MigrationError(format("Migration table %s does not contain column %s", table, c)));
553                         } else {
554                             ret.callback();
555                         }
556                     }), hitch(ret, "errback"));
557                 }
558 
559 
560             }), hitch(ret, "errback"));
561             return ret;
562         }
563     },
