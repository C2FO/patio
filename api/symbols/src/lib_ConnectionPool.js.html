 25         /**
 26          * Checks all deferred connection requests.
 27          */
 28         __checkQueries:function(){
 29             var fc = this.freeCount, def, defQueue = this.__deferredQueue;
 30             while (fc-- >= 0 && defQueue.count) {
 31                 def = defQueue.dequeue();
 32                 var conn = this.getObject();
 33                 if (conn) {
 34                     def.callback(conn);
 35                 } else {
 36                     //we didnt get a conneciton so assume we're out.
 37                     break;
 38                 }
 39                 fc--;
 40             }
 41         },

 44          * Performs a query on one of the connection in this Pool.
 45          *
 46          * @return {comb.Promise} A promise to called back with a connection.
 47          */

 49             var ret = new comb.Promise();
 50             //todo override getObject to make async so creating a connetion can execute setup sql
 51             var conn = this.getObject();
 52             if (!conn) {
 53                 //we need to deffer it
 54                 this.__deferredQueue.enqueue(ret);

 56                 ret.callback(conn);
 57             }
 58             return ret;

 61         /**

 68         returnObject:function(obj){
 69             if (this.count <= this.__maxObjects) {
 70                 this.validate(obj).then(comb.hitch(this, function(valid){
 71                     if (valid) {
 72                         this.__freeObjects.enqueue(obj);
 73                         var index;
 74                         if ((index = this.__inUseObjects.indexOf(obj)) > -1) {
 75                             this.__inUseObjects.splice(index, 1);
 76                         }
 77                         this.__checkQueries();
 78                     } else {
 79                         this.removeObject(obj);
 80                     }
 81                 }));
 82             } else {
 83                 this.removeObject(obj);
 84             }

115         endAll:function(){
116             this.__ending = true;
117             var conn, fQueue = this.__freeObjects, count = this.count, ps = [];
118             while ((conn = this.__freeObjects.dequeue()) != undefined) {
119                 ps.push(this.closeConnection(conn));
120             }
121             var inUse = this.__inUseObjects;
122             for (var i = inUse.length - 1; i >= 0; i--) {
123                 ps.push(this.closeConnection(inUse[i]));
124             }
125             this.__inUseObjects.length = 0;
126             var pls = new comb.PromiseList(ps);
127             ps.length === 0 && pls.callback();
