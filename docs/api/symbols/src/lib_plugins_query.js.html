 40         /**
 41          * Forces the reload of the data for a particular model instance.  The Promise returned is resolved with the
 42          * model.
 43          *
 44          * @example
 45          *
 46          * myModel.reload().then(function(model){
 47          *    //model === myModel
 48          * });
 49          *
 50          * @return {comb.Promise} resolved with the model instance.
 51          */
 52         __reload:function () {
 53             var ret = new Promise();
 54             if (!this.__isNew) {
 55                 this.dataset.naked().filter(this._getPrimaryKeyQuery()).one().then(hitch(this, function (values) {
 56                     this.__set(values, true);
 57                     ret.callback(this);
 58                 }), hitch(ret, "errback"));
 59             } else {
 60                 ret.callback(this);
 61             }
 62             return ret;

 65         /**
 66          * This method removes the instance of the model. If the model {@link patio.Model#isNew} then the promise is
 67          * resolved with a 0 indicating no rows were affected. Otherwise the model is removed, primary keys are cleared
 68          * and the model's isNew flag is set to true.
 69          *
 70          * @example
 71          * myModel.remove().then(function(){
 72          *     //model is deleted
 73          *     assert.isTrue(myModel.isNew);
 74          * });
 75          *
 76          * //dont use a transaction to remove this model
 77          * myModel.remove({transaction : false}).then(function(){
 78          *     //model is deleted
 79          *     assert.isTrue(myModel.isNew);
 80          * });
 81          *
 82          * @param {Object} [options] additional options.
 83          * @param {Boolean} [options.transaction] boolean indicating if a transaction should be used when
 84          * removing the model.
 85          *
 86          * @return {comb.Promise} called back after the deletion is successful.
 87          */
 88         remove:function (options) {
 89             if (!this.__isNew) {
 90                 return this._checkTransaction(options, comb.hitch(this, function () {
 91                     var ret = new Promise();
 92                     this._hook("pre", "remove").then(comb.hitch(this, function () {
 93                         this._remove().then(comb.hitch(this, function () {
 94                             this._hook("post", "remove").then(comb.hitch(this, function () {
 95                                 this._clearPrimaryKeys();
 96                                 this.__isNew = true;
 97                                 ret.callback(this);
 98                             }), comb.hitch(ret, "errback"));
 99                         }), comb.hitch(ret, "errback"));
100                     }), comb.hitch(ret, "errback"));
101                     return ret;
102                 }));
103             } else {
104                 return new comb.Promise().callback(0);
105             }
106         },

131          * Updates a model. This action checks if the model is not new and values have changed.
132          * If the model is new then the {@link patio.Model#save} action is called.
133          *
134          * @example
135          *
136          * //set values before saving
137          * someModel.update({
138          *      myVal1 : "newValue1",
139          *      myVal2 : "newValue2",
140          *      myVal3 : "newValue3"
141          *      }).then(function(){
142          *     //do something
143          * });
144          *
145          * //set vals before saving and don't use a transaction
146          * someModel.update({
147          *      myVal1 : "newValue1",
148          *      myVal2 : "newValue2",
149          *      myVal3 : "newValue3"
150          *      }, {transaction : false}).then(function(){
151          *     //do something
152          * });
153          *
154          * //or
155          *
156          * someModel.myVal1 = "newValue1";
157          * someModel.myVal2 = "newValue2";
158          * someModel.myVal3 = "newValue3";
159          *
160          * //update model with current values
161          * someModel.update().then(function(){
162          *     //do something
163          * });
164          *
165          *
166          * //don't use a transaction
167          * someModel.update(null, {transaction : false});
168          *
169          * @param {Object} [vals] optional values hash to set on the model before saving.
170          * @param {Object} [options] additional options.
171          * @param {Boolean} [options.transaction] boolean indicating if a transaction should be used when
172          * updating the model.
173          *
174          * @return {comb.Promise} resolved when the update action has completed.
175          */
176         update:function (vals, options) {
177             if (!this.__isNew) {
178                 return this._checkTransaction(options, comb.hitch(this, function () {
179                     comb.isHash(vals) && this.__set(vals);
180                     var saveChange = !comb.isEmpty(this.__changed);
181                     var ret = new Promise();
182                     this._hook("pre", "update").then(comb.hitch(this, function () {
183                         (saveChange ? this._update() : new Promise().callback()).then(comb.hitch(this, function () {
184                             this._hook("post", "update").then(comb.hitch(ret, "callback", this), comb.hitch(ret, "errback"));
185                         }), comb.hitch(ret, "errback"));
186                     }), comb.hitch(ret, "errback"));
187 
188                     return ret;
189 
190                 }));
191             } else if (this.__isNew && this.__isChanged) {
192                 return this.save(vals, options);
193             } else {
194                 return new comb.Promise().callback(this);
195             }
196         },

202                 .then(comb.hitch(ret, "callback"), comb.hitch(ret, "errback"));
203             this.__changed = {};
204             return ret;

297         /**
298          * Retrieves a record by the primaryKey/s of a table.
299          *
300          * @example
301          *
302          * var User = patio.getModel("user");
303          *
304          * User.findById(1).then(function(userOne){
305          *
306          * });
307          *
308          * //assume the primary key is a compostie of first_name and last_name
309          * User.findById(["greg", "yukon"]).then(function(userOne){
310          *
311          * });
312          *
313          *
314          * @param {*} id the primary key of the record to find.
315          *
316          * @return {comb.Promise} called back with the record or null if one is not found.
317          */
318         findById:function (id) {
319             var pk = this.primaryKey;
320             pk = pk.length == 1 ? pk[0] : pk;
321             var q = {};
322             if (comb.isArray(id) && comb.isArray(pk)) {
323                 if (id.length === pk.length) {
324                     pk.forEach(function (k, i) {
325                         q[k] = id[i]
326                     });
327                 } else {
328                     throw new ModelError("findById : ids length does not equal the primaryKeys length.");
329                 }
330             } else {
331                 q[pk] = id;
332             }
333             return this.filter(q).one();
334         },

378          * User.update({password : null}, function(){
379          *  return this.lastAccessed.lte(comb.date.add(new Date(), "year", -1));
380          * }, {transaction : false});
381          *
382          * @param {Object} vals a hash of values to update. See {@link patio.Dataset#update}.
383          * @param query a filter to apply to the UPDATE.  See {@link patio.Dataset#filter}.

394             return this._checkTransaction(options, comb.hitch(this, function () {

397                     dataset = dataset.filter(query);

403         /**
404          * Remove(delete) models. This can be used to do a mass delete of models.
405          *
406          * @example
407          * var User = patio.getModel("user");
408          *
409          * //remove all users
410          * User.remove();
411          *
412          * //remove all users who's names start with a.
413          * User.remove({name : /A%/i});
414          *
415          * //remove all users who's names start with a, without a transaction.
416          * User.remove({name : /A%/i}, {transaction : false});
417          *
418          * @param {Object} [q] query to filter the rows to remove. See {@link patio.Dataset#filter}.
419          * @param {Object} [options] additional options.
420          * @param {Boolean} [options.transaction] boolean indicating if a transaction should be used when
421          * removing the models.
422          * @param {Boolean} [options.load=true] boolean set to prevent the loading of each model. This is more efficient
423          * but the pre/post remove hooks not be notified of the deletion.
424          *
425          * @return {comb.Promise} called back when the removal completes.
426          */
427         remove:function (q, options) {
428             options = options || {};
429             var loadEach = comb.isBoolean(options.load) ? options.load : true;
430             //first find all records so we call alert the middleware for each model
431             return this._checkTransaction(options, comb.hitch(this, function () {
432                 var ds = this.dataset;
433                 ds = ds.filter.call(ds, q);
434                 if (loadEach) {
435                     return ds.map(function (r) {
436                         //todo this sucks find a better way!
437                         return r.remove(options);
438                     });
439                 } else {
440                     return ds.remove();
441                 }
442             }));
443         },

474          * Save either a new model or list of models to the database.
475          *

477          * var Student = patio.getModel("student");
478          * Student.save([
479          *      {
480          *          firstName:"Bob",
481          *          lastName:"Yukon",

485          *      {
486          *          firstName:"Greg",

488          *          gpa:3.689,
489          *          classYear:"Sohpmore"
490          *      },
491          *      {
492          *          firstName:"Sara",
493          *          lastName:"Malloc",
494          *          gpa:4.0,
495          *          classYear:"Junior"
496          *      },
497          *      {
498          *          firstName:"John",
499          *          lastName:"Favre",
500          *          gpa:2.867,
501          *          classYear:"Junior"

503          *      {
504          *          firstName:"Kim",
505          *          lastName:"Bim",
506          *          gpa:2.24,
507          *          classYear:"Senior"
508          *      },
509          *      {
510          *          firstName:"Alex",
511          *          lastName:"Young",
512          *          gpa:1.9,
513          *          classYear:"Freshman"
514          *      }
515          * ]).then(function(users){
516          *     //work with the users
517          * });
518          *
519          * Save a single record
520          * MyModel.save(m1);
521          *
522          * @param {patio.Model|Object|patio.Model[]|Object[]} record the record/s to save.
523          * @param {Object} [options] additional options.
524          * @param {Boolean} [options.transaction] boolean indicating if a transaction should be used when
525          * saving the models.
526          *
527          * @return {comb.Promise} called back with the saved record/s.
528          */
529         save:function (items, options) {
530             options = options || {};
531             return this._checkTransaction(options, hitch(this, function () {
532                 var ps;
533                 if (comb.isArray(items)) {
534                     ps = items.map(function (o) {
535                         if (!comb.isInstanceOf(o, this)) {
536                             o = new this(o);
537                         }
538                         return o.save(null, options);
539                     }, this);
540                     var ret = new comb.Promise();
541                     return new PromiseList(ps, true).then(hitch(ret, "callback"), hitch(ret, "errback"));
542                 } else {
543                     var ret = new comb.Promise();
544                     try {
545                         if (!comb.isInstanceOf(items, this)) {
546                             items = new this(items);
547                         }
548                         ret = items.save(null, options);
549                     } catch (e) {
550                         ret.errback(e);
551                     }
552                     return ret;
553                 }
554             }));
555         }
