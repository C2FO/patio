 79         constructor:function (options, patio, filter) {
 80             options = options || {};
 81             if (!options.model) {
 82                 throw new Error("Model is required for " + this.type + " association");
 83             }
 84             this._model = options.model;
 85             this.patio = patio;
 86             this.__opts = options;
 87             !comb.isUndefined(options.isCascading) && (this.isCascading = options.isCascading);
 88             this.filter = filter;
 89             this.readOnly = comb.isBoolean(options.readOnly) ? options.readOnly : false;
 90             this.__hooks =
 91             {before:{add:null, remove:null, "set":null, load:null}, after:{add:null, remove:null, "set":null, load:null}};
 92             var hooks = ["Add", "Remove", "Set", "Load"];
 93             ["before", "after"].forEach(function (h) {
 94                 hooks.forEach(function (a) {
 95                     var hookName = h + a, hook;
 96                     if (comb.isFunction((hook = options[hookName]))) {
 97                         this.__hooks[h][a.toLowerCase()] = hook;
 98                     }
 99                 }, this);
100             }, this);
101             this.fetchType = options.fetchType || fetch.LAZY;
102         },

104         _callHook:function (hook, action, args) {
105             var func = this.__hooks[hook][action], ret;
106             if (comb.isFunction(func)) {
107                 ret = func.apply(this, args);
108             }
109             return ret;

132         _checkAssociationKey:function (parent) {
133             var q = {};
134             this._setAssociationKeys(parent, q);
135             return Object.keys(q).every(function (k) {
136                 return q[k] != null
137             });
138         },

140         _getAssociationKey:function () {
141             var options = this.__opts, key, ret = [], lk, rk;
142             if (!comb.isUndefinedOrNull((key = options.key))) {
143                 if (this.supportsStringKey && comb.isString(key)) {
144                     //normalize the key first!
145                     ret = [
146                         [this.isOwner ? this.defaultLeftKey : key],
147                         [this.isOwner ? key : this.defaultRightKey]
148                     ];
149                 } else if (this.supportsHashKey && comb.isHash(key)) {
150                     var leftKey = Object.keys(key)[0];
151                     var rightKey = key[leftKey];
152                     ret = [
153                         [leftKey],
154                         [rightKey]
155                     ];
156                 } else if (this.supportsCompositeKey && comb.isArray(key)) {
157                     ret = [
158                         [key],
159                         null
160                     ];
161                 }
162             } else if (this.supportsLeftAndRightKey && (!comb.isUndefinedOrNull((lk = options.leftKey))
163                     && !comb.isUndefinedOrNull((rk = options.rightKey)))) {
164                 ret = [
165                     array.toArray(lk), array.toArray(rk)
166                 ];
167             } else {
168                 //todo handle composite primary keys
169                 ret = [
170                     [this.defaultLeftKey],
171                     [this.defaultRightKey]
172                 ];
173             }
174             return ret;
175         },
176 
177 
178         _setAssociationKeys:function (parent, model, val) {
179             var keys = this._getAssociationKey(parent), leftKey = keys[0], rightKey = keys[1];
180             if (leftKey && rightKey) {
181                 for (var i = 0; i < leftKey.length; i++) {
182                     model[rightKey[i]] = !comb.isUndefined(val) ? val : parent[leftKey[i]];
183                 }
184             } else {
185                 leftKey.forEach(function (k) {
186                     model[k] = !comb.isUndefined(val) ? val : parent[k];
187                 });
188             }
189         },
190 
191         _setDatasetOptions:function (ds) {
192             var options = this.__opts || {};
193             var order, limit, distinct, select, query;
194             //allow for multi key ordering
195             if (!comb.isUndefined((select = this.select))) {
196                 ds = ds.select.apply(ds, array.toArray(select));
197             }
198             if (!comb.isUndefined((query = options.query)) || !comb.isUndefined((query = options.conditions))) {
199                 ds = ds.filter(query);
200             }
201             if (comb.isFunction(this.filter)) {
202                 var ret = this.filter.apply(this, [ds]);
203                 if (comb.isInstanceOf(ret, ds._static)) {
204                     ds = ret;
205                 }
206             }
207             if (!comb.isUndefined((distinct = options.distinct))) {
208                 ds = ds.limit.apply(ds, array.toArray(distinct));
209             }
210             if (!comb.isUndefined((order = options.orderBy)) || !comb.isUndefined((order = options.order))) {
211                 ds = ds.order.apply(ds, array.toArray(order));
212             }
213             if (!comb.isUndefined((limit = options.limit))) {
214                 ds = ds.limit.apply(ds, array.toArray(limit));
215             }
216             return ds;
217

220         /**
221          *Filters our associated dataset to load our association.
222          *
223          *@return {Dataset} the dataset with all filters applied.
224          **/
225         _filter:function (parent) {
226             var options = this.__opts || {};
227             var ds;
228             if (!comb.isUndefined((ds = options.dataset)) && comb.isFunction(ds)) {
229                 ds = ds.apply(parent, [parent]);
230             }
231             if (!ds) {
232                 var q = {};
233                 this._setAssociationKeys(parent, q);
234                 ds = this.model.dataset.naked().filter(q);
235                 var recip = this.model._findAssociation(this);
236                 recip && (recip = recip[1]);
237                 ds.rowCb = comb.hitch(this, function (item) {
238                     var ret = new comb.Promise();
239                     var model = this._toModel(item, true);
240                     recip && recip.__setValue(model, parent);
241                     //call hook to finish other model associations
242                     model._hook("post", "load").then(hitch(this, function () {
243                         ret.callback(model);
244                     }), hitch(ret, "errback"));
245                     return ret;
246                 });
247             }
248 
249             return this._setDatasetOptions(ds);
250         },
251 
252         __setValue:function (parent, model) {
253             parent.__associations[this.name] = this._fetchMethod
254                     == "all" ? !comb.isArray(model) ? [model] : model : comb.isArray(model) ? model[0] : model;
255             return parent.__associations[this.name];
256         },
257 
258         fetch:function (parent) {
259             var ret = new Promise();
260             if (this._checkAssociationKey(parent)) {
261                 this._filter(parent)[this._fetchMethod]().then(hitch(this, function (result) {
262                     this.__setValue(parent, result);
263                     ret.callback(result);
264                     parent = null;
265                 }), hitch(ret, "errback"));
266             } else {
267                 this.__setValue(parent, null);
268                 ret.callback(null);
269             }
270             return ret;
271         },

403             var Model = this.model;

414         /**
415          * Method to inject functionality into a model. This method alters the model
416          * to prepare it for associations, and initializes all required middleware calls
417          * to fulfill requirements needed to loaded the associations.
418          *
419          * @param {Model} parent the model that is having an associtaion set on it.
420          * @param {String} name the name of the association.
421          */
422         inject:function (parent, name) {
423             this.name = name;
424             var self = this;
425             this.parent = parent;
426             parent.prototype.__defineGetter__(name, function () {
427                 return  self._getter(this);
428             });
429             parent.prototype.__defineGetter__(this.associatedDatasetName, function () {
430                 return  self._filter(this);
431             });
432 
433             if (!this.readOnly && this.createSetter) {
434                 //define a setter because we arent read only
435                 parent.prototype.__defineSetter__(name, function (vals) {
436                     self._setter(vals, this);
437                 });
438             }
439 
440             //set up all callbacks
441             ["pre", "post"].forEach(function (op) {
442                 ["save", "update", "remove", "load"].forEach(function (type) {
443                     parent[op](type, function (next) {
444                         return self["_" + op + type.charAt(0).toUpperCase() + type.slice(1)](next, this);
445                     });
446                 }, this);
447             }, this);
448         },
