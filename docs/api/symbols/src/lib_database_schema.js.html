 67         /**
 68          * Adds an index to a table for the given columns
 69          *
 70          * <p>
 71          *     This method is a shortcut to {@link patio.Database#alterTable} with an
 72          *     addIndex call.
 73          * </p>
 74          * @example
 75          * DB.addIndex("test", "name", {unique : true});
 76          *      //=> 'CREATE UNIQUE INDEX test_name_index ON test (name)'
 77          * DB.addIndex("test", ["one", "two"]);
 78          *      //=> ''CREATE INDEX test_one_two_index ON test (one, two)''
 79          *
 80          * @param {String} table the table to add the index to.
 81          * @param {String|String[]} columns the name of the column/s to create an index for.
 82          * @param {Object} [options] additional options that can be used when adding an index.
 83          * @param {Boolean} [options.unique] set to true if this this index should have a UNIQUE constraint.
 84          * @param {Boolean} [options.ignoreErrors] set to true to ignore errors.
 85          *
 86          * @return {Promise} a promise that is resolved when the CREATE INDEX action is complete.
 87          * */
 88         addIndex:function (table, columns, options) {
 89             options = options || {};
 90             var ignoreErrors = options.ignoreErrors === true;
 91             var ret = new Promise();
 92             this.alterTable(table,
 93                 function () {
 94                     this.addIndex(columns, options)
 95                 }).then(hitch(ret, "callback"), function (err) {
 96                     if (!ignoreErrors) {
 97                         ret.errback(err);
 98                     } else {
 99                         ret.callback();
100                     }
101                 });
102             return ret;
103         },

233          *  <p>
234          *      <b>NOTE:</b> The block is invoked in the scope of the table that is being altered. The block
235          *      is also called with the table as the first argument. Within the block you must use
236          *      <b>this</b>(If the block has not been bound to a different scope), or the table object
237          *      that is passed in for all alter table operations. See {@link patio.AlterTableGenerator} for

245          *  </p>
246          *
247          * @example
248          *   //using the table object
249          *   DB.alterTable("items", function(table){
250          *      //you must use the passed in table object.
251          *     table.addColumn("category", "text", {default : 'javascript'});
252          *     table.dropColumn("category");
253          *     table.renameColumn("cntr", "counter");
254          *     table.setColumnType("value", "float");
255          *     table.setColumnDefault("value", "float");
256          *     table.addIndex(["group", "category"]);
257          *     table.dropIndex [:group, :category]
258          *   });
259          *
260          *   //using this
261          *    DB.alterTable("items", function(){
262          *     this.addColumn("category", "text", {default : 'javascript'});
263          *     this.dropColumn("category");
264          *     this.renameColumn("cntr", "counter");
265          *     this.setColumnType("value", "float");
266          *     this.setColumnDefault("value", "float");
267          *     this.addIndex(["group", "category"]);
268          *     this.dropIndex [:group, :category]
269          *   });
270          *
271          *   //This will not work
272          *   DB.alterTable("items", comb.hitch(someObject, function(){
273          *    //This is called in the scope of someObject so this
274          *    //will not work and will throw an error
275          *     this.addColumn("category", "text", {default : 'javascript'});
276          *   }));
277          *
278          *    //This will work
279          *   DB.alterTable("items", comb.hitch(someObject, function(table){

281          *    //use the table argument
282          *     table.category("text", {default : 'javascript'});
283          *   }));
284          *
285          *

290          * */

292             if (comb.isFunction(generator)) {
293                 block = generator;
294                 generator = new AlterTableGenerator(this, block);
295             }
296             var ret = new Promise();
297             this.__alterTableSqlList(name, generator.operations).then(hitch(this, function (res) {
298                 var sqls = array.flatten(res.map(function (r) {return r[1]}));
299                 var l = sqls.length;
300                 var drop = hitch(this, function (i) {
301                     if (i < l) {

303                         this.executeDdl(sql).then(hitch(this, drop, i), hitch(ret, "errback"));
304                     } else {
305                         this.removeCachedSchema(name);
306                         ret.callback();
307                     }
308                 });
309                 drop(0);
310             }));
311             return ret;
312         },

599         /**
600          * @private
601          *  The SQL to execute to modify the DDL for the given table name.  op
602          * should be one of the operations returned by the AlterTableGenerator.
603          * */
604         __alterTableSql:function (table, op) {
605             var ret = new comb.Promise();
606             var quotedName = op.name ? this.__quoteIdentifier(op.name) : null;
607             var alterTableOp = null;
608             switch (op.op) {
609                 case "addColumn":
610                     alterTableOp = format("ADD COLUMN %s", this.__columnDefinitionSql(op));
611                     break;
612                 case "dropColumn":
613                     alterTableOp = format("DROP COLUMN %s", quotedName);
614                     break;
615                 case "renameColumn":
616                     alterTableOp = format("RENAME COLUMN %s TO %s", quotedName, this.__quoteIdentifier(op.newName));
617                     break;
618                 case "setColumnType":
619                     alterTableOp = format("ALTER COLUMN %s TYPE %s", quotedName, this.typeLiteral(op));
620                     break;
621                 case "setColumnDefault":
622                     alterTableOp = format("ALTER COLUMN %s SET DEFAULT %s", quotedName, this.literal(op["default"]));
623                     break;
624                 case "setColumnNull":
625                     alterTableOp = format("ALTER COLUMN %s %s NOT NULL", quotedName, op.null ? "DROP" : "SET");
626                     break;
627                 case "addIndex":
628                     return ret.callback(this.__indexDefinitionSql(table, op));
629                     break;
630                 case "dropIndex":
631                     return ret.callback(this.__dropIndexSql(table, op));
632                     break;
