1677 /**
1678  * @class Subclass of {@link moose.sql.ComplexExpression} where the expression results
1679  * in a boolean value in SQL.

1681  * @augments moose.sql.ComplexExpression
1682  * @augments moose.sql.BooleanMethods
1683  * @name BooleanExpression
1684  * @memberOf moose.sql
1685  */
1686 var BooleanExpression = comb.define([ComplexExpression, BooleanMethods], {
1687     static:{
1688         /**@lends moose.sql.BooleanExpression*/
1689 
1690         /**
1691          * Invert the expression, if possible.  If the expression cannot
1692          * be inverted, it throws an {@link moose.error.ExpressionError}.  An inverted expression should match
1693          * everything that the uninverted expression did not match, and vice-versa, except for possible issues with
1694          * SQL NULL (i.e. 1 == NULL is NULL and 1 != NULL is also NULL).
1695          *
1696          * @example
1697          *   BooleanExpression.invert(sql.a) //=> NOT "a"
1698          *
1699          * @param {moose.sql.BooleanExpression} expression
1700          * the expression to invert.
1701          *
1702          * @return {moose.sql.BooleanExpression} the inverted expression.
1703          */
1704         invert:function (expression) {
1705             if (comb.isInstanceOf(expression, BooleanExpression)) {
1706                 var op = expression.op, newArgs;
1707                 if (op == "AND" || op == "OR") {
1708                     newArgs = [OPERTATOR_INVERSIONS[op]].concat(expression.args.map(function (arg) {
1709                         return BooleanExpression.invert(arg);
1710                     }));
1711                     return BooleanExpression.fromArgs(newArgs);
1712                 } else {
1713                     newArgs = [OPERTATOR_INVERSIONS[op]].concat(expression.args);
1714                     return BooleanExpression.fromArgs(newArgs);
1715                 }
1716             } else if (comb.isInstanceOf(expression, StringExpression) || comb.isInstanceOf(expression, NumericExpression)) {
1717                 throw new ExpressionError(comb.string.format("Cannot invert %4j", [expression]));
1718             } else {
1719                 return new BooleanExpression("NOT", expression);
1720             }
1721         },
1722         /**
1723          * Take pairs of values (e.g. a hash or array of two element arrays)
1724          * and converts it to a {@link moose.sql.BooleanExpression}.  The operator and args
1725          * used depends on the case of the right (2nd) argument:
1726          *
1727          * <pre class='code'>
1728          * BooleanExpression.fromValuePairs({a : [1,2,3]}) //=> a IN (1,2,3)
1729          * BooleanExpression.fromValuePairs({a : true}); // a IS TRUE;
1730          * BooleanExpression.fromValuePairs({a : /^A/i}); // a *~ '^A'
1731          * </pre>
1732          *
1733          * If multiple arguments are given, they are joined with the op given (AND
1734          * by default, OR possible).  If negate is set to true,
1735          * all subexpressions are inverted before used.
1736          * <pre class='code'>
1737          * BooleanExpression.fromValuePairs({a : [1,2,3], b : true}) //=> a IN (1,2,3) AND b IS TRUE
1738          * BooleanExpression.fromValuePairs({a : [1,2,3], b : true}, "OR") //=> a IN (1,2,3) OR b IS TRUE
1739          * BooleanExpression.fromValuePairs({a : [1,2,3], b : true}, "OR", true) //=> a NOT IN (1,2,3) AND b IS NOT TRUE
1740          * </pre>
1741          * @param {Object} a object to convert to a {@link moose.sql.BooleanExpression}
1742          * @param {String} [op="AND"] Boolean operator to join each subexpression with.
1743          * <pre class="code">
1744          *     BooleanExpression.fromValuePairs({a : [1,2,3], b : true}, "OR") //=> a IN (1,2,3) OR b IS TRUE
1745          * </pre>
1746          * @param {Boolean} [negate=false] set to try to invert the {@link moose.sql.BooleanExpression}.
1747          * <pre class="code">
1748          * BooleanExpression.fromValuePairs({a : [1,2,3], b : true}, "OR", true) //=> a NOT IN (1,2,3) AND b IS NOT TRUE
1749          * </pre>
1750          * @return {moose.sql.BooleanExpression} expression composed of sub expressions built from the hash.
1751          */
1752         fromValuePairs:function (a, op, negate) {
1753             !Dataset && (Dataset = require("./dataset"));
1754             op = op || "AND", negate = negate || false;
1755             var pairArr = [];
1756             var isArr = comb.isArray(a) && Expression.isConditionSpecifier(a);
1757             for (var k in a) {
1758                 var v = isArr ? a[k][1] : a[k], ret;
1759                 k = isArr ? a[k][0] : k;
1760                 if (comb.isArray(v) || comb.isInstanceOf(v, Dataset)) {
1761                     k = comb.isArray(k) ? k.map(function (i) {
1762                         return comb.isString(i) ? new Identifier(i) : i
1763                     }) : k;
1764                     ret = new BooleanExpression("IN", k, v);
1765                 } else if (comb.isInstanceOf(v, NegativeBooleanConstant)) {
1766                     ret = new BooleanExpression("ISNOT", k, v.constant);
1767                 } else if (comb.isInstanceOf(v, BooleanConstant)) {
1768                     ret = new BooleanExpression("IS", k, v.constant);
1769                 } else if (comb.isNull(v) || comb.isBoolean(v)) {
1770                     ret = new BooleanExpression("IS", k, v);
1771                 } else if (comb.isHash(v)) {
1772                     ret = BooleanExpression.fromValuePairs(v);
1773                 } else if (comb.isRexExp(v)) {
1774                     ret = StringExpression.like(k, v);
1775                 } else {
1776                     ret = new BooleanExpression("EQ", sql.stringToIdentifier(k), v);
1777                 }
1778                 negate && (ret = BooleanExpression.invert(ret));
1779                 pairArr.push(ret);
1780             }
1781             //if We just have one then return the first otherwise create a new Boolean expression
1782             return pairArr.length == 1 ? pairArr[0] : BooleanExpression.fromArgs([op].concat(pairArr));
1783         }
1784     }
1785 }).as(sql, "BooleanExpression");
