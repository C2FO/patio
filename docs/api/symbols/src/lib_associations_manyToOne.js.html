1 var comb = require("comb"),
  2   hitch = comb.hitch,
  3   Promise = comb.Promise,
  4   PromiseList = comb.PromiseList,
  5   _Association = require("./_Association");
  6 /**
  7  * @class Class to define a many to one association.
  8  *
  9  * </br>
 10  * <b>NOT to be instantiated directly</b>
 11  * Its just documented for reference.
 12  *
 13  * @name ManyToOne
 14  * @augments patio.associations.Association
 15  * @memberOf patio.associations
 16  *
 17  * */
 18 module.exports = exports = comb.define(_Association, {
 19   instance:{
 20     _fetchMethod:"one",
 21 
 22     type:"manyToOne",
 23 
 24     isOwner:false,
 25 
 26     _preSave:function(next, model) {
 27       var assoc;
 28       if (this.associationLoaded(model) && (assoc = this.getAssociation(model)) != null && assoc.isNew) {
 29         assoc.save().both(hitch(this, function(assoc) {
 30           var recip = this.model._findAssociation(this);
 31           if (recip) {
 32             //set up our association
 33             recip[1]._setAssociationKeys(assoc, model);
 34           }
 35           next();
 36         }));
 37       } else {
 38         next();
 39       }
 40     },
 41 
 42 
 43     //override
 44     //@see _Association
 45     _postLoad:function(next, model) {
 46       if (this.isEager() && !this.associationLoaded(model)) {
 47         this.fetch(model).then(next, function(e) {
 48           throw e;
 49         });
 50       } else {
 51         next();
 52       }
 53     },
 54 
 55     //override
 56     //@see _Association
 57     _setter:function(val, model) {
 58       if (!comb.isUndefinedOrNull(val)) {
 59         val = this._toModel(val);
 60         this.__setValue(model, val);
 61       } else if (!model.isNew && comb.isNull(val)) {
 62         var keys = this._getAssociationKey(model)[0].forEach(function(k) {
 63           model[k] = null;
 64         });
 65       }
 66     }
 67   }
 68 });
