 55      *  patio.identifierInputMethod = "toUpperCase";
 56      *  //convert to camelCase
 57      *  patio.identifierInputMethod = "camelize";
 58      *  //convert to underscore
 59      *  patio.identifierInputMethod = "underscore";
 60      *
 61      *
 62      *  //CHANGING IDENTIFIER OUTPUT METHOD

 64      *  //use whatever the db returns
 65      *   patio.identifierOutputMethod = null;

 68      *  //convert to camelCase

124     /**
125      * Returns a {@link patio.Database} object that can be used to for querying.
126      *
127      * <p>This method is the entry point for all interactions with a database including getting
128      * {@link patio.Dataset}s for creating queries(see {@link patio.Database#from}).
129      * </p>
130      *
131      * <p>The {@link patio.Database} returned can also be used to create({@link patio.Database#createTable}),
132      * alter(@link patio.Database#alterTable}), rename({@link patio.Database#renameTable}), and
133      * drop({@link patio.Database#dropTable}) as well as many other {@link patio.Database} actions.
134      * </p>
135      *
136      * @example
137      *
138      * //connect using an object
139      * var DB = patio.createConnection({
140      *              host : "127.0.0.1",

157      * }).then(function(){
158      *     //tables created!!!

162      *                        This may be a database connetion string or object.
163      * @params {Number} [options.maxConnections = 10] the number of connections to pool.
164      * @params {Number} [options.minConnections = 3] the number of connections to pool.
165      * @param {String} [options.type = "mysql"] the type of database to communicate with.

182     /**
183      * This method allows one to connect to a database and immediately execute code.
184      * For connection options @see patio#createConnection
185      *
186      * @example
187      *
188      *
189      * var DB;
190      * var CONNECT_STRING = "dummyDB://test:testpass@localhost/dummySchema";
191      * var connectPromise = patio.connectAndExecute(CONNECT_STRING, function (db) {
192      *      db.dropTable("test");
193      *      db.createTable("test", function () {
194      *          this.primaryKey("id");
195      *          this.name(String);
196      *          this.age(Number);
197      *      });
198      * });
199      *
200      * connectPromise.then(function (db) {
201      *      DB = db;
202      *      //do more stuff!
203      *  });
204      *
205      * @param {String|Object} options @see patio#createConnection
206      * @param {Function} cb the function to callback once connected.
207      *
208      * @returns {comb.Promise} a promise that is resolved once the database execution has finished.
209      */
210     connectAndExecute:function(options, cb) {
211       if (!comb.isFunction(cb)) {
212         throw new PatioError("callback must be a function");
213       }
214       var db = this.createConnection.apply(this, arguments);
215       return comb.executeInOrder(db, patio, function(db, patio) {
216         cb(db, patio);
217         return db;
218       });
219     },

232      * or a directory if an index.js file is used.
233      *
234      * <p>
235      *     To import a group of model files you can do the following:
236      *      <pre class="code">
237      * patio.import(__dirname + "/models/Flight.js",
238      *              __dirname + "/models/Airport.js",

247      *                      Airport = patio.getModel("airport"),
248      *                      Airplane = patio.getModel("airplane");
249      *              });
250      *      </pre>
251      *     Another approach is to create an index.js file inside of a directory that requires all of the models needed.
252      *     then use {@link patio#import} on that file.
253      *     <pre class="code">

272     /**
273      * This method is used to create a {@link patio.Model} object. This method returns a Promise which will
274      * be resolved once the model has been loaded.
275      *
276      * @example
277      * patio.addModel("flight", {
278      *      instance:{
279      *              toObject:function () {
280      *                  var obj = this._super(arguments);
281      *                  obj.weekdays = this.weekdaysArray;
282      *                  obj.legs = this.legs.map(function (l) {
283      *                      return l.toObject();
284      *                  });
285      *                  return obj;
286      *              },
287      *
288      *              _setWeekdays:function (weekdays) {
289      *                  this.weekdaysArray = weekdays.split(",");
290      *                  return weekdays;
291      *              }
292      *      },
293      *
294      *      static:{
295      *
296      *          init:function () {
297      *              this.oneToMany("legs", {
298      *                  model:"flightLeg",
299      *                  orderBy:"scheduledDepartureTime",
300      *                  fetchType:this.fetchType.EAGER
301      *              });
302      *          },
303      *
304      *          byAirline:function (airline) {
305      *              return this.filter({airline:airline}).all();
306      *          },
307      *
308      *          arrivesAt:function (airportCode) {
309      *              return this.join(this.flightLeg.select("flightId").filter({arrivalCode:airportCode}).distinct(), {flightId:"id"}).all();
310      *          },
311      *
312      *          departsFrom:function (airportCode) {
313      *              return this.join(this.flightLeg.select("flightId").filter({departureCode:airportCode}).distinct(), {flightId:"id"}).all();
314      *          },
315      *
316      *          getters:{
317      *              flightLeg:function () {
318      *                  if (!this.__flightLeg) {
319      *                      this.__flightLeg = this.patio.getModel("flightLeg");
320      *                  }
321      *                  return this.__flightLeg;
322      *              }
323      *          }
324      *      }
325      *  });
326      *
327      *
328      * @param {String|patio.Dataset} table the table to use as the base for the model.
329      * @param {Object} proto an object to be used as the prototype for the model. See
330      * <a href="http://pollenware.github.com/comb/symbols/comb.html#.define">comb.define</a>.
331      * @param [Object[]] [proto.plugins] this can be used to specify additional plugins to use such as.
332      * <ul>
333      *     <li>{@link patio.plugins.TimeStampPlugin</li>
334      *     <li>{@link patio.plugins.CachePlugin</li>
335      * </ul>
336      *
337      *
338      *
339      */
340     addModel:function(table, proto) {
341       var ret = new Promise();
342       model.create(table, proto).then(hitch(this, function(model) {
343         ret.callback(model);
344       }), hitch(ret, "errback"));
345       return ret;
346     },

380     /**
381      * @private
382      *
383      * Proxies all {@link patio#addModel} calls to allow for the easy import of a group of models that
384      * are in a directory.
385      * @param file
386      */
387     __addModelProxy:function(file) {
388       var promises = [];
389       var repl = [];
390 
391       var orig = this["addModel"];
392       repl.push({name:"addModel", orig:orig});
393       this["addModel"] = function(arg1, arg2) {
394         try {
395           var ret;
396           ret = orig.apply(this, arguments);
397           promises.push(comb.when(ret));
398         } catch (e) {
399           promises.push(new Promise().errback(e));
400         }
401         return ret;
402       };
403       require(file);
404       if (promises.length == 0) {
405         promises.push(new Promise().callback());
406       }
407       return new PromiseList(promises, true).both(hitch(this, function() {
408         repl.forEach(function(o) {
409           this[o.name] = o.orig;
410         }, this)
411       }));
412     },
