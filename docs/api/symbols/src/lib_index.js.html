 94          * alter(@link patio.Database#alterTable}), rename({@link patio.Database#renameTable}), and
 95          * drop({@link patio.Database#dropTable}) as well as many other {@link patio.Database} actions.

 99          *

115          * //...do something
116          * DB.createTable("myTable", function(){

118          *    this.value("integer");
119          * }).then(function(){
120          *     //tables created!!!
121          * });

123          * @param {String|Object} options the options used to initialize the database connection.
124          *                        This may be a database connetion string or object.

126          * @params {Number} [options.minConnections = 3] the number of connections to pool.
127          * @param {String} [options.type = "mysql"] the type of database to communicate with.

144         /**
145          * This method allows one to connect to a database and immediately execute code.
146          * For connection options @see patio#createConnection
147          *
148          * @example
149          *
150          *
151          * var DB;
152          * var CONNECT_STRING = "dummyDB://test:testpass@localhost/dummySchema";
153          * var connectPromise = patio.connectAndExecute(CONNECT_STRING, function (db) {
154          *      db.dropTable("test");
155          *      db.createTable("test", function () {
156          *          this.primaryKey("id");
157          *          this.name(String);
158          *          this.age(Number);
159          *      });
160          * });
161          *
162          * connectPromise.then(function (db) {
163          *      DB = db;
164          *      //do more stuff!
165          *  });
166          *
167          * @param {String|Object} options @see patio#createConnection
168          * @param {Function} cb the function to callback once connected.
169          *
170          * @returns {comb.Promise} a promise that is resolved once the database execution has finished.
171          */
172         connectAndExecute:function (options, cb) {
173             if (!comb.isFunction(cb)) {
174                 throw new PatioError("callback must be a function");
175             }
176             var db = this.createConnection.apply(this, arguments);
177             return comb.executeInOrder(db, patio, function (db, patio) {
178                 cb(db, patio);
179                 return db;
180             });
181         },

281          *                      this.__flightLeg = this.patio.getModel("flightLeg");

291          * @param {patio.Model|patio.Model[]} Parent models of this model.
292          *                  See {@link patio.plugins.ClassTableInheritancePlugin}.

344         /**
345          * @private
346          *
347          * Proxies all {@link patio#addModel} calls to allow for the easy import of a group of models that
348          * are in a directory.
349          * @param file
350          */
351         __addModelProxy:function (file) {
352             var promises = [];
353             var repl = [];
354 
355             var orig = this["addModel"];
356             repl.push({name:"addModel", orig:orig});
357             this["addModel"] = function (arg1, arg2) {
358                 try {
359                     var ret;
360                     ret = orig.apply(this, arguments);
361                     promises.push(comb.when(ret));
362                 } catch (e) {
363                     promises.push(new Promise().errback(e));
364                 }
365                 return ret;
366             };
367             require(file);
368             if (promises.length == 0) {
369                 promises.push(new Promise().callback());
370             }
371             return new PromiseList(promises, true).both(hitch(this, function () {
372                 repl.forEach(function (o) {
373                     this[o.name] = o.orig;
374                 }, this)
375             }));
376         },

386         /**
387          * Migrates the database using migration files found in the supplied directory.
388          * <br/>
389          * <br/>
390          * <div>
391          *     <h3>Integer Migrations</h3>
392          *      Integer migrations are the simpler of the two migrations but are less flexible than timestamp based migrations.
393          *      In order for patio to determine which versions to use the file names must end in <versionNumber>.js where
394          *      versionNumber is a integer value representing the version number. <b>NOTE:</b>With integer migrations
395          *      missing versions are not allowed.
396          *      <br/>
397          *      <br/>
398          *      An example directory structure might look like the following:
399          *
400          *      <pre class="code">
401          * -migrations
402          *      - createFirstTables.0.js
403          *      - shortDescription.1.js
404          *      - another.2.js
405          *      .
406          *      .
407          *      .
408          *      -lastMigration.n.js
409          *      </pre>
410          *      In order to easily identify where certain schema alterations have taken place it is a good idea to provide a brief
411          *      but meaningful migration name.
412          *      <pre class="code">
413          *          createEmployee.0.js
414          *          alterEmployeeNameColumn.1.js
415          *      </pre>
416          *</div>
417          *
418          * <div>
419          *     <h3>Timestamp Migrations</h3>
420          *      Timestamp migrations are the more complex of the two migrations but offer greater flexibility especially
421          *      with development teams. This is because Timestamp migrations do not require consecutive version numbers,
422          *      ,allow for duplicate version numbers(but this should be avoided), keeps track of all currently applied migrations,
423          *      and it will merge missing migrations. In order for patio to determine the order of the migration files
424          *      the file names must end in <timestamp>.js where the timestamp can be any form of a time stamp.
425          *      <pre class="code">
426          * //yyyyMMdd
427          * 20110131
428          * //yyyyMMddHHmmss
429          * 20110131123940
430          * //unix epoch timestamp
431          * 1328035161
432          *      </pre>
433          *      as long as it is greater than 20000101 other wise it will be assumed to be part of an integer migration.
434          *      <br/>
435          *      <br/>
436          *      An example directory structure might look like the following:
437          *
438          *      <pre class="code">
439          * -migrations
440          *      - createFirstTables.1328035161.js
441          *      - shortDescription.1328035360.js
442          *      - another.1328035376.js
443          *      .
444          *      .
445          *      .
446          *      -lastMigration.n.js
447          *      </pre>
448          *      In order to easily identify where certain schema alterations have taken place it is a good idea to provide a brief
449          *      but meaningful migration name.
450          *      <pre class="code">
451          *          createEmployee.1328035161.js
452          *          alterEmployeeNameColumn.1328035360.js
453          *      </pre>
454          *</div>
455          *
456          * <b>NOTE:</b>If you start with IntegerBased migrations and decide to transition to Timestamp migrations the
457          * patio will attempt the migrate the current schema to the timestamp based migration schema.
458          *
459          * <div>
460          * In order to run a migraton all one has to do is call patio.migrate(DB, directory, options);
461          *
462          * <pre class="code">
463          *  var DB = patio.connect("my://connection/string");
464          *  patio.migrate(DB, __dirname + "/migrations").then(function(){
465          *      console.log("migrations finished");
466          *  });
467          * </pre>
468          *
469          * <b>Example migration file</b>
470          * <pre class="code">
471          *
472          * //Up function used to migrate up a version
473          * exports.up = function(db) {
474          *   //create a new table
475          *   db.createTable("company", function() {
476          *       this.primaryKey("id");
477          *       this.companyName(String, {size : 20, allowNull : false});
478          *   });
479          *   db.createTable("employee", function(table) {
480          *       this.primaryKey("id");
481          *       this.firstName(String);
482          *       this.lastName(String);
483          *       this.middleInitial("char", {size : 1});
484          *   });
485          *};
486          *
487          * //Down function used to migrate down version
488          *exports.down = function(db) {
489          *    db.dropTable("employee", "company");
490          *};
491          * </pre>
492          *
493          *</div>
494          *
495          * @param {String|patio.Database} db the database or connection string to a database to migrate.
496          * @param {String} directory directory that the migration files reside in
497          * @param {Object} [opts={}] optional parameters.
498          * @param {String} [opts.column] the column in the table that version information should be stored.
499          * @param {String} [opts.table] the table that version information should be stored.
500          * @param {Number} [opts.target] the target migration(i.e the migration to migrate up/down to).
501          * @param {String} [opts.current] the version that the database is currently at if the current version
502          * is not provided it is retrieved from the database.
503          *
504          * @return {Promise} a promise that is resolved once the migration is complete.
505          */
506         migrate:function (db) {
507             db = comb.isString(db) ? this.connect(db) : db;
508             var args = comb.argsToArray(arguments);
509             args.splice(0, 1);
510             return migrate.run.apply(migrate, [db].concat(args));
511         },
