1 var comb = require("comb"),
  2     sql = require("../sql").sql,
  3     hitch = comb.hitch,
  4     Promise = comb.Promise,
  5     PromiseList = comb.PromiseList,
  6     OneToMany = require("./oneToMany"),
  7     AssociationError = require("../errors").AssociationError;

  9 var LOGGER = comb.logging.Logger.getLogger("comb.associations.ManyToMany");
 10 /**
 11  * @class Class to define a manyToMany association.
 12  *
 13  * </br>
 14  * <b>NOT to be instantiated directly</b>
 15  * Its just documented for reference.
 16  *
 17  * @name ManyToMany
 18  * @augments patio.associations.OneToMany
 19  * @memberOf patio.associations
 20  *
 21  * @param {String} options.joinTable the joinTable of the association.
 22  *
 23  *
 24  * @property {String} joinTable the join table used in the relation.
 25  * */
 26 module.exports = exports = comb.define(OneToMany, {
 27     instance:{
 28 
 29         type:"manyToMany",
 30 
 31         _fetchMethod:"all",
 32 
 33         supportsStringKey:false,
 34 
 35         supportsCompositeKey:false,
 36 
 37 
 38         _filter:function(parent){
 39             var keys = this._getAssociationKey(parent);
 40             var options = this.__opts || {};
 41             var ds;
 42             if (!comb.isUndefined((ds = options.dataset)) && comb.isFunction(ds)) {
 43                 ds = ds.apply(parent, [parent]);
 44             }
 45             if (!ds) {
 46                 ds = this.model.dataset.naked().innerJoin(this.joinTableName, comb.array.zip(keys[1], this.modelPrimaryKey.map(function(k){
 47                     return sql.stringToIdentifier(k);
 48                 })).concat(comb.array.zip(keys[0], this.parentPrimaryKey.map(function(k){
 49                     return parent[k]
 50                 }))))
 51 
 52             }
 53 
 54             return this._setDatasetOptions(ds);
 55         },
 56 
 57         _setAssociationKeys:function(parent, model, val){
 58             var keys = this._getAssociationKey(parent),
 59                 leftKey = keys[0],
 60                 parentPk = this.parentPrimaryKey;
 61             if (!(leftKey && leftKey.length == parentPk.length)) {
 62                 throw new AssociationError("Invalid leftKey for " + this.name + " : " + leftKey);
 63             }
 64             for (var i = 0; i < leftKey.length; i++) {
 65                 model[leftKey[i]] = !comb.isUndefined(val) ? val : parent[parentPk[i]];
 66             }
 67         },
 68 
 69 
 70         __createJoinTableInsertRemoveQuery:function(model, item){
 71             var q = {};
 72             var keys = this._getAssociationKey(model),
 73                 leftKey = keys[0],
 74                 rightKey = keys[1],
 75                 parentPk = this.parentPrimaryKey,
 76                 modelPk = this.modelPrimaryKey;
 77             if (!(leftKey && leftKey.length == parentPk.length)) {
 78                 throw new AssociationError("Invalid leftKey for " + this.name + " : " + leftKey);
 79             }
 80             if (!(rightKey && rightKey.length == modelPk.length)) {
 81                 throw new AssociationError("Invalid rightKey for " + this.name + " : " + rightKey);
 82             }
 83             for (var i = 0; i < leftKey.length; i++) {
 84                 q[leftKey[i]] = model[parentPk[i]];
 85             }
 86             for (var i = 0; i < rightKey.length; i++) {
 87                 q[rightKey[i]] = item[modelPk[i]];
 88             }
 89             return q;
 90         },
 91 
 92         _preRemove:function(next, model){
 93             if (this.isOwner && !this.isCascading) {
 94                 var q = {};
 95                 this._setAssociationKeys(model, q);
 96                 this.joinTable.where(q).remove().then(next, function(e){
 97                     throw e;
 98                 });
 99             } else {
100                 next();
101             }
102         },
103 
104 
105         addAssociation:function(item, model, reload){
106             reload = comb.isBoolean(reload) ? reload : false;
107             var ret = new Promise().callback(model);
108             if (!comb.isUndefinedOrNull(item)) {
109                 if (!model.isNew) {
110                     item = this._toModel(item);
111                     var loaded = this.associationLoaded(model);
112                     var recip = this.model._findAssociation(this), save = item.isNew;
113                     ret = model._checkTransaction(comb.hitch(this, function(){
114                         return comb.executeInOrder(item, model, this, this.joinTable, this.parent, function(item, model, self, joinTable, parent){
115                             save && item.save();
116                             joinTable.insert(self.__createJoinTableInsertRemoveQuery(model, item));
117                             if (recip) {
118                                 recip[1].__setValue(item, [model]);
119                             }
120                             if (loaded && reload) {
121                                 return parent._reloadAssociationsForType(self.type, self.model, model);
122                             } else {
123                                 return model;
124                             }
125                         });
126                     }));
127                 } else {
128                     item = this._toModel(item);
129                     var items = this.getAssociation(model);
130                     if (comb.isUndefinedOrNull(items)) {
131                         this.__setValue(model, [item]);
132                     } else {
133                         items.push(item);
134                     }
135                 }
136             }
137             return ret;
138         },
139 
140         removeItem:function(item, model, remove, reload){
141             reload = comb.isBoolean(reload) ? reload : false;
142             remove = comb.isBoolean(remove) ? remove : false;
143             var ret = new Promise().callback(model);
144             if (!comb.isUndefinedOrNull(item)) {
145                 if (!model.isNew) {
146                     if (comb.isInstanceOf(item, this.model) && !item.isNew) {
147                         var loaded = this.associationLoaded(model), remove = remove && !item.isNew;
148                         var ret = new comb.Promise();
149                         model._checkTransaction(comb.hitch(this, function(){
150                             return comb.executeInOrder(item, this.joinTable, model, this, function(item, joinTable, model, self){
151                                 if (remove) {
152                                     item.remove();
153                                 } else {
154                                     joinTable.where(self.__createJoinTableInsertRemoveQuery(model, item))["remove"]();
155                                 }
156                                 return model;
157                             });
158                         })).then(comb.hitch(this, function(model){
159                             if (loaded && reload) {
160                                 return this.parent._reloadAssociationsForType(this.type, this.model, model).then(comb.hitch(ret, "callback"), comb.hitch(ret, "errback"));
161                             } else {
162                                 ret.callback(model);
163                             }
164                         }), comb.hitch(ret, "errback"));
165                         return ret;
166                     }
167                 } else {
168                     item = this._toModel(item);
169                     var items = this.getAssociation(model), index;
170                     if (!comb.isUndefinedOrNull(items) && (index = items.indexOf(item)) != -1) {
171                         items.splice(index, 1);
172                     }
173                 }
174             }
175             return ret;
176         },
177 
178 
179         getters:{
180 
181             select:function(){
182                 if (!this.__select) {
183                     this.__select = this.__opts.select || new this.patio.SQL.ColumnAll(this.model.tableName);
184                 }
185                 return this.__select;
186             },
187 
188             defaultLeftKey:function(){
189                 return this.__opts.leftKey || this.parent.tableName + "Id";
190             },
191 
192             defaultRightKey:function(){
193                 return this.__opts.rightKey || this.model.tableName + "Id";
194             },
195 
196             parentPrimaryKey:function(){
197                 return this.__opts.leftPrimaryKey || this.parent.primaryKey;
198             },
199 
200             modelPrimaryKey:function(){
201                 return this.__opts.rightPrimaryKey || this.model.primaryKey;
202             },
203 
204             joinTableName:function(){
205                 if (!this._joinTable) {
206                     var options = this.__opts;
207                     var joinTable = options.joinTable;
208                     if (comb.isUndefined(joinTable)) {
209                         var defaultJoinTable = this.defaultJoinTable;
210                         if (comb.isUndefined(defaultJoinTable)) {
211                             throw new Error("Unable to determine jointable for " + this.name)
212                         } else {
213                             this._joinTable = defaultJoinTable;
214                         }
215                     } else {
216                         this._joinTable = joinTable;
217                     }
218                 }
219                 return this._joinTable;
220             },
221 
222             //returns our join table model
223             joinTable:function(){
224                 if (!this.__joinTableDataset) {
225                     var ds = this.__joinTableDataset = this.model.dataset.db.from(this.joinTableName), model = this.model, options = this.__opts;
226                     var identifierInputMethod = comb.isUndefined(options.identifierInputMethod) ? model.identifierInputMethod : options.identifierInputMethod,
227                         identifierOutputMethod = comb.isUndefined(options.identifierOutputMethod) ? model.identifierOutputMethod : options.identifierOutputMethod;
228                     identifierInputMethod && (ds.identifierInputMethod = identifierInputMethod);
229                     identifierOutputMethod && (ds.identifierOutputMethod = identifierOutputMethod);
230                 }
231                 return this.__joinTableDataset;
232             },
233 
234             defaultJoinTable:function(){
235                 var ret;
236                 var recip = this.model._findAssociation(this);
237                 if (recip && recip.length) {
238                     var names = [comb.pluralize(this._model), comb.pluralize(recip[1]._model)].sort();
239                     names[1] = names[1].charAt(0).toUpperCase() + names[1].substr(1);
240                     ret = names.join("");
241                 }
242                 return ret;
243             }
244         }
245     }
