344                 }
345             }, this);

348             if (tableAliasNum) {

705                 cb = args[args.length - 1];
706                 args.pop();
707             } else {
708                 cb = null;
709             }
710             type = args.shift(), table = args.shift(), expr = args.shift(), options = args.shift();
711             expr = comb.isUndefined(expr) ? null : expr, options = comb.isUndefined(options) ? {} : options;
712

714             var usingJoin = comb.isArray(expr) && expr.length && expr.every(function (x) {

735             if (comb.isInstanceOf(table, Dataset)) {
736                 if (!tableAlias) {

738                     tableAlias = this._datasetAlias(tableAliasNum);
739                 }
740                 tableName = tableAlias;
741             } else {
742                 if (table.hasOwnProperty("tableName")) {
743                     table = table.tableName;
744                 }
745                 if (comb.isArray(table)) {
746                     table = table.map(this.stringToIdentifier, this);
747                 } else {
748                     table = comb.isString(table) ? this.stringToIdentifier(table) : table;
749                     var parts = this._splitAlias(table), implicitTableAlias = parts[1];

787             }

1931          * @private
1932          *
1933          * SQL expression object based on the expr type.  See {@link moose.Dataset#filter}
1934          */
1935         _filterExpr:function (expr, cb) {
1936             expr = (comb.isUndefined(expr) || comb.isNull(expr) || (comb.isArray(expr) && !expr.length)) ? null : expr;
1937             if (expr && cb) {
1938                 return new BooleanExpression("AND", this._filterExpr(expr), this._filterExpr(cb))
1939             } else if (cb) {
1940                 expr = cb
1941             }
1942             if (comb.isInstanceOf(expr, Expression)) {
1943                 if (comb.isInstanceOf(expr, NumericExpression, StringExpression)) {
1944                     throw new QueryError("Invalid SQL Expression type : " + expr);
1945                 }
1946                 return expr;
1947             } else if (comb.isArray(expr)) {
1948                 if (expr.length) {
1949                     var first = expr[0];
1950                     if (comb.isString(first)) {
1951                         return new PlaceHolderLiteralString(first, expr.slice(1), true);
1952                     } else if (Expression.isConditionSpecifier(expr)) {
1953                         return BooleanExpression.fromValuePairs(expr)
1954                     } else {
1955                         return BooleanExpression.fromArgs(["AND"].concat(expr.map(function (e) {
1956                             return this._filterExpr(e);
1957                         }, this)));
1958                     }
1959                 }
1960             } else if (comb.isFunction(expr)) {
1961                 return this._filterExpr(expr.call(sql, sql));
1962             } else if (comb.isBoolean(expr)) {
1963                 return new BooleanExpression("NOOP", expr);
1964             } else if (comb.isString(expr) || comb.isInstanceOf(expr, LiteralString)) {
1965                 return new LiteralString("(" + expr + ")");
1966             } else if (comb.isHash(expr)) {
1967                 if (!comb.isEmpty(expr)) {
1968                     return this.__filterObject(expr);
1969                 }
1970             } else {
1971                 throw new QueryError("Invalid filter argument");
1972             }

1975         /**
1976          * @private
1977          *
1978          * This builds an expression from a hash
1979          *
1980          * @example
1981          *
1982          *  Dataset._filterObject({a : 1}) //=> WHERE (a = 1)
1983          *  Dataset._filterObject({x : {gt : 1}}) //=> WHERE (x > 1)
1984          *  Dataset._filterObject({x : {gt : 1}, a : 1}) //=> WHERE ((x > 1) AND (a = 1))
1985          *  Dataset._filterObject({x : {like : "name"}}) //=> WHERE (x LIKE 'name')
1986          *  Dataset._filterObject({x : {iLike : "name"}}) //=> WHERE (x LIKE 'name')
1987          *  Dataset._filterObject({x : {between : [1,10]}}) //=> WHERE ((x >= 1) AND (x <= 10))
1988          *  Dataset._filterObject({x : {notBetween : [1,10]}}) //=> WHERE ((x < 1) OR (x > 10))
1989          *  Dataset._filterObject({x : {neq : 1}}) //=> WHERE (x != 1)
1990          *
1991          * @param {Object} expr the expression we need to create an expression out of
1992          * @param {String} [key=null] the key that the hash corresponds to
1993          *
1994          * @return {moose.sql.Expression} an expression to use in the filter
1995          */
1996         __filterObject:function (expr, key) {
1997             var pairs = [], opts, newKey;
1998             var twoArityOperators = this._static.TWO_ARITY_OPERATORS;
1999             for (var k in expr) {
2000                 var v = expr[k];
2001                 if (comb.isHash(v)) { //its a hash too filter it too!
2002                     pairs.push(this.__filterObject(v, k));
2003                 } else if (key && (twoArityOperators[k.toUpperCase()] || k.match(/between/i))) {
2004                     //its a two arrity operator (e.g. '=', '>')
2005                     newKey = key.split(",");
2006                     if (newKey.length > 1) {
2007                         //this represents a hash where the key represents two columns
2008                         //(e.g. {"col1,col2" : 1}) => WHERE (col1 = 1 AND col2 = 1)
2009                         pairs = pairs.concat(newKey.map(function (k) {
2010                             //filter each column with the expression
2011                             return this.__filterObject(expr, k);
2012                         }, this));
2013                     } else {
2014                         opts = {};
2015                         opts[newKey[0]] = v;
2016                         if (k.match(/^like$/)) {
2017                             //its a like clause {col : {like : "hello"}}
2018                             pairs.push(StringExpression.like.apply(StringExpression, (newKey.concat(comb.isArray(v) ? v : [v]))));
2019                         } else if (k.match(/^iLike$/)) {
2020                             //its a like clause {col : {iLike : "hello"}}
2021                             pairs.push(StringExpression.like.apply(StringExpression, (newKey.concat(comb.isArray(v) ? v : [v]).concat({caseInsensitive:true}))));
2022                         } else if (k.match(/between/i)) {
2023                             //its a like clause {col : {between : [1,10]}}
2024                             pairs.push(this.__createBetweenExpression(opts, k == "notBetween"));
2025                         } else {
2026                             //otherwise is just a boolean expressio
2027                             //it its not a valid operator then we
2028                             //BooleanExpression with throw an error
2029                             pairs.push(this.__createBoolExpression(k, [
2030                                 [newKey[0], expr[k]]
2031                             ]));
2032                         }
2033                     }
2034                 } else {
2035                     //we're not a twoarity operator
2036                     //so we create a boolean expression out of it
2037                     newKey = k.split(",");
2038                     if (newKey.length == 1) {
2039                         newKey = new Identifier(newKey[0]);
2040                     }
2041                     opts = [
2042                         [newKey, v]
2043                     ];
2044                     pairs.push(BooleanExpression.fromValuePairs(opts));
2045                 }
2046             }
2047             //if the total of pairs is one then we just return the first element
2048             //otherwise we join them all with an AND
2049             return pairs.length == 1 ? pairs[0] : BooleanExpression.fromArgs(["AND"].concat(pairs));
2050         },
