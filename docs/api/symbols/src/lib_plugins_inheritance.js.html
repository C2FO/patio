 15 /**
 16  * @class This plugin enables
 17  * <a href="http://www.martinfowler.com/eaaCatalog/classTableInheritance.html" target="patioapi">
 18  *     class table inheritance
 19  * </a>.
 20  *
 21  *<div>
 22  *     Consider the following table model.
 23  *     <pre class="code">
 24  *          employee
 25  *            - id
 26  *            - name (varchar)
 27  *            - kind (varchar)
 28  *     /                          \
 29  * staff                        manager
 30  *   - id (fk employee)           - id (fk employee)
 31  *   - manager_id (fk manger)     - numStaff (number)
 32  *                                 |
 33  *                              executive
 34  *                                - id (fk manager)
 35  *     </pre>
 36  *  <ul>
 37  *      <li><b>employee</b>: This is the parent table of all employee instances.</li>
 38  *      <li><b>staff</b>: Table that inherits from employee where and represents.</li>
 39  *      <li><b>manager</b>: Another subclass of employee.</li>
 40  *      <li><b>executive</b>: Subclass of manager that also inherits from employee through inhertiance</li>
 41  *  </ul>
 42  * <p>
 43  * When setting up you tables the parent table should contain a String column that contains the "kind" of class it is.
 44  * (i.e. employee, staff, manager, executive). This allows the plugin to return the proper instance type when querying
 45  * the tables.
 46  * </p>
 47  * <p>
 48  *     All other tables that inherit from employee should contain a foreign key to their direct super class that is the
 49  *     same name as the primary key of the parent table(<b>employee</b>). So, in the
 50  *     above example <b>staff</b> and <b>manager</b> both contain foreign keys to employee and <b>executive</b> contains
 51  *     a foreign key to <b>manager</b> and they are all named <b>id</b>.
 52  * </p>
 53  *</div>
 54  *
 55  * To set up you models the base super class should contain the ClassTableInheritancePlugin
 56  *
 57  * <pre class="code">
 58  * patio.addModel("employee", {
 59  *      plugins : [patio.plougins.ClassTableInheritancePlugin],
 60  *      static:{
 61  *          init:function () {
 62  *              this._super(arguments);
 63  *              this.configure({key : "kind"});
 64  *          }
 65  *      }
 66  *  });
 67  * </pre>
 68  * All sub classes should just inherit their super class
 69  * <pre class="code">
 70  * var Employee = patio.getModel("employee");
 71  *  patio.addModel("staff", Employee, {
 72  *      static:{
 73  *          init:function () {
 74  *              this._super(arguments);
 75  *              this.manyToOne("manager", {key : "managerId", fetchType : this.fetchType.EAGER});
 76  *          }
 77  *      }
 78  * });
 79  * patio.addModel("manager", Employee, {
 80  *      static:{
 81  *          init:function () {
 82  *              this._super(arguments);
 83  *              this.oneToMany("staff", {key : "managerId", fetchType : this.fetchType.EAGER});
 84  *          }
 85  *      }
 86  * });
 87  *
 88  * </pre>
 89  * Executive inherits from manager, and through inheritance will also receive the oneToMany relationship with staff
 90  * <pre class="code">
 91  * var Manager = require("manager");
 92  * patio.addModel("executive",  Manager);
 93  * </pre>
 94  *
 95  * Working with models
 96  *
 97  * <pre class="code">
 98  * comb.when(
 99  *     new Employee({name:"Bob"}).save(),
100  *     new Staff({name:"Greg"}).save(),
101  *     new Manager({name:"Jane"}).save(),
102  *     new Executive({name:"Sue"}).save()
103  * ).then(function(){
104  *      Employee.all().then(function(emps){
105  *          var bob = emps[0], greg = emps[1], jane = emps[2], sue = emps[3];
106  *          console.log(bob instanceof Employee); //true
107  *          console.log(greg instanceof Employee);  //true
108  *          console.log(greg instanceof Staff);  //true
109  *          console.log(jane instanceof Employee);  //true
110  *          console.log(jane instanceof Manager);  //true
111  *          console.log(sue instanceof Employee);  //true
112  *          console.log(sue instanceof Manager);  //true
113  *          console.log(sue instanceof Executive);  //true
114  *      });
115  * });
116  * </pre>
117  * @name ClassTableInheritancePlugin
118  * @member patio.plugins
119  */
120 comb.define(null, {
121 
122     instance:{
123 
124         // Delete the row from all backing tables, starting from the
125         // most recent table and going through all superclasses.
126         _remove:function () {
127             var q = this._getPrimaryKeyQuery();
128             return new comb.PromiseList(this._static.__ctiTables.slice().reverse().map(function (table) {
129                 return this.db.from(table).filter(q).remove();
130             }, this), true);
131         },
132 
133         // Save each column according to the columns in each table
134         _save:function () {
135             var thisStatic = this._static;
136             var ret = new comb.Promise();
137             if (thisStatic === this._static.__ctiBaseModel) {
138                 return this._super(arguments);
139             } else {
140                 var q = this._getPrimaryKeyQuery(), pk = this.primaryKey[0];
141                 var tables = thisStatic.__ctiTables, ctiColumns = thisStatic.__ctiColumns;
142                 var insertTable = comb.hitch(this, function (index) {
143                     if (index < tables.length) {
144                         var table = tables[index];
145                         var cols = ctiColumns[table], insert = {};
146                         cols.forEach(function (c) {
147                             if (index != 0 || (index == 0 && pk.indexOf(c) == -1)) {
148                                 insert[c] = this[c];
149                             }
150                         }, this);
151                         this.db.from(table).insert(insert).then(comb.hitch(this, function (id) {
152                             if (id) {
153                                 this.__ignore = true;
154                                 //how to handle composite keys.
155                                 this[pk] = id;
156                                 this.__ignore = false;
157                             }
158                             insertTable(++index)
159                         }), comb.hitch(ret, "errback"));
160                     } else {
161                         this.__isNew = false;
162                         this.__isChanged = false;
163                         this._saveReload().then(comb.hitch(this, function () {
164                             ret.callback(this);
165                         }), comb.hitch(ret, "errback"));
166                     }
167                 });
168                 insertTable(0);
169             }
170             return ret;
171         },
172 
173         // update each column according to the columns in each table
174         _update:function () {
175             var q = this._getPrimaryKeyQuery(), changed = this.__changed;
176             var ret = new comb.Promise(), modelStatic = this._static,
177                 ctiColumns = modelStatic.__ctiColumns,
178                 tables = modelStatic.__ctiTables;
179             new comb.PromiseList(tables.map(function (table) {
180                 var cols = ctiColumns[table], update = {};
181                 cols.forEach(function (c) {
182                     if (!comb.isUndefined(changed[c])) {
183                         update[c] = changed[c];
184                     }
185                 }, this);
186                 return comb.isEmpty(update) ? new comb.Promise().callback() : this.db.from(table).filter(q).update(update);
187             }, this), true)
188                 .chain(comb.hitch(this, "_updateReload"), comb.hitch(ret, "errback"))
189                 .then(comb.hitch(ret, "callback"), comb.hitch(ret, "errback"));
190             return ret;
191         }
192 
193 
194     },
195 
196     static:{
197         /**@lends patio.plugins.ClassTableInheritancePlugin*/
198 
199         /**
200          * Configures the plugins with the provided options. <b>Note:</b> This should only be called in the
201          * initial parent class.
202          *
203          * @param {Object} [opts] Additional options to configure behavior of the plugin
204          * @param {String} [opts.key="key"] the column in the base table that contains the name of the subclass.
205          */
206         configure:function (opts) {
207             if (!this.__configured) {
208                 var baseModel = this.__ctiBaseModel = this;
209                 var key = this.__ctiKey = opts.key || "key";
210                 this.__ctiModels = [this];
211                 this.__ctiTables = [this.tableName];
212                 var cols = this.__ctiColumns = {};
213                 cols[this.tableName] = this.columns;
214                 this.dataset.rowCb = comb.hitch(this, function (r) {
215                     if (key) {
216                         var model = this.patio.getModel(r[key]);
217                         if (model != baseModel) {
218                             var q = {};
219                             model.primaryKey.forEach(function (k) {
220                                 q[k] = r[k];
221                             }, this);
222                             var ret = new comb.Promise();
223                             model.dataset.naked().filter(q).one().then(function (vals) {
224                                 model.load(vals).then(comb.hitch(ret, "callback"), comb.hitch(ret, "errback"));
225                             }, comb.hitch(ret, "errback"));
226                             return ret;
227                         } else {
228                             return this.load(r);
229                         }
230                     } else {
231                         return this.load(r);
232                     }
233                 });
234                 this.pre("save", function (next) {
235                     if (key) {
236                         this[key] = this.tableName.toString();
237                     }
238                     next();
239                 });
240                 this.__configured = true;
241             }
242         },
243 
244         /**
245          * <b>Not typically called by user code</b>. Sets up subclass inheritance.
246          *
247          * @param {patio.Model} model model that this class is inheriting from.
248          */
249         inherits:function (model) {
250             this._super(arguments);
251             var ctiKey = this.__ctiKey = model.__ctiKey;
252             this.__ctiTables = model.__ctiTables.slice();
253             this.__ctiModels = model.__ctiModels.slice();
254             this.__ctiModels.push(this);
255             this.__ctiTables.push(this.tableName);
256             this.__ctiColumns = comb.merge({}, model.__ctiColumns);
257             this.__ctiColumns[this.tableName] = this.columns;
258             this.__ctiBaseModel = model.__ctiBaseModel;
259             //copy over our schema
260             var newSchema = comb.merge({}, this.__schema);
261             var schemas = model.__ctiModels.map(
262                 function (m) {
263                     return m.schema;
264                 }).reverse();
265             schemas.forEach(function (s) {
266                 for (var i in s) {
267                     newSchema[i] = s[i];
268                 }
269             }, this);
270             this._setSchema(newSchema);
271             this._setDataset(model.dataset.join(this.tableName, this.primaryKey));
272             this._setPrimaryKey(this.__ctiBaseModel.primaryKey);
273             this.pre("save", function (next) {
274                 if (ctiKey) {
275                     this[ctiKey] = this.tableName.toString();
276                 }
277                 next();
278             });
279         }
280     }
281 }).as(exports, "ClassTableInheritance");
