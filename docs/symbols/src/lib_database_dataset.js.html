1 var comb = require("comb"), hitch = comb.hitch, errors = require("../errors"), NotImplemented = errors.NotImplemented, Dataset = require("../dataset");
  2 
  3 var Database = comb.define(null, {
  4     instance : {
  5 
  6         /*
  7          Fetches records for an arbitrary SQL statement. If a block is given,
  8 
  9          * it is used to iterate over the records:
 10          *
 11          *   DB.fetch('SELECT * FROM items'){|r| p r}
 12          *
 13          * The +fetch+ method returns a dataset instance:
 14          *
 15          *   DB.fetch('SELECT * FROM items').all
 16          *
 17          * +fetch+ can also perform parameterized queries for protection against SQL
 18          * injection:
 19          *
 20          *   DB.fetch('SELECT * FROM items WHERE name = ?', my_name).all
 21          *   */
 22         fetch : function() {
 23             var ret = new comb.Promise();
 24             var args = comb.argsToArray(arguments);
 25             var block = comb.isFunction(args[args.length - 1]) ? args.pop() : null;
 26             var ds = this.dataset.withSql.apply(this.dataset, args);
 27             if (block) {
 28                 ds.forEach(block).then(function() {
 29                     ret.callback(ds);
 30                 }, comb.hitch(ret, "errback"));
 31                 return ret;
 32             } else {
 33                 return ds;
 34             }
 35         },
 36 
 37         /*
 38          * Returns a new dataset with the +from+ method invoked. If a block is given,
 39          * it is used as a filter on the dataset.
 40          *
 41          *   DB.from(:items) # SELECT * FROM items
 42          *   DB.from(:items){id > 2} # SELECT * FROM items WHERE (id > 2)
 43          * */
 44         from : function() {
 45             var args = comb.argsToArray(arguments);
 46             var block = comb.isFunction(args[args.length - 1]) ? args.pop() : null;
 47             var ds = this.dataset.from.apply(this.dataset, args);
 48             return block ? ds.filter(block) : ds;
 49         },
 50 
 51         /*
 52          * Returns a new dataset with the select method invoked.
 53          *
 54          *   DB.select(1) # SELECT 1
 55          *   DB.select{server_version{}} # SELECT server_version()
 56          *   DB.select(:id).from(:items) # SELECT id FROM items
 57          *   */
 58         select : function() {
 59             return this.dataset.select.apply(this.dataset, arguments);
 60         },
 61 
 62         getters : {
 63             dataset : function() {
 64                 return new Dataset(this);
 65             }
 66         }
 67 
 68     }
 69 }).as(module);
