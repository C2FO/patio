100          *

393         /* The SQL to execute to modify the DDL for the given table name.  op
394          * should be one of the operations returned by the AlterTableGenerator.
395          * */
396         __alterTableSql : function(table, op) {
397             var ret = new comb.Promise();
398             var quotedName = op.name ? this.__quoteIdentifier(op.name) : null;
399             var alterTableOp = null;
400             switch (op.op) {
401                 case "addColumn":
402                     alterTableOp = format("ADD COLUMN %s", this.__columnDefinitionSql(op));
403                     break;
404                 case "dropColumn":
405                     alterTableOp = format("DROP COLUMN %s", quotedName);
406                     break;
407                 case "renameColumn":
408                     alterTableOp = format("RENAME COLUMN %s TO %s", quotedName, this.__quoteIdentifier(op.newName));
409                     break;
410                 case "setColumnType":
411                     alterTableOp = format("ALTER COLUMN %s TYPE %s", quotedName, this.typeLiteral(op));
412                     break;
413                 case "setColumnDefault":
414                     alterTableOp = format("ALTER COLUMN %s SET DEFAULT %s", quotedName, this.literal(op["default"]));
415                     break;
416                 case "setColumnNull":
417                     alterTableOp = format("ALTER COLUMN %s %s NOT NULL", quotedName, op.null ? "DROP" : "SET");
418                     break;
419                 case "addIndex":
420                     return ret.callback(this.__indexDefinitionSql(table, op));
421                     break;
422                 case "dropIndex":
423                     return ret.callback(this.__dropIndexSql(table, op));
424                     break;
425                 case "addConstraint":
426                     alterTableOp = format("ADD %s", this.__constraintDefinitionSql(op));
427                     break;
428                 case "dropConstraint":
429                     alterTableOp = format("DROP CONSTRAINT %s", quotedName);
430                     break;
431                 default :
432                     throw new DatabaseError("Invalid altertable operator");
433             }
434             return ret.callback(format("ALTER TABLE %s %s", this.__quoteSchemaTable(table), alterTableOp));
435         },

495         //SQL DDL fragment specifying a constraint on a table.
496         __constraintDefinitionSql : function(constraint) {
497             var sql = constraint.name ? format("CONSTRAINT %s ", this.__quoteIdentifier(constraint.name)) : "";
498             switch (constraint.type) {
499                 case "check":
500                     var check = constraint.check;
501                     sql += format("CHECK %s", this.__filterExpr(comb.isArray(check) && check.length == 1 ? check[0] : check));
502                     break;
503                 case "primaryKey":
504                     sql += format("PRIMARY KEY %s", this.literal(constraint.columns));
505                     break;
506                 case "foreignKey":
507                     sql += this.__columnReferencesTableConstraintSql(constraint);
508                     break;
509                 case "unique":
510                     sql += format("UNIQUE %s", this.literal(constraint.columns));
511                     break;
512                 default:
513                     throw new DatabaseError(format("Invalid constriant type %s, should be 'check', 'primaryKey', foreignKey', or 'unique'", contraint.type));
514             }
515             return sql;
