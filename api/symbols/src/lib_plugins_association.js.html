218         /**
219          * Associates a related model with the current model. The following types are
220          * supported:
221          *
222          * <ul>
223          *     <li><b>oneToMany</b> - Foreign key in associated model's table points to this
224          *         model's primary key.   Each current model object can be associated with
225          *         more than one associated model objects.  Each associated model object
226          *         can be associated with only one current model object.</li>
227          *     <li><b>manyToOne</b> - Foreign key in current model's table points to
228          *         associated model's primary key.  Each associated model object can
229          *         be associated with more than one current model objects.  Each current
230          *         model object can be associated with only one associated model object.</li>
231          *     <li><b>oneToOne</b> - Similar to one_to_many in terms of foreign keys, but
232          *         only one object is associated to the current object through the
233          *         association.  The methods created are similar to many_to_one, except
234          *         that the one_to_one setter method saves the passed object./li>
235          *     <li><b>manyToMany</b> - A join table is used that has a foreign key that points
236          *         to this model's primary key and a foreign key that points to the
237          *         associated model's primary key.  Each current model object can be
238          *         associated with many associated model objects, and each associated
239          *         model object can be associated with many current model objects.</li>
240          * </ul>
241          *
242          * @param {patio.Model.ONE_TO_ONE|patio.Model.ONE_TO_MANY|patio.Model.MANY_TO_ONE|patio.Model.MANY_TO_MANY} type the
243          *          type of association that is to be created.
244          * @param {String} name the name of the association, the name specified will be exposed as a property on instances
245          * of the model.
246          @param {Object} [options] additional options.
247          * The following options can be supplied:
248          * <ul>
249          *   <li><b>model</b> - The associated class or its name. If not  given, uses the association's name,
250          *    which is singularized unless the type is MANY_TO_ONE or ONE_TO_ONE</li>
251          *   <li><b>query</b> - The conditions to use to filter the association, can be any argument passed
252          *   to {@link patio.Dataset#filter}.</li>
253          *   <li><b>dataset</b> - A function that is called in the scope of the model and called with the model as the
254          *     first argument. The function must return a dataset that can be used as the base for all dataset
255          *     operations.<b>NOTE:</b>The dataset returned will have all options applied to it.</li>
256          *   <li><b>distinct</b> Use the DISTINCT clause when selecting associated objects.
257          *   See {@link patio.Dataset#distinct}.</li>
258          *   <li><b>limit</b> : Limit the number of records to the provided value.  Use
259          *     an array with two elements for the value to specify a limit (first element) and an
260          *     offset (second element). See {@link patio.Dataset#limit}.</li>
261          *   <li><b>order</b> : the column/s order the association dataset by.  Can be a
262          *     one or more columns.
263          *     See {@link patio.Dataset#order}.</li>
264          *   <li><b>readOnly</b> : Do not add a setter method (for MANY_TO_ONE or ONE_TO_ONE associations),
265          *     or add/remove/removeAll methods (for ONE_TO_MANY and MANY_TO_MANY associations).</li>
266          *   <li><b>select</b> : the columns to select.  Defaults to the associated class's
267          *     tableName.* in a MANY_TO_MANY association, which means it doesn't include the attributes from the
268          *     join table.  If you want to include the join table attributes, you can
269          *     use this option, but beware that the join table attributes can clash with
270          *     attributes from the model table, so you should alias any attributes that have
271          *     the same name in both the join table and the associated table.</li>
272          *   </ul>
273          *   ManyToOne additional options:
274          *   <ul>
275          *      <li><b>key</b> : foreignKey in current model's table that references
276          *          associated model's primary key.  Defaults to : "{tableName}Id".  Can use an
277          *          array of strings for a composite key association.</li>
278          *       <li><b>primaryKey</b> : column in the associated table that the <b>key</b> option references.
279          *              Defaults to the primary key of the associated table. Can use an
280          *              array of strings for a composite key association.</li>
281          *   </ul>
282          * OneToMany and OneToOne additional options:
283          * <ul>
284          *      <li><b>key</b> : foreign key in associated model's table that references
285          *             current model's primary key, as a string.  Defaults to
286          *             "{thisTableName}Id".  Can use an array of columns for a composite key association.</li>
287          *      <li><b>primaryKey</b> : column in the current table that <b>key</b> option references.
288          *             Defaults to primary key of the current table. Can use an array of strings for a
289          *             composite key association.</li>
290          * </ul>
291          *
292          * ManyToMany additional options:
293          *   <ul>
294          *      <li><b>joinTable</b> : name of table that includes the foreign keys to both
295          *          the current model and the associated model.  Defaults to the name
296          *          of current model and name of associated model, pluralized,
297          *          sorted, and joined with '' and camelized.
298          *      <li><b>leftKey</b> : foreign key in join table that points to current model's
299          *          primary key. Defaults to :"{tableName}Id".
300          *          Can use an array of strings for a composite key association.
301          *      <li><b>leftPrimaryKey</b> - column in current table that <b>leftKey</b> points to.
302          *          Defaults to primary key of current table.  Can use an array of strings for a
303          *          composite key association.
304          *      <li><b>rightKey</b> : foreign key in join table that points to associated
305          *          model's primary key.  Defaults to Defaults to :"{associatedTableName}Id".
306          *          Can use an array of strings for a composite key association.
307          *      <li><b>rightPrimaryKey</b> : column in associated table that <b>rightKey</b> points to.
308          *          Defaults to primary key of the associated table.  Can use an
309          *          array of strings for a composite key association.
310          *  </ul>
311          * @param {Function} [filter] optional function to filter the dataset after all other options have been applied.
312          *
313          */
314         associate:function (type, name, options, filter) {
315             if (comb.isFunction(options)) {
316                 filter = options;
317                 options = {};
318             }
319             this.__associations = this.__associations || {manyToMany:{}, oneToMany:{}, manyToOne:{}, oneToOne:{}};
320             var assoc = new associations[type](comb.merge({model:name}, options), this.patio, filter);
321             assoc.inject(this, name);
322             this.__associations[type][name] = assoc;
323             return this;
324         },

326         __isReciprocalAssociation:function (assoc, pAssoc) {
327             var keys = assoc._getAssociationKey(), leftKey = keys[0], rightKey = keys[1];
328             var pKeys = pAssoc._getAssociationKey(), pLeftKey = pKeys[0], pRightKey = pKeys[1];
329             return leftKey.every(function (k, i) {
330                 return pRightKey[i] == k
331             }) && rightKey.every(function (k, i) {
332                 return pLeftKey[i] == k
333             }) && assoc.parent == pAssoc.model;
334         },

336         _findAssociation:function (assoc) {
337             var type = assoc.type, recipTypes = RECIPROCAL_ASSOC[type];
338             for (var i in recipTypes) {
339                 var recipType = recipTypes[i];
340                 var potentialAssociations = this.__associations[recipType];
341                 var ret = null, found = false;
342                 for (var i in potentialAssociations) {
343                     var pAssoc = potentialAssociations[i];
344                     if (this.__isReciprocalAssociation(assoc, pAssoc)) {
345                         ret = [i, pAssoc], found = true;
346                         break;
347                     }
348                 }
349                 if (found) {
350                     break;
351                 }
352 
353             }
354             return ret;
355         },

370         _findAssociationsForType:function (type, clazz) {
371             var associations = this.__associations[type], ret = [];
372             for (var i in associations) {
373                 var assoc = associations[i];
374                 if (assoc.model == clazz) {
375                     ret.push(assoc);
376                 }
377             }
378             return ret;
