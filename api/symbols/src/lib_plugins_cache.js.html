1 var comb = require("comb"),
  2     hitch = comb.hitch,
  3     Promise = comb.Promise,
  4     PromiseList = comb.PromiseList,
  5     Hive = require("hive-cache");
  6 
  7 
  8 var hive;
  9 var i = 0;
 10 
 11 var LOGGER = comb.logging.Logger.getLogger("patio.plugins.CachePlugin");
 12 /**
 13  * @class Adds in memory caching support for models.
 14  *
 15  * @example
 16  *
 17  * var MyModel = patio.addModel("testTable", {
 18  *     plugins : [patio.plugins.CachePlugin];
 19  * });
 20  *
 21  * //NOW IT WILL CACHE
 22  *
 23  * @name CachePlugin
 24  * @memberOf patio.plugins
 25  */
 26 exports.CachePlugin = comb.define(null, {
 27     instance:{
 28 
 29         constructor:function () {
 30             if (!hive) hive = new Hive();
 31             this._super(arguments);
 32             this.post("load", this._postLoad);
 33         },
 34 
 35         reload:function () {
 36             var ret = new Promise();
 37             this._super(arguments).then(hitch(this, function (m) {
 38                 hive.replace((this.tableName.toString() + this.primaryKeyValue.toString()), m);
 39                 ret.callback(m);
 40             }), hitch(ret, "errback"));
 41             return ret;
 42         },
 43 
 44         _postLoad:function (next) {
 45             hive.replace(this.tableName + this.primaryKeyValue, this);
 46             next();
 47         },
 48 
 49         update:function (options, errback) {
 50             var ret = new Promise();
 51             this._super(arguments).then(hitch(this, function (val) {
 52                 hive.remove(this.tableName + this.primaryKeyValue, val);
 53                 ret.callback(val);
 54             }), hitch(ret, "errback"));
 55             return ret;
 56         },
 57 
 58         remove:function (errback) {
 59             hive.remove(this.primaryKeyValue);
 60             var ret = this._super(arguments);
 61             return ret;
 62         }
 63     },
 64 
 65     static:{
 66 
 67         findById:function (id) {
 68             var cached = hive.get(this.tableName + id);
 69             if (!cached) {
 70                 return this._super(arguments);
 71             } else {
 72                 var ret = new Promise();
 73                 ret.callback(cached);
 74                 return ret;
 75             }
 76         }
 77     }
 78 });
