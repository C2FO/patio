265                 }
266                 var raiseOnError = clazz.raiseOnTypecastError;
267                 if (raiseOnError === true && comb.isUndefinedOrNull(value) && colSchema.allowNull === false) {

270                 try {
271                     value = clazz.db.typecastValue(type, value);
272                 } catch (e) {

513         _checkTransaction:function (opts, cb) {
514             if (comb.isFunction(opts)) {
515                 cb = opts;
516                 opts = {};
517             } else {
518                 opts = opts || {};
519             }
520             var ret = new comb.Promise(), retVal = null, errored = false;
521             if (this.useTransaction(opts)) {
522                 this.db.transaction(opts, hitch(this, function () {
523                     return comb.when(cb(), function () {
524                         retVal = comb.argsToArray(arguments);
525                     }, function () {
526                         retVal = comb.argsToArray(arguments);
527                         errored = true;
528                     });
529                 })).then(hitch(this, function () {
530                     ret[errored ? "errback" : "callback"].apply(ret, retVal);
531                 }), hitch(this, function () {
532                     if(errored){
533                         ret.errback.apply(ret, retVal);
534                     }else{
535                         ret.errback.apply(ret, arguments);
536                     }
537                 }));
538                 return ret;
539             } else {
540                 return comb.when(cb(), hitch(ret, "callback"), hitch(ret, "errback"));
541             }

758 var checkAndAddDBToTable = function (db, table) {
759     if (!table.contains(db)) {
760         table.set(db, {});
761         var handle = comb.connect(db, "onDisconnect", function () {
762             comb.disconnect(handle);
763             MODELS.remove(db);
764         });
765     }
766 };

769 exports.create = function (name, supers, modelOptions) {
770     !patio && (patio = require("./index"));
771     var db, ds, tableName, ret = new comb.Promise()
772     if (comb.isString(name)) {
773         tableName = name;
774         db = patio.defaultDatabase;
775         ds = db.from(tableName);
776     } else if (comb.isInstanceOf(name, patio.Dataset)) {
777         ds = name;
778         tableName = ds.firstSourceAlias;
779         db = ds.db;
780     }
781     checkAndAddDBToTable(db, MODELS);
782     checkAndAddDBToTable(db, DEFERRED_MODELS);
783     var hasSuper = false, deffer = false;
784     if (comb.isHash(supers) || comb.isUndefinedOrNull(supers)) {
785         modelOptions = supers;
786         supers = [Model];
787     } else if (comb.isString(supers)) {
788         //if this happens then assume its the name of a table that is deferred.
789         //check if it is loaded
790         if ((model = MODELS.get(db)[checkAndTransformName(supers)]) != null) {
791             supers = [model]
792         } else {
793             //its not loaded check if it is deferred
794             if ((model = DEFERRED_MODELS.get(db)[checkAndTransformName(supers)]) != null) {
795                 DEFERRED_MODELS.get(db)[tableName] = ret;
796                 model.then(function (SuperModel) {
797                     DEFERRED_MODELS.get(db)[tableName] = null;
798                     exports.create(name, SuperModel, modelOptions).then(hitch(ret, "callback"), hitch(ret, "errback"));
799                 }, hitch(ret, "errback"));
800             } else {
801                 //couldnt find it thrown an error!
802                 return ret.errback(new ModelError("Invalid super class " + supers));
803             }
804             return ret;
805         }
806     } else {
807         hasSuper = true;
808     }
809     supers = comb.array.toArray(supers);
810     var model;
811 
812     if ((model = MODELS.get(db)[checkAndTransformName(tableName.toString())]) != null) {
813         ret.callback(model);
814     } else if ((model = DEFERRED_MODELS.get(db)[checkAndTransformName(tableName.toString())]) != null) {
815         ret = model;
816     } else {
817         DEFERRED_MODELS.get(db)[tableName] = ret;
818         var DEFAULT_PROTO = {instance:{}, static:{}};
819         modelOptions = merge(DEFAULT_PROTO, modelOptions || {});
820         modelOptions.instance._hooks = ["save", "update", "remove", "load"];
821         modelOptions.instance.__hooks = {pre:{}, post:{}};
822         //Mixin the column setter/getters
823         db.schema(tableName).then(function (schema) {
824             modelOptions.static.__schema = schema;
825             modelOptions.static.__tableName = tableName;
826             modelOptions.static.__db = db;
827             model = comb.define(supers.concat(modelOptions.plugins || []), modelOptions);
828             if (hasSuper) {
829                 supers.forEach(hitch(model, "inherits"));
830             }
831             ["pre", "post"].forEach(function (op) {
832                 var optionsOp = modelOptions[op];
833                 if (optionsOp) {
834                     for (var i in optionsOp) {
835                         model[op](i, optionsOp[i]);
836                     }
837                 }
838             });
839             MODELS.get(db)[tableName] = model;
840             //clear it out
841             DEFERRED_MODELS.get(db)[tableName] = null;
842             ret.callback(model);
843         }, comb.hitch(ret, "errback"));
844     }
845     return ret;
846 };

848 var checkAndTransformName = function (name) {
849     return applyColumnTransformMethod(name, Model.camelize === true ? "camelize" : Model.underscore === true ? "underscore" : null);

852 exports.getModel = function (name, db) {
853     if (!comb.isInstanceOf(name, Model)) {
854         !patio && (patio = require("./index"));
855         db = db || patio.defaultDatabase;
856         var ret = null;
857         if (MODELS.contains(db)) {
858             ret = MODELS.get(db)[checkAndTransformName(name)];
859         }
860     } else {
861         ret = name;
862     }
863     return ret;
864 };
