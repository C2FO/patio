 43     /**
 44      * Adds the given graph aliases to the list of graph aliases to use,
 45      * unlike {@link patio.Dataset#setGraphAliases}, which replaces the list (the equivalent
 46      * of {@link patio.Dataset#selectMore} when graphing).  See {@link patio.Dataset#setGraphAliases}.
 47      *
 48      * @example
 49      *  var DB = patio.defaultDatabase;
 50      *  // SELECT ..., table.column AS someAlias
 51      *   DB.from("table").addGraphAliases({someAlias : ["table", "column"]);
 52      *   //returns from graphing
 53      *   // => {table : {column : someAlias_value, ...}, ...}
 54      *
 55      *   @param {Object} graphAliases the graph aliases to use.
 56      *          Where key is the alias name and the value is an array where arr[0] = 'tableName' arr[1] = "colName'.
 57      *
 58      *   @return {patio.Dataset} deep copy of the original dataset with the added graphAliases.
 59      */
 60     addGraphAliases:function(graphAliases) {
 61       var ds = this.selectMore.apply(this, this.__graphAliasColumns(graphAliases));
 62       ds.__opts.graphAliases = comb.merge((ds.__opts.graphAliases || (ds.__opts.graph ? ds.__opts.graph.columnAliases : {}) || {}), graphAliases);
 63       return ds;
 64     },

 66     /**
 67      * Allows you to join multiple datasets/tables and have the result set
 68      * split into component tables.
 69      *
 70      * This differs from the usual usage of join, which returns the result set
 71      * as a single hash.
 72      *
 73      * <pre class="code">
 74      *
 75      * //CREATE TABLE artists (id INTEGER, name TEXT);
 76      * //CREATE TABLE albums (id INTEGER, name TEXT, artist_id INTEGER);
 77      *
 78      * var DB = patio.defaultDatabase, ds = db.from("artists");
 79      * ds.leftAuterJoin("albums", {artistId : "id"}).first
 80      *   //=> {id : albums.id, name : albums.name, artist_id : albums.artist_id}
 81      *
 82      * var p = comb.executeInOrder(ds, function(ds){
 83      *   var graphedDs = ds.graph("albums", {artist_id : "id"});
 84      *   return graphedDs.first();
 85      * });
 86      * p.then(function(obj){
 87      *    //obj == {artists : {
 88      *                  id : artists.id,
 89      *                  name : artists.name
 90      *                  },
 91      *             albums : {
 92      *                  id : albums.id,
 93      *                  name : albums.name,
 94      *                  artist_id=>albums.artist_id
 95      *                  }
 96      *             }
 97      * });
 98      * </pre>
 99      *
100      * Using a join such as leftOuterJoin, the attribute names that are shared between
101      * the tables are combined in the single return hash.  You can get around that by
102      * using {@link patio.Dataset#select} with correct aliases for all of the columns, but it is simpler to
103      * use {@link patio.Dataset#graph} and have the result set split for you.  In addition, {@link patio.Dataset#graph} respects
104      * any {@link patio.Dataset#rowCb} of the current dataset and the datasets you use with {@link patio.Dataset#graph}.
105      *
106      * If you are graphing a table and all columns for that table are null, this
107      * indicates that no matching rows existed in the table, so graph will return null.
108      * instead of a hash with all nil values:
109      *
110      * <pre class="code">
111      *   // Psuedo code there will be promises returned
112      *   /// If the artist doesn't have any albums
113      *
114      *   var DB = patio.defaultDatabase, ds = db.from("artists");
115      *   var obj = ds.graph(:albums, :artist_id=>:id).first()
116      *   //obj == { artists : {id : artists.id, name : artists.name}, albums : null};
117      * </pre>
118      *
119      *
120      * @parameter {String|patio.Dataset|Object} dataset This can be a string (representing a table), another {@link patio.Dataset},
121      *            or an object that has a dataset property that returns a string or dataset.
122      * @parameter joinConditions Any condition(s) allowed by {@link patio.Dataset#joinTable}.
123      *
124      * @parameter {Object} [options] options to use when creating the graph
125      *
126      * @parameter {String|sql.LiteralString|sql.Identifier} [options.fromSelfAlias] The alias to use when the
127      *                      receiver is not a graphed dataset but it contains multiple FROM tables or a JOIN.
128      *                      In this case, the receiver is wrapped in a {@link patio.Dataset#fromSelf} before graphing,
129      *                      and this option determines the alias to use.
130      * @parameter {String|sql.LiteralString|sql.Identifier} [options.implicitQualifier] The qualifier of implicit conditions,
131      *              see {@link patio.Dataset#joinTable}.
132      * @parameter [String] [options.joinType="leftOuter"] The type of join to use (passed to {@link patio.Dataset#joinTable}.).
133      * @parameter [String[]|sql.LiteralString[]|sql.Identifier[]|Boolean] [options.select] An array of columns to select.  When not used, selects
134      *            all columns in the given dataset.  When set to false, selects no
135      *            columns and is like simply joining the tables, though graph keeps
136      *            some metadata about the join that makes it important to use {@link patio.Dataset#graph} instead
137      *            of {@link patio.Dataset#joinTable}
138      * @parameter {String|sql.LiteralString|sql.Identifier} [options.tableAlias] The alias to use for the table.  If not specified, doesn't
139      *                 alias the table.  You will get an error if the the alias (or table) name is
140      *                 used more than once.
141 
142      * @parameter {Function} block  A function that is passed to {@link patio.Dataset#joinTable}.
143      **/
144     graph:function(dataset, joinConditions, options, block) {
145       var ret = new comb.Promise();
146       var args = comb.argsToArray(arguments).slice(1);
147       block = comb.isFunction(args[args.length - 1]) ? args.pop() : null;
148       joinConditions = args.shift() || null;
149       options = args.shift() || {};
150       // Allow the use of a model, dataset, or symbol as the first argument
151       // Find the table name/dataset based on the argument
152       dataset.hasOwnProperty("dataset") && (dataset = dataset.dataset);
153       var tableAlias = options.tableAlias, table;
154       if (comb.isString(dataset)) {
155         table = sql.identifier(dataset);
156         dataset = this.db.from(dataset);
157         comb.isUndefinedOrNull(tableAlias) && (tableAlias = table);
158       } else if (comb.isInstanceOf(dataset, Dataset)) {
159         if (dataset.isSimpleSelectAll) {
160           table = dataset.__opts.from[0];
161           comb.isUndefinedOrNull(tableAlias) && (tableAlias = table);
162         } else {
163           table = dataset;
164           comb.isUndefinedOrNull(tableAlias) && (tableAlias = this._datasetAlias((this.__opts.numDatasetSources || 0) + 1));
165         }
166       } else {
167         throw new QueryError("The dataset arg should be a string, dataset or model");
168       }
169 
170       var aliases;
171       // Only allow table aliases that haven't been used
172       var thisOpts = this.__opts, thisOptsGraph = thisOpts.graph;
173       if (comb.isObject(thisOptsGraph) && comb.isHash((aliases = thisOptsGraph.tableAliases)) && !comb.isUndefinedOrNull(aliases[tableAlias.value])) {
174         raiseAliasError(options);
175       }
176       // Use a from_self if this is already a joined table
177       var ds = (!thisOptsGraph && (thisOpts.from.length > 1 || thisOpts.join)) ? this.fromSelf({alias:options.fromSelfAlias || this.firstSourceAlias}) : this;
178 
179       // Join the table early in order to avoid cloning the dataset twice
180       ds = ds.joinTable(options.joinType || "leftOuter", table, joinConditions, {tableAlias:tableAlias, implicitQualifier:options.implicitQualifier}, block);
181       var opts = ds.__opts;
182 
183       // Whether to include the table in the result set
184       var addTable = comb.isBoolean(options.select) ? options.select : true;
185       // Whether to add the columns to the list of column aliases
186       var addColumns = comb.isUndefinedOrNull(opts.graphAliases);
187       // Setup the initial graph data structure if it doesn't exist
188       var graph;
189       var populateGraphPromise = new comb.Promise();
190       if (comb.isUndefinedOrNull((graph = opts.graph))) {
191         var master = this._toTableName(ds.firstSourceAlias);
192         (master == tableAlias) && raiseAliasError(options);
193         // Master hash storing all .graph related information
194         graph = opts.graph = {};
195         // Associates column aliases back to tables and columns
196         var columnAliases = graph.columnAliases = {};
197         // Associates table alias (the master is never aliased)
198         var tableAliases = graph.tableAliases = {};
199         tableAliases[master] = this;
200 
201         // All columns in the master table are never
202         // aliased, but are not included if set_graph_aliases
203         // has been used.
204         if (addColumns) {
205           var select = opts.select = [];
206           this.columns.then(hitch(this, function(cols) {
207             cols.forEach(function(column) {
208               columnAliases[column] = [master, column];
209               select.push(new sql.QualifiedIdentifier(master, column));
210             });
211             populateGraphPromise.callback(graph);
212           }), hitch(ret, "errback"));
213         } else {
214           populateGraphPromise.callback(graph);
215         }
216       } else {
217         populateGraphPromise.callback(graph);
218       }
219       populateGraphPromise.then(hitch(this, function(graph) {
220         // Add the table alias to the list of aliases
221         // Even if it isn't been used in the result set,
222         // we add a key for it with a nil value so we can check if it
223         // is used more than once
224         var tableAliases = graph.tableAliases;
225         tableAliases[tableAlias] = addTable ? dataset : null;
226 
227         // Add the columns to the selection unless we are ignoring them
228         if (addTable && addColumns) {
229           var select = opts.select;
230           var columnAliases = graph.columnAliases;
231 
232           // Which columns to add to the result set
233           var dsColPromise;
234           if (options.select) {
235             dsColPromise = new comb.Promise().callback(options.select)
236           } else {
237             dsColPromise = dataset.columns;
238           }
239           // If the column hasn't been used yet, don't alias it.
240           // If it has been used, try tableColumn.
241           dsColPromise.then(hitch(this, function(cols) {
242             cols.forEach(function(column) {
243               var colAlias, identifier;
244               if (columnAliases[column]) {
245                 var columnAlias = format("%s_%s", [tableAlias, column]);
246                 colAlias = columnAlias, identifier = new sql.QualifiedIdentifier(tableAlias, column).as(columnAlias);
247               } else {
248                 colAlias = column, identifier = new sql.QualifiedIdentifier(tableAlias, column);
249               }
250               columnAliases[colAlias] = [tableAlias, column];
251               select.push(identifier)
252             });
253             ret.callback(ds);
254           }), hitch(ret, "errback"));
255         } else {
256           ret.callback(ds);
257         }
258       }), hitch(ret, "errback"));
259       return ret;
260     },

263     /**
264      * This allows you to manually specify the graph aliases to use
265      * when using graph.  You can use it to only select certain
266      * columns, and have those columns mapped to specific aliases
267      * in the result set.  This is the equivalent of {@link patio.Dataset#select} for a
268      * graphed dataset, and must be used instead of {@link patio.Dataset#select} whenever
269      * graphing is used.
270      *
271      * @example
272      *
273      * var DB = patio.defaultDatabase, ds = DB.from("artists");
274      * var p = comb.executeInOrder(ds, function(ds){
275      *     var graphedDs = ds.graph("albums", {artist_id : id});
276      *     //SELECT artists.name AS artist_name, albums.name AS album_name, 42 AS forty_two FROM table
277      *     return graphedDs.setGraphAliases({artist_name : ["artists", "name"],
278      *                       album_name : ["albums", "name"],
279      *                       forty_two : ["albums", "fourtwo", 42]).first();
280      * });
281      *
282      * p.then(function(obj){
283      *    //obj == {artists : {name : artists.name}, albums : {name : albums.name, fourtwo : 42}}
284      * });
285      *
286      * @parameter {Object} graphAliases Should be a hash with keys being  column aliases, and values being
287      *                  arrays with two or three elements. The first element of the array should be the table alias,
288      *                  and the second should be the actual column name symbol. If the array
289      *                  has a third element, it is used as the value returned, instead of
290      *                  tableAlias.columnName.
291      *
292      **/
293     setGraphAliases:function(graphAliases) {
294       var ds = this.select.apply(this, this.__graphAliasColumns(graphAliases));
295       ds.__opts.graphAliases = graphAliases;
296       return ds;
297     },

308     //Transform the hash of graph aliases to an array of columns
309     __graphAliasColumns:function(graphAliases) {
310       var ret = [];
311       if (comb.isArray(graphAliases)) {
312         var newGraphAliases = {};
313         for (var i in graphAliases) {
314           newGraphAliases[graphAliases[i][0]] = graphAliases[i][1];
315         }
316         graphAliases = newGraphAliases;
317       }
318       for (var colAlias in graphAliases) {
319         var tc = graphAliases[colAlias];
320         var identifier = tc[2] || new sql.QualifiedIdentifier(tc[0], tc[1]);
321         if (tc[2] || tc[1] != colAlias) {
322           identifier = new sql.AliasedExpression(identifier, colAlias);
323         }
324         ret.push(identifier);
325       }
326       return ret;
327     },

330     /**
331      * Fetch the rows, split them into component table parts,
332      * transform and run the {@link patio.Dataset#rowCb} on each part (if applicable),
333      * and yield a hash of the parts.
334      * */
335     graphEach:function(cb) {
336       // Reject tables with nil datasets, as they are excluded from
337       // the result set
338       var datasets = comb.array.toArray(this.__opts.graph.tableAliases).filter(function(e) {
339         return !comb.isUndefinedOrNull(e[1])
340       });
341       // Get just the list of table aliases into a local variable, for speed
342       var tableAliases = datasets.map(function(e) {
343         return e[0]
344       });
345 
346       datasets = datasets.map(function(e) {
347         return [e[0], e[1], e[1].rowCb]
348       });
349       // Use the manually set graph aliases, if any, otherwise
350       // use the ones automatically created by .graph
351       var columnAliases = this.__opts.graphAliases || this.__opts.graph.columnAliases;
352       var ret = new comb.Promise();
353       return this.fetchRows(this.selectSql, function(r) {
354         var graph = {};
355         // Create the sub hashes, one per table
356         tableAliases.forEach(function(ta) {
357           graph[ta] = {};
358         });
359         // Split the result set based on the column aliases
360         // If there are columns in the result set that are
361         // not in column_aliases, they are ignored
362         for (var colAlias in columnAliases) {
363           var tc = columnAliases[colAlias];
364           var ta = tc[0], column = tc[1];
365           !graph[ta] && (graph[ta] = {});
366           graph[ta][column] = r[colAlias];
367         }
368         datasets.forEach(function(d) {
369           var ta = d[0], ds = d[1], dsCb = d[2];
370           var g = graph[ta];
371           if (!comb.isEmpty(g) && Object.keys(g).some(function(x) {
372             return !comb.isUndefinedOrNull(g[x]);
373           })) {
374             graph[ta] = dsCb ? dsCb(g) : g;
375           } else {
376             graph[ta] = null;
377           }
378         });
379         cb(graph);
380       });
381     }
