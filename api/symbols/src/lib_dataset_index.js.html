270         /**
271          * Converts a string to an {@link patio.sql.Identifier}, {@link patio.sql.QualifiedIdentifier},
272          * or {@link patio.sql.AliasedExpression}, depending on the format:
273          *
274          * <ul>
275          *      <li>For columns : table__column___alias.</li>
276          *      <li>For tables : schema__table___alias.</li>
277          * </ul>
278          * each portion of the identifier is optional. See example below
279          *
280          * @example
281          *
282          * ds.stringToIdentifier("a") //= > new patio.sql.Identifier("a");
283          * ds.stringToIdentifier("table__column"); //=> new patio.sql.QualifiedIdentifier(table, column);
284          * ds.stringToIdentifier("table__column___alias");
285          *      //=> new patio.sql.AliasedExpression(new patio.sql.QualifiedIdentifier(table, column), alias);
286          *
287          * @param {String} name the name to covert to an an {@link patio.sql.Identifier}, {@link patio.sql.QualifiedIdentifier},
288          * or {@link patio.sql.AliasedExpression}.
289          *
290          * @return  {patio.sql.Identifier|patio.sql.QualifiedIdentifier|patio.sql.AliasedExpression} an identifier generated based on the name string.
291          */
292         stringToIdentifier:function(name){
293             if (comb.isString(name)) {
294                 var parts = this._splitString(name);
295                 var schema = parts[0], table = parts[1], alias = parts[2];
296                 return (schema && table && alias
297                     ? new AliasedExpression(new QualifiedIdentifier(schema, table), alias)
298                     : (schema && table
299                     ? new QualifiedIdentifier(schema, table)
300                     : (table && alias
301                     ? new AliasedExpression(new Identifier(table), alias) : new Identifier(table))));
302             } else {
303                 return name;
304             }
305         },

307         /**
308          * Can either be a string or null.
309          *
310          *
311          * @example
312          * //columns
313          *  table__column___alias //=> table.column as alias
314          *  table__column //=> table.column
315          *  //tables
316          *  schema__table___alias //=> schema.table as alias
317          *  schema__table //=> schema.table
318          *
319          * //name and alias
320          * columnOrTable___alias //=> columnOrTable as alias
321          *
322          *
323          *
324          * @return {String[]} an array with the elements being:
325          * <ul>
326          *      <li>For columns :[table, column, alias].</li>
327          *      <li>For tables : [schema, table, alias].</li>
328          * </ul>
329          */
330         _splitString:function(s){
331             var ret, m;
332             if ((m = s.match(this._static.COLUMN_REF_RE1)) != null) {
333                 ret = m.slice(1);
334             }
335             else if ((m = s.match(this._static.COLUMN_REF_RE2)) != null) {
336                 ret = [null, m[1], m[2]];
337             }
338             else if ((m = s.match(this._static.COLUMN_REF_RE3)) != null) {
339                 ret = [m[1], m[2], null];
340             }
341             else {
342                 ret = [null, s, null];
343             }
344             return ret;
345         },
