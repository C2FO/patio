 75     /**
 76      * Add a column with the given name, type, and opts to the DDL.
 77      *
 78      * <pre class="code">
 79      *  DB.createTable("test", function(){
 80      *      this.column("num", "integer");
 81      *          //=> num INTEGER
 82      *      this.column('name", String, {allowNull : false, "default" : "a");
 83      *          //=> name varchar(255) NOT NULL DEFAULT 'a'
 84      *      this.column("ip", "inet");
 85      *          //=> ip inet
 86      *   });
 87      * </pre>
 88      *
 89      * You can also create columns via method missing, so the following are
 90      * equivalent:
 91      * <pre class="code">
 92      * DB.createTable("test", function(){
 93      *   this.column("number", "integer");
 94      *   this.number("integer");
 95      * });
 96      * </pre>
 97      *
 98      * @param {String|patio.sql.Identifier} name the name of the column
 99      * @param type the datatype of the column.
100      * @param {Object} [opts] additional options
101      *
102      * @param [opts.default] The default value for the column.
103      * @param [opts.deferrable] This ensure Referential Integrity will work even if
104      *                reference table will use for its foreign key a value that does not
105      *                exists(yet) on referenced table. Basically it adds
106      *                DEFERRABLE INITIALLY DEFERRED on key creation.
107      * @param {Boolean} [opts.index]  Create an index on this column.
108      * @param {String|patio.sql.Identifier} [key] For foreign key columns, the column in the associated table
109      *         that this column references.  Unnecessary if this column references the primary key of the
110      *         associated table.
111      * @param {Boolean} [opts.allowNull]  Mark the column as allowing NULL values (if true),
112      *          or not allowing NULL values (if false).  If unspecified, will default
113      *          to whatever the database default is.
114      * @param {String} [opts.onDelete] Specify the behavior of this column when being deleted
115      *               ("restrict", "cascade", "setNull", "setDefault", "noAction").
116      * @param {String} [opts.onUpdate] Specify the behavior of this column when being updated
117      *               Valid options ("restrict", "cascade", "setNull", "setDefault", "noAction").
118      * @param {Boolean} [opts.primaryKey] Make the column as a single primary key column.  This should only
119      *                 be used if you have a single, non-autoincrementing primary key column.
120      * @param {Number} [opts.size] The size of the column, generally used with string
121      *          columns to specify the maximum number of characters the column will hold.
122      *          An array of two integers can be provided to set the size and the
123      *          precision, respectively, of decimal columns.
124      * @param {String} [opts.unique] Mark the column as unique, generally has the same effect as
125      *            creating a unique index on the column.
126      * @param {String} [opts.unsigned] Make the column type unsigned, only useful for integer
127      *              columns.
128      * @param {Array} [opts.elements] Available items used for set and enum columns.
129      *

136       }

320 exports.SchemaGenerator = function(db, block) {
321   var gen = new Generator(db);
322   var prox = comb.methodMissing(gen, function(name) {
323     return function(type, opts) {
324       name == name || null, opts = opts || {};
325       if (name) {
326         return this.column(name, type, opts);
327       } else {
328         throw new TypeError();
329       }
330     }
331   }, Generator);
332   block.apply(prox, [prox]);
333   gen.columns = prox.columns;
334   if (gen.__primaryKey && !gen.hasColumn(gen.primaryKeyName)) {
335     gen.columns.unshift(gen.__primaryKey);
336   }
337   return gen;
338 }
