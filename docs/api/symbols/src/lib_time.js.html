479      * Converts a datetime date string to a {@link sql.DateTime}
480      *
481      * @example
482      *
483      *  var dateTime = new sql.DateTime(2004, 1, 1, 12, 12, 12),
484      * offset = getTimeZoneOffset();
485      * patio.stringToDateTime('2004-02-01 12:12:12'); //=> dateTime
486      *
487      * patio.dateTimeFormat = patio.DATETIME_TWO_YEAR_FORMAT;
488      * patio.stringToDateTime('04-02-01 12:12:12-0600'); //=> dateTime
489      *
490      * patio.dateTimeFormat = patio.DATETIME_FORMAT_TZ;
491      * patio.stringToDateTime('2004-02-01 12:12:12-0600'); //=> dateTime
492      *
493      * patio.dateTimeFormat = patio.ISO_8601;
494      * patio.stringToDateTime('2004-02-01T12:12:12-0600'); //=> dateTime
495      *
496      * patio.dateTimeFormat = patio.ISO_8601_TWO_YEAR;
497      * patio.stringToDateTime('04-02-01T12:12:12-0600'); //=> dateTime
498      *
499      * @param {String} dt the string to convert to a {@link sql.DateTime}
500      * @param {String} [format=time.dateTimeFormat] the format to use when converting the date.
501      *
502      * @throws {PatioError} thrown if the conversion fails.
503      * @return {sql.DateTime} the {@link sql.DateTime}
504      */
505     stringToDateTime:function(dt, fmt) {
506       var useT = dt.indexOf("T") != -1;
507       //test if there is a T in the string so we can try to properly convert it
508       var format = fmt ? fmt : useT ? this.ISO_8601 : this.DEFAULT_DATETIME_FORMAT;
509       var ret = date.parse(dt, format);
510       //if the coversion failed try it with a time zone
511       !ret && (ret = date.parse(dt, this.DATETIME_FORMAT_TZ));
512       if (!ret && this.convertTwoDigitYears) {
513         //if we still fail and we need to convert two digit years try the twoYearFormat
514         var twoYearFormat = fmt ? fmt : useT ? this.ISO_8601_TWO_YEAR : this.DATETIME_TWO_YEAR_FORMAT;
515         ret = date.parse(dt, twoYearFormat);
516         //try with time zone
517         !ret && (ret = date.parse(dt, twoYearFormat + "Z"));
518       }
519       if (!ret) {
520         throw new PatioError("Unable to convert datetime: " + dt);
521       }
522       return new SQL.DateTime(ret);
523     },
