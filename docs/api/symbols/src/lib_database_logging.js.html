 64         /* Yield to the block, logging any errors at error level to all loggers,
 65          * and all other queries with the duration at warn or info level.
 66          * */
 67         __logAndExecute : function(sql, args, cb) {
 68             if (comb.isFunction(args)) {
 69                 cb = args;
 70                 args = null;
 71             }
 72             if (args) {
 73                 sql = format("%s; %j", sql, args);
 74             }
 75             var start = new Date();
 76             var ret = new comb.Promise();
 77             if (comb.isFunction(cb)) {
 78                 this.logDebug(format("Executing: %s", sql.trim()));
 79                 cb().then(comb.hitch(this, function() {
 80                     this.logInfo(format("Duration: % 4dms; %s", new Date() - start, sql.trim()));
 81                     ret.callback.apply(ret, arguments);
 82                 }), comb.hitch(this, function(err) {
 83                     var err = new QueryError(format("%s: %s", err.message, sql.trim()));
 84                     this.logError(err);
 85                     ret.errback.apply(ret, [err]);
 86                 }));
 87             } else {
 88                 throw "CB is required";
 89             }
 90             return ret;
 91         },
