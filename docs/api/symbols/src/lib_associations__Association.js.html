 76             this._model = options.model;

 79             !comb.isUndefined(options.isCascading) && (this.isCascading = options.isCascading);

131         _getAssociationKey:function () {
132             var options = this.__opts, key, ret = [], lk, rk;
133             if (!comb.isUndefinedOrNull((key = options.key))) {
134                 if (this.supportsStringKey && comb.isString(key)) {
135                     //normalize the key first!
136                     ret = [
137                         [this.isOwner ? this.defaultLeftKey : key],
138                         [this.isOwner ? key : this.defaultRightKey]
139                     ];
140                 } else if (this.supportsHashKey && comb.isHash(key)) {
141                     var leftKey = Object.keys(key)[0];
142                     var rightKey = key[leftKey];
143                     ret = [
144                         [leftKey],
145                         [rightKey]
146                     ];
147                 } else if (this.supportsCompositeKey && comb.isArray(key)) {
148                     ret = [
149                         [key],
150                         null
151                     ];
152                 }
153             } else if (this.supportsLeftAndRightKey && (!comb.isUndefinedOrNull((lk = options.leftKey)) && !comb.isUndefinedOrNull((rk = options.rightKey)))) {
154                 ret = [
155                     comb.array.toArray(lk),
156                     comb.array.toArray(rk)
157                 ];
158             } else {
159                 //todo handle composite primary keys
160                 ret = [
161                     [this.defaultLeftKey],
162                     [this.defaultRightKey]
163                 ];
164             }
165             return ret;

169         _setAssociationKeys:function (parent, model, val) {
170             var keys = this._getAssociationKey(parent), leftKey = keys[0], rightKey = keys[1];
171             if (leftKey && rightKey) {
172                 for (var i = 0; i < leftKey.length; i++) {
173                     model[rightKey[i]] = !comb.isUndefined(val) ? val : parent[leftKey[i]];
174                 }
175             } else {
176                 leftKey.forEach(function (k) {
177                     model[k] = !comb.isUndefined(val) ? val : parent[k];
178                 });
179             }

182         _setDatasetOptions:function (ds) {
183             var options = this.__opts || {};
184             var order, limit, distinct, select, query;
185             //allow for multi key ordering
186             if (!comb.isUndefined((select = this.select))) {
187                 ds = ds.select.apply(ds, comb.array.toArray(select));
188             }
189             if (!comb.isUndefined((query = options.query)) || !comb.isUndefined((query = options.conditions))) {
190                 ds = ds.filter(query);
191             }
192             if (comb.isFunction(this.filter)) {
193                 var ret = this.filter.apply(this, [ds]);
194                 if (comb.isInstanceOf(ret, ds._static)) {
195                     ds = ret;
196                 }
197             }
198             if (!comb.isUndefined((distinct = options.distinct))) {
199                 ds.limit.apply(ds, comb.array.toArray(distinct));
200             }
201             if (!comb.isUndefined((order = options.orderBy)) || !comb.isUndefined((order = options.order))) {
202                 ds.order.apply(ds, comb.array.toArray(order));
203             }
204             if (!comb.isUndefined((limit = options.limit))) {
205                 ds.limit.apply(ds, comb.array.toArray(limit));
206             }
207             return ds;

235         __loadModel:function (parent, model, setValue) {
236             var ret = new Promise();
237             setValue = comb.isBoolean(setValue) ? setValue : true;
238             if (comb.isArray(model)) {
239                 var pl = new PromiseList(model.map(hitch(this, function (m) {
240                     //dont set the value because we have an array
241                     //and it will be set in the callback
242                     return this.__loadModel(parent, m, false);
243                 })), true);
244                 if (setValue) {
245                     pl.then(hitch(this, function (models) {
246                         //now we can set the value!!!
247                         this.__setValue(parent, models);
248                         ret.callback(models);
249                     }), hitch(ret, "errback"));
250                 } else {
251                     ret = pl;
252                 }
253             } else {
254                 model = this._toModel(model, true);
255                 var recip = this.model._findAssociation(this);
256                 if (recip) {
257                     recip = recip[1];
258                     recip.__setValue(model, parent);
259                 }
260                 setValue && this.__setValue(parent, model);
261                 //call hook to finish other model associations
262                 model._hook("post", "load").then(hitch(this, function () {
263                     ret.callback(model);
264                 }), hitch(ret, "errback"));
265             }
266             return ret;

270         fetch:function (parent) {
271             var ret = new Promise();
272             if (this._checkAssociationKey(parent)) {
273                 this._filter(parent)[this._fetchMethod]().then(hitch(this, function (result) {
274                     var p;
275                     if ((comb.isArray(result) && result.length) || comb.isHash(result)) {
276                         p = this.__loadModel(parent, result);
277                     } else {
278                         this.__setValue(parent, result);
279                         p = new Promise().callback(result);
280                     }
281                     p.then(hitch(ret, "callback"), hitch(ret, "errback"));
282                 }), hitch(ret, "errback"));
283             } else {
284                 this.__setValue(parent, null);
285                 ret.callback(null);
286             }
287             return ret;
