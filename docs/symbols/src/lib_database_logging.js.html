 45         /* Yield to the block, logging any errors at error level to all loggers,
 46          * and all other queries with the duration at warn or info level.
 47          * */
 48         __logAndExecute : function(sql, args, cb) {
 49             if (comb.isFunction(args)) {
 50                 cb = args;
 51                 args = null;
 52             }
 53             if (args) {
 54                 sql = format("%s; %j", sql, args);
 55             }
 56             var start = new Date();
 57             var ret = new comb.Promise();
 58             if (comb.isFunction(cb)) {
 59                 cb().then(comb.hitch(this, function() {
 60                     this.logInfo(format("Duration: % 4dms; %s", new Date() - start, sql.trim()));
 61                     ret.callback.apply(ret, arguments);
 62                 }), comb.hitch(this, function(err) {
 63                     this.logError(format("Error   : %s: %s: %s", err.name, err.message, sql.trim()));
 64                     ret.errback.apply(ret, arguments);
 65                 }));
 66             } else {
 67                 throw "CB is required";
 68             }
 69             return ret;
 70         },
 71 
 72         /*Log the given SQL and then execute it on the connection, used by
 73          *the transaction code.
 74          * */
 75         __logConnectionExecute : function(conn, sql) {
 76             return this.__logAndExecute(sql, comb.hitch(this, function() {
 77                 return conn[this.connectionExecuteMethod](sql);
