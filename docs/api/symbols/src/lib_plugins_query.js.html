  7 /**
  8  *@class
  9  * This plugin adds the query support.
 10  * <ul>
 11  *     <li>filter</li>
 12  *     <li>findById</li>
 13  *     <li>count</li>
 14  *     <li>join</li>
 15  *     <li>where</li>
 16  *     <li>select</li>
 17  *     <li>all</li>
 18  *     <li>forEach</li>
 19  *     <li>first</li>
 20  *     <li>one</li>
 21  *     <li>last</li>
 22  * </ul>
 23  *
 24  * All queries require an action to be called on them before the results are fetched. The action methods are :
 25  *
 26  * <ul>
 27  *     <li>all</li>
 28  *     <li>forEach</li>
 29  *     <li>first</li>
 30  *     <li>one</li>
 31  *     <li>last</li>
 32  *
 33  * </ul>
 34  *
 35  * The action items accept a callback that will be called with the results. They also return a promise that will be
 36  * called with the results.
 37  *
 38  * <p>Assume we have an Employee model.</p>
 39  *
 40  * <p>
 41  *      <b>first</b></br>
 42  *      Get the record in a dataset, this query does not require an action method
 43  *      <pre class="code">
 44  *          Employee.first() => select * from employee limit 1
 45  *      </pre>
 46  * </p>
 47  *
 48  * <p>
 49  *      <b>filter</b></br>
 50  *      Sets the where clause on a query. See {@link Dataset}
 51  *      <pre class="code">
 52  *          //Equality Checks
 53  *          Employee.filter({eid : 1})
 54  *                  => select * from employee where eid = 1
 55  *          Employee.filter({eid : {gt : 1}})
 56  *                  => select * from employee where eid > 1
 57  *          Employee.filter({eid : {gte : 1}})
 58  *                  => select * from employee where eid >= 1
 59  *          Employee.filter({eid : {lt : 1}})
 60  *                  => select * from employee where eid < 1
 61  *          Employee.filter({eid : {lte : 1}})
 62  *                  => select * from employee where eid <= 1
 63  *          //Nested query in filter
 64  *          Employee.filter({eid : {gt : 1}, lastname : "bob"})
 65  *                  => select * from employee where eid > 1 and lastname = 'bob';
 66  *          Employee.filter({eid : [1,2,3], lastname : "bob"})
 67  *                  => select * from employee where eid in (1,2,3) and lastname = 'bob'
 68  *      </pre>
 69  * </p>
 70  * <p>
 71  *      <b>findById</b></br>
 72  *      Find a record in a dataset by id, this query does not require an action method
 73  *      <pre class="code">
 74  *          Employee.findById(1) => select * from employee where eid = 1
 75  *      </pre>
 76  * </p>
 77  * <p>
 78  *      <b>count</b></br>
 79  *      Find the number of records in a dataset, this query does not require an action method
 80  *      <pre class="code">
 81  *          Employee.count() => select count(*) as count from employee
 82  *          Employee.filter({eid : {gte : 1}}).count()
 83  *                  => select count(*) as count from employee  where eid > 1
 84  *      </pre>
 85  * </p>
 86  * <p>
 87  *      <b>join</b></br>
 88  *      Get Join two models together, this will not create model instances for the result.
 89  *      <pre class="code">
 90  *          Employee.join("words", {eid : "eid"}).where({"employee.eid" : 1})
 91  *                  => select * from employee inner join works on employee.id=works.id where employee.eid = 1
 92  *      </pre>
 93  * </p>
 94  *
 95  * <p>
 96  *      <b>Where</b></br>
 97  *      Sets the where clause on a query. See {@link Dataset}
 98  *      <pre class="code">
 99  *          //Equality Checks
100  *          Employee.where({eid : 1})
101  *                  => select * from employee where eid = 1
102  *          Employee.where({eid : {gt : 1}})
103  *                  => select * from employee where eid > 1
104  *          Employee.where({eid : {gte : 1}})
105  *                  => select * from employee where eid >= 1
106  *          Employee.where({eid : {lt : 1}})
107  *                  => select * from employee where eid < 1
108  *          Employee.where({eid : {lte : 1}})
109  *                  => select * from employee where eid <= 1
110  *          //Nested query in filter
111  *          Employee.where({eid : {gt : 1}, lastname : "bob"})
112  *                  => select * from employee where eid > 1 and lastname = 'bob';
113  *          Employee.where({eid : [1,2,3], lastname : "bob"})
114  *                  => select * from employee where eid in (1,2,3) and lastname = 'bob'
115  *      </pre>
116  * </p>
117  *
118  * <p>
119  *      <b>select</b></br>
120  *      Selects only certain columns to return, this will not create model instances for the result.
121  *      <pre class="code">
122  *          Employee.select(eid).where({firstname : { gt : "bob"}})
123  *                  => select eid from employee where firstname > "bob"
124  *      </pre>
125  * </p>
126  *
127  *
128  * <p>
129  *      <b>all, foreach, first, one, last</b></br>
130  *      These methods all act as action methods and fetch the results immediately. Each method accepts a query, callback, and errback.
131  *      The methods return a promise that can be used to listen for results also.
132  *      <pre class="code">
133  *          Employee.all()
134  *                  => select * from employee
135  *          Employee.forEach(function(){})
136  *                  => select * from employee
137  *          Employee.forEach({eid : [1,2,3]}, function(){}))
138  *                  => select * from employee where eid in (1,2,3)
139  *          Employee.one()
140  *                  => select * from employee limit 1
141  *      </pre>
142  * </p>
143  *
144  * @name QueryPlugin
145  * @memberOf patio.plugins
146  *
147  * @borrows Dataset#all as all
148  * @borrows Dataset#forEach as forEach
149  * @borrows Dataset#first as first
150  * @borrows Dataset#one as one
151  * @borrows Dataset#last as last
152  * @borrows SQL#join as join
153  * @borrows SQL#where as where
154  * @borrows SQL#select as select
155  *
156  */

161     _getPrimaryKeyQuery:function() {
162       var q = {};
163       this.primaryKey.forEach(function(k) {
164         q[k] = this[k];
165       }, this);
166       return q;
167     },

169     /**
170      * Force the reload of the data for a particular model instance.
171      *
172      * @example
173      *
174      * myModel.reload().then(function(myModel){
175      *    //work with this instance
176      * });
177      *
178      * @return {comb.Promise} called back with the reloaded model instance.
179      */
180     reload:function() {
181       var ret = new Promise();
182       if (!this.__isNew) {
183         ret = this.dataset.filter(this._getPrimaryKeyQuery()).one();
184       } else {
185         ret.callback(this);
186       }
187       return ret;
188     },

190     /**
191      * Remove this model.
192      *
193      * @param {Function} errback called in the deletion fails.
194      *
195      * @return {comb.Promise} called back after the deletion is successful
196      */
197     remove:function(options) {
198       if (!this.__isNew) {
199         return this._checkTransaction(options, comb.hitch(this, function() {
200           return comb.executeInOrder(this, this.dataset, function(m, ds) {
201             m._hook("pre", "remove");
202             var ret = ds.filter(m._getPrimaryKeyQuery()).remove();
203             m._hook("post", "remove");
204             return ret;
205           });
206         }));
207       } else {
208         return new comb.Promise().callback(0);
209       }
210     },
211 
212     /**
213      * Update a model with new values.
214      *
215      * @example
216      *
217      * someModel.update({
218      *      myVal1 : "newValue1",
219      *      myVal2 : "newValue2",
220      *      myVal3 : "newValue3"
221      *      }).then(..do something);
222      *
223      * //or
224      *
225      * someModel.myVal1 = "newValue1";
226      * someModel.myVal2 = "newValue2";
227      * someModel.myVal3 = "newValue3";
228      *
229      * someModel.update().then(..so something);
230      *
231      * @param {Object} [options] values to update this model with
232      * @param {Function} [errback] function to call if the update fails, the promise will errback also if it fails.
233      *
234      * @return {comb.Promise} called on completion or error of update.
235      */
236     update:function(vals, options) {
237       if (!this.__isNew && this.__isChanged) {
238         return this._checkTransaction(options, comb.hitch(this, function() {
239           comb.isHash(vals) && this.__set(vals);
240           var saveChanged = !comb.isEmpty(this.__changed);
241           return comb.executeInOrder(this, this.dataset.filter(this._getPrimaryKeyQuery()), function(m, ds) {
242             m._hook("pre", "update");
243             saveChanged && ds.update(m.__changed);
244             m._hook("post", "update");
245             return m;
246           });
247         }));
248       } else if (this.__isNew && this.__isChanged) {
249         return this.save(vals, options);
250       }
251     },
252 
253     /**
254      * Save a model with new values.
255      *
256      * @example
257      *
258      * someModel.save({
259      *      myVal1 : "newValue1",
260      *      myVal2 : "newValue2",
261      *      myVal3 : "newValue3"
262      *      }).then(..do something);
263      *
264      * //or
265      *
266      * someModel.myVal1 = "newValue1";
267      * someModel.myVal2 = "newValue2";
268      * someModel.myVal3 = "newValue3";
269      *
270      * someModel.save().then(..so something);
271      *
272      * @param {Object} [options] values to save this model with
273      * @param {Function} [errback] function to call if the save fails, the promise will errback also if it fails.
274      *
275      * @return {comb.Promise} called on completion or error of save.
276      */
277     save:function(vals, options) {
278       if (this.__isNew) {
279         return this._checkTransaction(options, comb.hitch(this, function() {
280           comb.isHash(vals) && this.__set(vals);
281           var pk = this._static.primaryKey[0];
282           return comb.executeInOrder(this, this.dataset, function(m, ds) {
283             m._hook("pre", "save");
284             var id = ds.insert(m._toObject());
285             m.__ignore = true;
286             m[pk] = id;
287             m.__ignore = false;
288             m.__isNew = false;
289             m.__isChanged = false;
290             m._hook("post", "save");
291             return m;
292           });
293         }));
294       } else {
295         return this.update(vals, options);
296       }
297     }

305     /**
306      * Filter a model to return a subset of results. {@link SQL#find}
307      *
308      * <p><b>This function requires all, forEach, one, last,
309      *       or count to be called inorder for the results to be fetched</b></p>
310      * @param {Object} [options] query to filter the dataset by.
311      * @param {Boolean} [hydrate=true] if true model instances will be the result of the query,
312      *                                  otherwise just the results will be returned.
313      *
314      *@return {Dataset} A dataset to query, and or fetch results.
315      */
316     filter:function(options, hydrate) {
317       var ds = this.dataset;
318       return ds.filter.apply(ds, arguments);
319     },
320 
321     /**
322      * Retrieves a record by the primarykey of a table.
323      * @param {*} id the primary key record to find.
324      *
325      * @return {comb.Promise} called back with the record or null if one is not found.
326      */
327     findById:function(id) {
328       var pk = this.primaryKey;
329       pk = pk.length == 1 ? pk[0] : pk;
330       var q = {};
331       if (comb.isArray(pk)) {
332         pk.forEach(function(k, i) {
333           q[k] = id[i]
334         });
335       } else {
336         q[pk] = id;
337       }
338       return this.filter(q).one();
339     },
340 
341 
342     find:function(id) {
343       var args = comb.argsToArray(arguments);
344       var query = args.every(function(h) {
345         return comb.isHash(h)
346       });
347       if (query) {
348         return this.filter.apply(this, args).one();
349       } else {
350         return this.findById.apply(this, args).one();
351       }
352     },
353 
354     /**
355      * Update multiple rows with a set of values.
356      *
357      * @param {Object} vals the values to set on each row.
358      * @param {Object} [options] query to limit the rows that are updated
359      * @param {Function} [callback] function to call after the update is complete.
360      * @param {Function} [errback] function to call if the update errors.
361      *
362      * @return {comb.Promise|Dataset} if just values were passed in then a dataset is returned and exec has to be
363      *                                  called in order to complete the update.
364      *                                If options, callback, or errback are provided then the update is executed
365      *                                and a promise is returned that will be called back when the update completes.
366      */
367     update:function(vals, /*?object*/options) {
368       return this._checkTransaction(options, comb.hitch(this, function() {
369         var args = comb.argsToArray(arguments);
370         var dataset = this.dataset;
371         if (args.length > 1) {
372           dataset = dataset.filter(options);
373         }
374         return dataset.update(vals);
375       }));
376     },

378     /**
379      * Remove rows from the Model.
380      *
381      * @param {Object} [q] query to filter the rows to remove
382      * @param {Function} [errback] function to call if the removal fails.
383      *
384      * @return {comb.Promise} called back when the removal completes.
385      */
386     remove:function(q, options) {
387       //first find all records so we call alert all associations and all other crap that needs to be
388       //done in middle ware
389       return this._checkTransaction(options, comb.hitch(this, function() {
390         var p = new Promise();
391         var ds = this.dataset;
392         this.dataset.filter(ds, arguments).all(function(items) {
393           //todo this sucks find a better way!
394           var pl = items.map(function(r) {
395             return r.remove();
396           });
397           new PromiseList(pl).then(hitch(p, "callback"), hitch(p, "errback"));
398         }, hitch(p, "errback"));
399         return p;
400       }));
401     },
402 
403     removeById:function(q, options) {
404       //first find all records so we call alert all associations and all other crap that needs to be
405       //done in middle ware
406       return this._checkTransaction(options, comb.hitch(this, function() {
407         var p = new Promise();
408         this.findById(1).then(function(model) {
409           if (model) {
410             model.remove().then(hitch(p, "callback"), hitch(p, "errback"));
411           } else {
412             p.callback(0);
413           }
414         }, hitch(p, "errback"));
415         return p;
416       }));
417     },

419     /**
420      * Save either a new model or list of models to the database.
421      *
422      * @example
423      *
424      * //Save a group of records
425      * MyModel.save([m1,m2, m3]);
426      *
427      * Save a single record
428      * MyModel.save(m1);
429      *
430      * @param {Array|Object} record the record/s to save to the database
431      * @param {Function} [errback] function to execute if the save fails
432      *
433      * @return {comb.Promise} called back with the saved record/s.
434      */
435     save:function(items, options) {
436       return this._checkTransaction(options, hitch(this, function() {
437         var ps;
438         if (comb.isArray(items)) {
439           ps = items.map(function(o) {
440             return this.save(o);
441           }, this);
442           var ret = new comb.Promise();
443           return new PromiseList(ps, true).then(hitch(ret, "callback"), hitch(ret, "errback"));
444         } else {
445           var ret = new comb.Promise();
446           try {
447             ret = new this(items).save();
448           } catch (e) {
449             ret.errback(e);
450           }
451           return ret;
452         }
453       }));
454     }
