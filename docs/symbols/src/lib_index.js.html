 19 var Moose = comb.singleton(Time, {

 25         /**
 26          * A singleton class that acts as the entry point for all actions performed in moose.
 27          *
 28          * @example
 29          *
 30          * var moose = require("moose");
 31          *
 32          * moose.createConnection(....);
 33          *
 34          * moose.camelize = true;
 35          * moose.quoteIdentifiers=false;
 36          *
 37          * moose.createModel("my_table");
 38          *
 39          *
 40          *  //CHANGING IDENTIFIER INPUT METHOD
 41          *
 42          *
 43          *  //use whatever is passed in
 44          *   moose.identifierInputMethod = null;
 45          *  //convert to uppercase
 46          *  moose.identifierInputMethod = "toUpperCase";
 47          *  //convert to camelCase
 48          *  moose.identifierInputMethod = "camelize";
 49          *  //convert to underscore
 50          *  moose.identifierInputMethod = "underscore";
 51          *
 52          *
 53          *  //CHANGING IDENTIFIER OUTPUT METHOD
 54          *
 55          *  //use whatever the db returns
 56          *   moose.identifierOutputMethod = null;
 57          *  //convert to uppercase
 58          *  moose.identifierOutputMethod = "toUpperCase";
 59          *  //convert to camelCase
 60          *  moose.identifierOutputMethod = "camelize";
 61          *  //convert to underscore
 62          *  moose.identifierOutputMethod = "underscore";
 63          *
 64          *  //TURN QUOTING OFF
 65          *   moose.quoteIdentifiers = false
 66          *
 67          * @constructs
 68          * @augments moose.Time
 69          * @param options
 70          *
 71          * @property {moose.Database} defaultDatabase the default database to use, this property can only be used after the conneciton has
 72          *                             initialized.
 73          * @property {Object[]} adapters that are available to use.
 74          *
 75          *
 76          * @property {moose.Database[]} DATABASES An array of databases that are currently connected.
 77          *
 78          *
 79          * @property {String} identifierInputMethod  Set the method to call on identifiers going into the database.  This affects
 80          *  how identifiers are sent to the database. So if you use camelCased and the db identifiers are all underscored
 81          *  use camelize. The method can include
 82          * <ul>
 83          *     <li>toUpperCase</li>
 84          *     <li>toLowerCase</li>
 85          *     <li>camelize</li>
 86          *     <li>underscore</li>
 87          *     <li>Other String instance method names.</li>
 88          * </ul>
 89          *
 90          * moose uses toUpperCase identifiers in all SQL strings for most databases.
 91          *
 92          *
 93          * @property {String} identifierOutputMethod Set the method to call on identifiers coming out of the database.  This affects
 94          * the how identifiers are represented by calling the method on them.
 95          * The method can include
 96          * <ul>
 97          *     <li>toUpperCase</li>
 98          *     <li>toLowerCase</li>
 99          *     <li>camelize</li>
100          *     <li>underscore</li>
101          *     <li>Other String instance method names.</li>
102          * </ul>
103          * most database implementations in moose use toLowerCase
104          *
105          * @property {Boolean} quoteIdentifiers Set whether to quote identifiers for all databases by default. By default,
106          * moose quotes identifiers in all SQL strings.
107          *
108          */
109         constructor:function () {
110             this._super(arguments);
111             var constants = SQL.Constants;
112             for (var i in constants) {
113                 this[i] = constants[i];
114             }
115         },
116 
117         /**
118          * Initialize the connection information, and prepare moose to communicate with the DB.
119          * All models, and schemas, and models that are created before this method has been called will be deferred.
120          *
121          * @example
122          *
123          * //connect using an object
124          * moose.createConnection({
125          *              host : "127.0.0.1",
126          *              port : 3306,
127          *              type : "mysql",
128          *              maxConnections : 1,
129          *              minConnections : 1,
130          *              user : "test",
131          *              password : "testpass",
132          *              database : 'test'
133          * });
134          * //connect using a connection string
135          * var CONNECT_STRING = "mysql://test:testpass@localhost:3306/test?maxConnections=1&minConnections=1";
136          * moose.createConnection(CONNECT_STRING);
137          *
138          * @param {String|Object} options the options used to initialize the database connection.
139          *                        This may be a database connetion string or object.
140          * @params {Number} [options.maxConnections = 10] the number of connections to pool.
141          * @params {Number} [options.minConnections = 3] the number of connections to pool.
142          * @param {String} [options.type = "mysql"] the type of database to communicate with.
143          * @params {String} options.user the user to authenticate as.
144          * @params {String} options.password the password of the user.
145          * @params {String} options.database the name of the database to use, the database
146          *                                   specified here is the default database for all connections.
147          */
148         createConnection:function (options) {
149             return Database.connect(options);
150         },
151 
152         /**
153          *  @see moose#createConnection
154          */
155         connect:function () {
156             return this.createConnection.apply(this, arguments);
157         },
158 
159         /**
160          * This method allows one to connect to a database and immediately execute code.
161          * For connection options @see moose#createConnection
162          *
163          * @example
164          *
165          *
166          * var DB;
167          * var CONNECT_STRING = "dummyDB://test:testpass@localhost/dummySchema";
168          * var connectPromise = moose.connectAndExecute(CONNECT_STRING, function (db) {
169          *      db.dropTable("test");
170          *      db.createTable("test", function () {
171          *          this.primaryKey("id");
172          *          this.name(String);
173          *          this.age(Number);
174          *      });
175          * });
176          *
177          * connectPromise.then(function (db) {
178          *      DB = db;
179          *      //do more stuff!
180          *  });
181          *
182          * @param {String|Object} options @see moose#createConnection
183          * @param {Function} cb the function to callback once connected.
184          *
185          * @returns {comb.Promise} a promise that is resolved once the database execution has finished.
186          */
187         connectAndExecute:function (options, cb) {
188             if (!comb.isFunction(cb)) throw new MooseError("callback must be a function");
189             var db = this.createConnection.apply(this, arguments);
190             return comb.executeInOrder(db, moose, function (db, moose) {
191                 cb(db, moose);
192                 return db;
193             });
194         },
195 
196         /**
197          * Disconnects all databases in use.
198          *
199          * @return {comb.Promise} a promise that is resolved once all databases have disconnected.
200          */
201         disconnect:function () {
202             return new PromiseList(this.DATABASES.map(function (d) {
203                 return d.disconnect()
204             }), true);
205         },
