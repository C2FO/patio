117                     value = null;
118                 }
119                 var raiseOnError = clazz.raiseOnTypecastError;
120                 if (raiseOnError === true && comb.isUndefinedOrNull(value) && colSchema.allowNull === false) {
121                     throw new ModelError("null is not allowed for the " + column + " column.");
122                 }
123                 try {
124                     value = clazz.db.typecastValue(type, value);
125                 } catch (e) {
126                     if (raiseOnError === true) {
127                         throw e;
128                     }
129                 }

351 /*Adds a setter to an object*/
352 var addSetter = function(name, col, setterFunc){
353     return function(val){
354         var ignore = this.__ignore;
355         if (col.primaryKey && !ignore) {
356             throw new ModelError("Cannot set primary key of model " + this._static.tableName);
357         }
358         val = this._typeCastValue(name, val);
359         if (!this.isNew && !ignore) {
360             this.__isChanged = true;
361             this.__changed[name] = val;
362         }
363         setterFunc = this["_set" + name.charAt(0).toUpperCase() + name.substr(1)];
364         this.__values[name] = comb.isFunction(setterFunc) ? setterFunc.call(this, val) : val;
365     };
366 };

378 exports.create = function(name, modelOptions){
379     !patio && (patio = require("./index"));
380     modelOptions = modelOptions || {};
381     var modelInstance = modelOptions.instance || {},
382         modelGetters = modelInstance.getters || {},
383         modelSetters = modelInstance.setters || {},
384         modelStatic = modelOptions.static || {},
385         modelStaticGetters = modelStatic.getters || {},
386         modelStaticSetters = modelStatic.setters || {};
387     //do this so we can correctly map the columns
388     var outputMethod = (modelStatic.camelize === true || Model.camelize === true) ? "camelize" : (modelStatic.underscore === true || Model.underscore === true) ? "underscore" : modelStatic.identifierOutputMethod;
389     var inputMethod = (modelStatic.camelize === true || Model.camelize === true) ? "underscore" : (modelStatic.underscore === true || Model.underscore === true) ? "camelize" : modelStatic.identifierInputMethod;
390     //do this so we can correctly map the columns
391     var columnOutputTransformMethod = function(val){
392         return applyColumnTransformMethod(val, outputMethod);
393     };
394     var columnInputTransformMethod = function(val){
395         return applyColumnTransformMethod(val, inputMethod);
396     };
397 
398     var db, ds, tableName;
399     if (comb.isString(name)) {
400         tableName = columnOutputTransformMethod(name);
401         db = patio.defaultDatabase;
402         ds = db.from(tableName);
403     } else if (comb.isInstanceOf(name, patio.Dataset)) {
404         ds = name;
405         tableName = ds.firstSourceAlias;
406         db = ds.db;
407     }
408     if (!MODELS.contains(db)) {
409         MODELS.set(db, {});
410         var handle = comb.connect(db, "onDisconnect", function(){
411             comb.disconnect(handle);
412             MODELS.remove(db);
413         });
414     }
415     //Create the default proto
416     var proto = {
417         instance:{
418             patio:patio,
419             __dataset:null,
420             _hooks:["save", "update", "remove", "load"],
421             __hooks:{pre:{}, post:{}},
422             getters:{
423                 primaryKey:function(){
424                     return this._static.primaryKey;
425                 },
426 
427                 tableName:function(){
428                     return this._static.tableName;
429                 },
430 
431                 dataset:function(){
432                     return this.__dataset || this._static.dataset;
433                 },
434 
435                 db:function(){
436                     return this._static.db;
437                 },
438 
439                 schema:function(){
440                     return this._static.schema;
441                 },
442 
443                 columns:function(){
444                     return this._static.columns;
445                 }
446             },
447             setters:{}
448         },
449         static:{
450             __columns:null,
451 
452             __schema:null,
453 
454             __primaryKey:null,
455 
456             getters:{
457                 /**@lends patio.Model*/
458 
459                 /**
460                  * The name of the table all instances of the this {@link patio.Model} use.
461                  * @field
462                  * @type String
463                  */
464                 tableName:function(){
465                     return tableName;
466                 },
467 
468                 /**
469                  * The database all instances of this {@link patio.Model} use.
470                  * @field
471                  * @type patio.Database
472                  */
473                 db:function(){
474                     return db;
475                 },
476 
477                 /**
478                  * A dataset to use to retrieve instances of this {@link patio.Model{ from the database. The dataset
479                  * has the {@link patio.Dataset#rowCb} set to create instances of this model.
480                  * @field
481                  * @type patio.Dataset
482                  */
483                 dataset:function(){
484                     var ds = db.from(tableName);
485                     ds.rowCb = hitch(this, function(vals){
486                         var ret = new Promise();
487                         var m = new this(vals, true);
488                         //call the hooks!
489                         m._hook("post", "load").then(hitch(ret, "callback", m), hitch(ret, "errback"));
490                         return ret;
491                     });
492                     this.identifierInputMethod && (ds.identifierInputMethod = this.identifierInputMethod);
493                     this.identifierOutputMethod && (ds.identifierOutputMethod = this.identifierOutputMethod);
494                     return ds;
495                 },
496 
497                 /**
498                  * A list of columns this models table contains.
499                  * @field
500                  * @type String[]
501 
502                  */
503                 columns:function(){
504                     return this.__columns;
505                 },
506 
507                 /**
508                  * The schema of this {@link patio.Model}'s table. See {@link patio.Database#schema} for details
509                  * on the schema object.
510                  * @field
511                  * @type Object
512                  */
513                 schema:function(){
514                     return this.__schema;
515                 },
516 
517                 /**
518                  * The primaryKey column/s of this {@link patio.Model}
519                  * @field
520                  */
521                 primaryKey:function(){
522                     return this.__primaryKey.slice(0);
523                 },
524 
525                 /**
526                  * A reference to the global {@link patio}.
527                  * @field
528                  */
529                 patio:function(){
530                     return patio;
531                 }
532             },
533             setters:{}
534         }
535     };
536     var instance = proto.instance,
537         getters = instance.getters,
538         setters = instance.setters,
539         static = proto.static || {},
540         staticGetters = static.getters || {},
541         staticSetters = static.setters || {};
542     //remove these so they dont override our proto
543     delete modelInstance.getters;
544     delete modelInstance.setters;
545     delete modelStatic.getters;
546     delete modelStatic.setters;
547     //Mixin the column setter/getters
548     var ret = new comb.Promise();
549     comb.executeInOrder(ds, db,
550         function(ds, db){
551             return [ds.columns, db.schema(tableName), db.indexes(tableName)];
552         }).then(function(values){
553             var columns = values[0].map(columnOutputTransformMethod), schema = values[1], indexes = values[2];
554             var pks = [];
555             for (var i in schema) {
556                 var col = schema[i];
557                 var name = columnOutputTransformMethod(i);
558                 col.primaryKey && pks.push(name);
559                 getters[name] = addGetter(name);
560                 setters[name] = addSetter(name, col);
561             }
562             modelStatic.__columns = columns;
563             modelStatic.__schema = schema;
564             modelStatic.__primaryKey = pks;
565             modelStatic.__indexes = indexes;
566             //Define super and mixins
567             //By Default we include Query and Association plugins
568             var parents = [Model].concat(modelOptions.plugins || []);
569 
570             //START MERGE OF PASSED PROTO
571             var merge = comb.merge;
572 
573             merge(instance, modelInstance);
574             merge(getters, modelGetters);
575             merge(setters, modelSetters);
576             merge(static, modelStatic);
577             merge(staticGetters, modelStaticGetters);
578             merge(staticSetters, modelStaticSetters);
579             //END MERGE OF PASSED PROTO
580             //Create return model
581             var model = comb.define(parents, proto);
582             //mixin pre and post functions
583             ["pre", "post"].forEach(function(op){
584                 var optionsOp = modelOptions[op];
585                 if (optionsOp) {
586                     for (var i in optionsOp) {
587                         model[op](i, optionsOp[i]);
588                     }
589                 }
590             });
591             MODELS.get(db)[tableName] = model;
592             ret.callback(model);
593         }, comb.hitch(ret, "errback"));
594     return ret;
