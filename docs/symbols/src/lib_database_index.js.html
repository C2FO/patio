1 var comb = require("comb"),
  2     format = comb.string.format,
  3     hitch = comb.hitch,
  4     sql = require("../sql").sql,
  5     ConnectionPool = require("../ConnectionPool"),
  6     DatabaseError = require("../errors").DatabaseError,
  7     ConnectDB = require("./connect"),
  8     DatasetDB = require("./dataset"),
  9     DefaultsDB = require("./defaults"),
 10     LoggingDB = require("./logging"),
 11     QueryDB = require("./query"),
 12     SchemaDB = require("./schema"), moose;
 13 
 14 var DATABASES = [];
 15 var Database = comb.define([ConnectDB, DatasetDB, DefaultsDB, LoggingDB, QueryDB, SchemaDB], {
 16 
 17     instance : {
 18 
 19         constructor : function(opts) {
 20             opts = opts || {};
 21             !moose && (moose = require("../index"));
 22             this.moose = moose;
 23             this._super(arguments, [opts]);
 24             opts = comb.merge(this.connectionPoolDefaultOptions, opts);
 25             this.schemas = {};
 26             this.type = opts.type;
 27             this.defaultSchema = opts.defaultSchema || this.defaultSchemaDefault;
 28             this.preparedStatements = {};
 29             this.opts = opts;
 30             this.pool = ConnectionPool.getPool(opts, hitch(this, "createConnection"), hitch(this, "closeConnection"), hitch(this, "validate"));
 31             DATABASES.push(this);
 32         },
 33 
 34         /*
 35          *Cast the given type to a literal type
 36          *
 37          *   DB.cast_type_literal(Float) # double precision
 38          *   DB.cast_type_literal(:foo) # foo
 39          *   */
 40         castTypeLiteral : function(type) {
 41             return this.typeLiteral({type : type});
 42         },
 43 
 44 
 45         /*
 46          Proxy the literal call to the dataset.
 47          *
 48          *   DB.literal(1) # 1
 49          *   DB.literal(:a) # a
 50          *   DB.literal('a') # 'a'
 51          *   */
 52         literal : function(v) {
 53             return this.dataset.literal(v)
 54         },
 55 
 56         /*
 57          * Typecast the value to the given column_type. Calls
 58          * typecast_value_#{column_type} if the method exists,
 59          * otherwise returns the value.
 60          * This method should raise Sequel::InvalidValue if assigned value
 61          * is invalid.
 62          * */
 63         typecastValue : function(columnType, value) {
 64             if (comb.isNull(value) || comb.isUndefined(value)) return null;
 65             var meth = "typecastValue" + columnType.charAt(0).toUpperCase() + columnType.substr(1).toLowerCase();
 66             try {
 67                 if (comb.isFunction(this[meth])) {
 68                     return this[meth](value);
 69                 } else {
 70                     return value;
 71                 }
 72             } catch(e) {
 73                 throw new DatabaseError("Unable to convert " + columnType);
 74             }
 75         },
 76 
 77 
 78         // Typecast the value to true, false, or nil
 79         typecastValueBoolean : function(value) {
 80             if (value == false || 0 || "0" || value.match(/^f(alse)?$/i)) {
 81                 return false;
 82             } else {
 83                 return comb.isEmpty(value) ? null : true;
 84             }
 85         },
 86 
 87         // Typecast the value to a Date
 88         typecastValueDate : function(value) {
 89             if (comb.isDate(value)) {
 90                 return value;
 91             } else if (comb.isString(value)) {
 92                 var ret = comb.date.parse(value, this.DATE_PATTERN);
 93                 if (ret == null) {
 94                     throw new DatabaseError(format("Invalid value for date %j", [value]));
 95                 }
 96                 return ret;
 97             } else if (comb.isHash(value) && !comb.isEmpty(value)) {
 98                 return new Date(value.year, value.month, value.day);
 99             } else {
100                 throw new DatabaseError(format("Invalid value for date %j", [value]));
101             }
102         },
103 
104         // Typecast the value to a DateTime or Time depending on Sequel.datetime_class
105         typecastValueDatetime : function(value) {
106             var ret;
107             if(comb.isInstanceOf(value, sql.DateTime)){
108                 return value;
109             }else if (comb.isDate(value)) {
110                 ret = value;
111             }else if (comb.isHash(value) && !comb.isEmpty(value)) {
112                 ret = new Date(value.year, value.month, value.day, value.hour, value.minute, value.second);
113 
114             } else if (comb.isString(value)) {
115                 ret = comb.date.parse(value, this.DATE_TIME_PATTERN);
116                 if (ret == null) {
117                     throw new DatabaseError(format("Invalid value for datetime %j", [value]));
118                 }
119             } else {
120                 throw new DatabaseError(format("Invalid value for datetime %j", [value]));
121             }
122             return new sql.DateTime(ret);
123         },
124 
125         // Typecast the value to a DateTime or Time depending on Sequel.datetime_class
126         typecastValueTimestamp : function(value) {
127             var ret;
128             if(comb.isInstanceOf(value, sql.TimeStamp)){
129                 return ret;
130             }else if (comb.isDate(value)) {
131                 ret = value;
132             }else if (comb.isHash(value) && !comb.isEmpty(value)) {
133                 ret = new Date(value.year, value.month, value.day, value.hour, value.minute, value.second);
134 
135             } else if (comb.isString(value)) {
136                 ret = comb.date.parse(value, this.DATE_TIME_PATTERN);
137                 if (ret == null) {
138                     throw new DatabaseError(format("Invalid value for timestamp %j", [value]));
139                 }
140             } else {
141                 throw new DatabaseError(format("Invalid value for timestamp %j", [value]));
142             }
143             return new sql.TimeStamp(ret);
144         },
145 
146         // Typecast the value to a Time
147         typecastValueTime : function(value) {
148             var ret;
149             if(comb.isInstanceOf(value, sql.Time)){
150                 return value;
151             }else if (comb.isDate(value)) {
152                 ret = value;
153             } else if (comb.isString(value)) {
154                 ret = comb.date.parse(value, this.TIME_PATTERN);
155                 if (ret == null) {
156                     throw new DatabaseError(format("Invalid value for time %j", [value]));
157                 }
158             } else if (comb.isHash(value) && !comb.isEmpty(value)) {
159                 ret = new Date(0, 0, 0, value.hour, value.minute, value.second);
160             } else {
161                 throw new DatabaseError(format("Invalid value for time %j", [value]));
162             }
163             return new sql.Time(ret);
164         },
165 
166         // Typecast the value to a BigDecimal
167         typecastValueDecimal : function(value) {
168             if (comb.isNumber(value)) {
169                 return value * 1.0;
170             } else {
171                 throw new DatabaseError(format("Invalid value for decimal %j", [value]));
172             }
173 
174         },
175 
176         // Typecast the value to a Float
177         typecastValueFloat : function(value) {
178             var ret = parseFloat(value);
179             if (isNaN(ret)) {
180                 throw new DatabaseError(format("Invalid value for float %j", [value]));
181             }
182             return ret;
183         },
184 
185         // Typecast the value to an Integer
186         typecastValueInteger : function(value) {
187             var ret = parseInt(value, 10);
188             if (isNaN(ret)) {
189                 throw new DatabaseError(format("Invalid value for integer %j", [value]));
190             }
191             return ret;
192         },
193 
194         // Typecast the value to a String
195         typecastValueString : function(value) {
196             return "" + value;
197         }
198 
199     },
200 
201     static : {
202         DATABASES : DATABASES
203     }

205 }).as(module);
